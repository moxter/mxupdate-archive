42  /***
43   * The class is used as handler for commands to compare local update files
44   * against current update code from the data base.
45   *
46   * @author The MxUpdate Team

49  public class CompareHandler
50          extends AbstractFileHandler

52      /***
53       * Opens for each defined file a compare editor.
54       *
55       * @param _files    files to compare
56       * @see CompareEditor
57       */
58      @Override
59      protected void execute(final List<IFile> _files)
60      {
61          for (final IFile file : _files)  {
62  
63              CompareUI.openCompareEditor(new CompareEditor(file));
64          }
65      }

67      /***
68       * Compare editor between the local update file and the current update file
69       * extracted from the data base.
70       */
71      class CompareEditor
72              extends SaveableCompareEditorInput
73      {
74          /***
75           * Local file which must be compared against the current object in the
76           * data base.
77           */
78          private final IFile file;
79  
80          /***
81           *
82           * @param _file     local update file
83           * @see #file
84           */
85          CompareEditor(final IFile _file)
86          {
87              super(new CompareConfiguration(), null);
88              this.getCompareConfiguration().setLeftEditable(true);
89              this.getCompareConfiguration().setLeftLabel(Messages.getString("CompareHandler.LocaleFile")); //$NON-NLS-1$
90              this.getCompareConfiguration().setRightEditable(false);
91              this.getCompareConfiguration().setRightLabel(Messages.getString("CompareHandler.DataBase")); //$NON-NLS-1$
92              this.file = _file;
93          }
94  
95          /***
96           * Dummy method because the method is only needed that
97           * {@link SaveableCompareEditorInput} is implemented.
98           */
99          @Override
100         protected void fireInputChange()
101         {
102         }
103 
104         /***
105          * Fetches for given local update file the related extracted code of
106          * the update file from the data base and returns an input editor used
107          * to compare.
108          *
109          * @param _progressMonitor  progress monitor
110          * @return input editor with the local update file and the current
111          *         update code from the data base
112          * @throws InvocationTargetException of the update file from the data
113          *                                   base could not be fetched or the
114          *                                   code of the update file could not
115          *                                   be converted to bytes
116          * @todo get bytes in UTF 8 must NOT be hard coded!
117          */
118         @Override
119         protected ICompareInput prepareCompareInput(final IProgressMonitor _progressMonitor)
120                 throws InvocationTargetException
121         {
122             _progressMonitor.beginTask(Messages.getString("CompareHandler.TaskReadFromDataBase"), 2); //$NON-NLS-1$
123 
124             // get current update code
125             final String cur;
126             try {
127                 cur = Activator.getDefault().getAdapter().extractCode(this.file);
128             } catch (final Exception ex) {
129                 throw new InvocationTargetException(ex);
130             }
131 
132             // convert
133             _progressMonitor.setTaskName(Messages.getString("CompareHandler.TaskConvertText")); //$NON-NLS-1$
134             final byte[] buf;
135             try {
136                 buf = cur.getBytes("UTF8"); //$NON-NLS-1$
137             } catch (final UnsupportedEncodingException ex) {
138                 throw new InvocationTargetException(ex);
139             }
140 
141             // and create new difference node
142             return new DiffNode(SaveableCompareEditorInput.createFileElement(this.file), new ByteBufferType(buf))  {
143                 @Override
144                 public String getName()
145                 {
146                     return Messages.getString("CompareHandler.Title", //$NON-NLS-1$
147                                               CompareHandler.CompareEditor.this.file.getName());
148                 }
149             };
150         }
151     }

169          *
170          * @param _buf  new buffer
171          */
172         ByteBufferType(final byte[] _buf)
173         {
174             this.buf = _buf;
175         }
176

181          * @return always <code>null</code>

183         public Image getImage()

185             return null;
186         }

188         /***

194         public String getName()
195         {
