21  package org.mxupdate.eclipse.util;

23  import java.io.ByteArrayInputStream;
24  import java.io.ByteArrayOutputStream;
25  import java.io.IOException;
26  import java.io.ObjectInputStream;
27  import java.io.ObjectOutputStream;
28  
29  import org.apache.commons.codec.binary.Base64;
30  
31  /***
32   * Adapter to the MX database.
33   *
34   * @author The MxUpdate Team
35   * @version $Id: CommunicationUtil.java 764 2010-03-27 13:22:36Z tim.moxter $
36   */
37  public final class CommunicationUtil
38  {
39      /***
40       * Private constructor so that this utility could not be initialized.
41       */
42      private CommunicationUtil()
43      {
44      }
45  
46      /***
47       * Encodes given <code>_object</code> to a string with <b>base64</b>.
48       *
49       * @param _object   object to encode
50       * @return encoded string
51       * @throws IOException if encode failed
52       */
53      public static String encode(final Object _object)
54          throws IOException
55      {
56          final ByteArrayOutputStream out = new ByteArrayOutputStream();
57          final ObjectOutputStream oos = new ObjectOutputStream(out);
58          oos.writeObject(_object);
59          oos.close();
60          return new String(Base64.encodeBase64(out.toByteArray()));
61      }
62  
63      /***
64       * Decodes given string value to an object of given type
65       * <code>&lt;T&gt;</code>. First the string is <b>base64</b> decoded, then
66       * the object instance is extracted from the decoded bytes via the Java
67       * &quot;standard&quot; feature of the {@link ObjectInputStream}.
68       *
69       * @param <T>   type of the object which must be decoded
70       * @param _arg  string argument with encoded instance of
71       *              <code>&lt;T&gt;</code>
72       * @return decoded object instance of given type <code>&lt;T&gt;</code>
73       * @throws IOException              if the value could not be decoded,
74       *                                  the decoder stream could not be
75       *                                  opened or the argument at given
76       *                                  <code>_index</code> is not defined
77       * @throws ClassNotFoundException   if the object itself could not be read
78       *                                  from decoder stream
79       */
80      @SuppressWarnings("unchecked")
81      public static <T> T decode(final String _arg)
82          throws IOException, ClassNotFoundException
83      {
84          final byte[] bytes = Base64.decodeBase64(_arg.getBytes());
85          final ByteArrayInputStream in = new ByteArrayInputStream(bytes);
86          final ObjectInputStream ois = new ObjectInputStream(in);
87          final T ret = (T) ois.readObject();
88          ois.close();
89          return ret;
90      }
