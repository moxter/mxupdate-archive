34  /***
35   * The JPO class is the plug-in to export configuration items (administration
36   * objects) or in other words with the class the TCL update file could be
37   * created depending on the configuration item type (administration type).
38   *
39   * @author The MxUpdate Team
40   * @version $Id: Export_mxJPO.java 609 2009-11-01 22:42:53Z tim.moxter $
41   */
42  public class Export_mxJPO
43      extends AbstractPlugin_mxJPO
44  {
45      /***
46       * Name of the key in the returned mapped for the &quot;name&quot; of the
47       * administration object (which administration object is exported).
48       */
49      private static final String RETURN_KEY_NAME = "name";
50  
51      /***
52       * Name of the key in the returned mapped for the &quot;code&quot; of the
53       * administration object (the TCL update code itself).
54       */
55      private static final String RETURN_KEY_CODE = "code";
56  
57      /***
58       * Name of the key in the returned mapped for the &quot;path&quot; of the
59       * administration object (where is the TCL update code located in the file
60       * system).
61       */
62      private static final String RETURN_KEY_PATH = "path";
63  
64      /***
65       * Name of the key in the returned mapped for the &quot;file name&quot; of
66       * the administration object (what is the name of the file).
67       */
68      private static final String RETURN_KEY_FILENAME = "filename";
69  
70      /***
71       * <p>All given administration objects are exported and the related update
72       * file is returned. The administration objects to export in
73       * <code>_args</code> are identified by the administration type (see
74       * {@link TypeDef_mxJPO}) and the MX names of the administration objects.
75       * </p>
76       * <p>The returned packed values (packed with method
77       * {@link AbstractPlugin_mxJPO#prepareReturn(String, String, Exception, Object)})
78       * includes the logging messages itself and for configuration item type
79       * a collection of exporting information as map. The exporting information
80       * map defines three keys:
81       * <ul>
82       * <li>{@link #RETURN_KEY_NAME}:        name of the configuration item</li>
83       * <li>{@link #RETURN_KEY_PATH}:        path within the file directory of
84       *                                      the configuration item (without
85       *                                      file name)</li>
86       * <li>{@link #RETURN_KEY_CODE}:        TCL update source code of the
87       *                                      configuration item</li>
88       * <li>{@link #RETURN_KEY_FILENAME}:    file name of the configuration
89       *                                      item</li>
90       * </ul></p>
91       * <p>If an exception is thrown the exception is packed in the returned
92       * map.</p>
93       *
94       * @param _context  MX context for this request
95       * @param _args     encoded arguments from the Eclipse plug-in:
96       *                  <ul>
97       *                  <li><b>{@link Map}&lt;{@link String},{@link Collection}
98       *                         &lt;{@link String}&gt;&gt;</b>
99       *                      <br/>a map depending on the (administration) type
100      *                      (as key) and the related list of (administration)
101      *                      MX names</li>
102      *                  <li><b>{@link Map}&lt;{@link String},{@link String}&gt;
103      *                      </b><br/>Sets the predefined parameter values (key
104      *                      of the map is name of the parameter, value of the
105      *                      map is the value of the parameter). Default is no
106      *                      predefined parameter valued (<code>null</code>).
107      *                      </li>
108      *                  </ul>
109      * @return packed return values in maps
110      * @see AbstractPlugin_mxJPO#prepareReturn(String, String, Exception, Object)
111      */
112     public Map<String,?> exportByName(final Context _context,
113                                       final String... _args)
114     {
115         Map<String,?> packedRet = null;
116         try  {
117             // initialize mapping
118             final Map<String,Collection<String>> exports = this.<Map<String,Collection<String>>>decode(_args, 0);
119             final Map<String,String> paramValues = this.<Map<String,String>>decode(_args, 1, null);
120             final ParameterCache_mxJPO paramCache = new ParameterCache_mxJPO(_context, true, paramValues);
121 
122             // export all objects depending on the type definitions
123             final Map<String,Collection<Map<String,String>>> ret = new HashMap<String,Collection<Map<String,String>>>();
124             for (final Map.Entry<String,Collection<String>> entry : exports.entrySet())  {
125                 final TypeDef_mxJPO typeDef = paramCache.getMapping().getTypeDef(entry.getKey());
126                 final Collection<Map<String,String>> extracts = new HashSet<Map<String,String>>();
127                 ret.put(entry.getKey(), extracts);
128                 for (final String mxName : entry.getValue())  {
129                     final AbstractObject_mxJPO instance = typeDef.newTypeInstance(mxName);
130                     final StringBuilder code = new StringBuilder();
131                     instance.export(paramCache, code);
132                     final Map<String,String> desc = new HashMap<String,String>();
133                     desc.put(Export_mxJPO.RETURN_KEY_NAME, instance.getName());
134                     desc.put(Export_mxJPO.RETURN_KEY_CODE, code.toString());
135                     desc.put(Export_mxJPO.RETURN_KEY_PATH, instance.getPath());
136                     desc.put(Export_mxJPO.RETURN_KEY_FILENAME, instance.getFileName());
137                     extracts.add(desc);
138                 }
139             }
140 
141             packedRet = this.prepareReturn(paramCache.getLogString(),
142                                            (String) null,
143                                            (Exception) null,
144                                            ret);
145         } catch (final Exception e)  {
146             packedRet = this.prepareReturn((String) null,
147                                            (String) null,
148                                            e,
149                                            null);
150         }
151         return packedRet;
152     }
