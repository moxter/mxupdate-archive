71       * Updates MX Update files depending on file names. The method is used from
72       * the MX Update Eclipse Plug-In if the MxUpdate Update Deployment tool has
73       * access to the files from the Eclipse Plug-In.
74       *
75       * @param _context  MX context for this request
76       * @param _args     encoded arguments from the Eclipse plug-in:
77       *                  <ul>
78       *                  <li><b>{@link Set}&lt;{@link String}&gt;</b><br/>set of
79       *                      file names which must be updated</li>
80       *                  <li><b>{@link Boolean}</b><br/> if set to <i>true</i>
81       *                      all included JPOs are compiled; otherwise no JPOs
82       *                      are compiled. Default is not to compile
83       *                      (<i>false</i>)</li>
84       *                  <li><b>{@link Map}&lt;{@link String},{@link String}&gt;
85       *                      </b><br/>Sets the predefined parameter values (key
86       *                      of the map is name of the parameter, value of the
87       *                      map is the value of the parameter). Default is no
88       *                      predefined parameter valued (<code>null</code>).
89       *                      </li>
90       *                  </ul>
91       * @return logging information from the update
92       * @throws Exception if the update failed
93       * @see #updateFiles(Context, Map, Set, boolean)
94       */
95      public String updateByName(final Context _context,
96                                 final String[] _args)
97              throws Exception
98      {
99          final Set<String> fileNames = this.<Set<String>>decode(_args, 0);
100         final boolean compile = this.<Boolean>decode(_args, 1, false);
101         final Map<String,String> paramValues = this.<Map<String,String>>decode(_args, 2, null);
102 
103         final Set<File> files = new HashSet<File>();
104         for (final String fileName : fileNames)  {
105             files.add(new File(fileName));
106         }
107 
108         return this.updateFiles(_context, paramValues, files, compile);

111     /***
112      * Updates the files depending on the file name with related file content.
113      * The method is used from the MX Update Eclipse Plug-In if the MxUpdate
114      * Update Deployment tool has <b>NO</b> access to the files from the
115      * Eclipse Plug-In (e.g. if a virtual machine is used, or MX is located on
116      * a server).
117      *
118      * @param _context  MX context for this request
119      * @param _args     encoded arguments from the Eclipse plug-in:
120      *                  <ul>
121      *                  <li><b>{@link Map}&lt;{@link String},{@link String}&gt;
122      *                      </b><br/>map of the file name and the related file
123      *                      content which must be updated</li>
124      *                  <li><b>{@link Boolean}</b><br/> if set to <i>true</i>
125      *                      all included JPOs are compiled; otherwise no JPOs
126      *                      are compiled. Default is not to compile
127      *                      (<i>false</i>)</li>
128      *                  <li><b>{@link Map}&lt;{@link String},{@link String}&gt;
129      *                      </b><br/>Sets the predefined parameter values (key
130      *                      of the map is name of the parameter, value of the
131      *                      map is the value of the parameter). Default is no
132      *                      predefined parameter valued (<code>null</code>).
133      *                      </li>
134      *                  </ul>
135      * @return logging information from the update
136      * @throws Exception if the update failed
137      * @see #updateFiles(Context, Map, Set, boolean)
138      */
139     public String updateByContent(final Context _context,
140                                   final String[] _args)
141             throws Exception
142     {
143         String ret = "";
144 
145         final Map<String,String> files = this.<Map<String,String>>decode(_args, 0);
146         final boolean compile = this.<Boolean>decode(_args, 1, false);
147         final Map<String,String> paramValues = this.<Map<String,String>>decode(_args, 2, null);
148 
149         // create temporary directory
150         final File tmpDir = File.createTempFile("Update", "tmp");
151         tmpDir.delete();
152         tmpDir.mkdir();
153 
154         final Set<File> localFiles = new HashSet<File>();
155         try  {
156             // store all files in temporary directory
157             for (final Map.Entry<String,String> fileEntry : files.entrySet())  {
158                 final String fileName = (new File(fileEntry.getKey())).getName();
159 
160                 final File localFile = new File(tmpDir, fileName);
161                 final Writer writer = new FileWriter(localFile);
162                 writer.write(fileEntry.getValue());
163                 writer.close();
164 
165                 localFiles.add(localFile);
166             }
167 
168             // and call update
169             ret = this.updateFiles(_context, paramValues, localFiles, compile);
170         } finally  {
171             // at least remove all temporary stuff
172             for (final File localFile : localFiles)  {
173                 localFile.delete();
174             }
175             tmpDir.delete();
176         }
177 
178         return ret;
179     }

181     /***
182      * Updates all configurations items specified through given set of MX
183      * update files.
184      *
185      * @param _context      MX context for this request
186      * @param _paramValues  predefined parameters
187      * @param _files        set of all files to update
188      * @param _compile      if <i>true</i> related JPOs are compiled; if
189      *                      <i>false</i> no JPOs are compiled
190      * @return logging information from the update
191      * @throws Exception if update for the <code>_files</code> failed
192      */
193     protected String updateFiles(final Context _context,
194                                  final Map<String,String> _paramValues,
195                                  final Set<File> _files,
196                                  final boolean _compile)
197             throws Exception
198     {
199         // initialize mapping
200         final ParameterCache_mxJPO paramCache = new ParameterCache_mxJPO(_context, true, _paramValues);
201 
202         // first found related type definition instances
203         final Map<File,AbstractObject_mxJPO> instances = new TreeMap<File,AbstractObject_mxJPO>();
204         for (final File file : _files)  {
205             boolean found = false;
206             for (final TypeDef_mxJPO typeDef : paramCache.getMapping().getAllTypeDefs())  {
207                 if (!typeDef.isFileMatchLast())  {
208                     final AbstractObject_mxJPO instance = typeDef.newTypeInstance(null);
209                     final String mxName = instance.extractMxName(paramCache, file);
210                     if (mxName != null)  {
211                         instances.put(file, typeDef.newTypeInstance(mxName));
212                         found  = true;
213                         break;
214                     }
215                 }
216             }
217             if (!found)  {
218                 for (final TypeDef_mxJPO typeDef : paramCache.getMapping().getAllTypeDefs())  {
219                     if (typeDef.isFileMatchLast())  {
220                         final AbstractObject_mxJPO instance = typeDef.newTypeInstance(null);
221                         final String mxName = instance.extractMxName(paramCache, file);
222                         if (mxName != null)  {
223                             instances.put(file, typeDef.newTypeInstance(mxName));
224                             found = true;
225                             break;
226                         }
227                     }
228                 }
229             }
230         }
231 
232         // check if objects must be created
233         final Map<TypeDef_mxJPO,Set<String>> allMxNames = new HashMap<TypeDef_mxJPO,Set<String>>();
234         for (final AbstractObject_mxJPO instance : instances.values())  {
235             if (!allMxNames.containsKey(instance.getTypeDef()))  {
236                 allMxNames.put(instance.getTypeDef(), instance.getMxNames(paramCache));
237             }
238             if (!allMxNames.get(instance.getTypeDef()).contains(instance.getName()))  {
239                 paramCache.logInfo("created " + instance.getTypeDef().getLogging() + " '" + instance.getName() + "'");
240                 instance.create(paramCache);
241             }
242         }
243 
244         // update
245         for (final Map.Entry<File,AbstractObject_mxJPO> instanceEntry : instances.entrySet())  {
246             paramCache.logInfo("updated " + instanceEntry.getValue().getTypeDef().getLogging()
247                     + " '" + instanceEntry.getValue().getName() + "'");
248             instanceEntry.getValue().update(paramCache, instanceEntry.getKey(), "");
249         }
250 
251         // at least compile (to be sure that all JPOs are updated)
252         if (_compile)  {
253             for (final Map.Entry<File,AbstractObject_mxJPO> instanceEntry : instances.entrySet())  {
254                 try  {
255                     if (instanceEntry.getValue().compile(paramCache))  {
256                         paramCache.logInfo("compiled " + instanceEntry.getValue().getTypeDef().getLogging()
257                                 + " '" + instanceEntry.getValue().getName() + "'");
258                     }
259                 } catch (final Exception e)  {
260                     paramCache.logInfo("compile of " + instanceEntry.getValue().getTypeDef().getLogging()
261                             + " '" + instanceEntry.getValue().getName() + "' failed:\n" + e.toString());
262                 }
263             }
264         }
265 
266         return paramCache.getLogString();
267     }
