75      /***
76       * Converts given string by escaping all special characters for TCL.
77       *
78       * @param _text     character stream to convert
79       * @return converted string
80       */
81      public static String convertTcl(final CharSequence _text)
82      {
83          return (_text != null)
84                 ? _text.toString().replaceAll("////", "////////")
85                                   .replaceAll("//\"", "////\"")
86                                   .replaceAll("//" + "$", "//////" + "$")
87                                   .replaceAll("//{", "////{")
88                                   .replaceAll("//}", "////}")
89                                   .replaceAll("//[", "////[")
90                                   .replaceAll("//]", "////]")
91                 : "";
92      }

109     /***
110      * Converts given MX name to a name which could be used within a file
111      * system. This must be done because some characters could not handled
112      * correctly from the file system.
113      * <ul>
114      * <li>a backslash is converted to &quot;@5C&quot;</li>
115      * <li>a slash is converted to &quot;@2F&quot;</li>
116      * <li>% is converted to &quot;@25&quot;</li>
117      * <li>@ is converted to &quot;@@&quot;</li>
118      * </ul>
119      *
120      * @param _name     MX name to convert
121      * @return converted file name
122      */
123     public static String convertToFileName(final String _name)
124     {
125         return _name.replaceAll("@", "@@")
126                     .replaceAll("%", "@25")
127                     .replaceAll("/", "@2F")
128                     .replaceAll("////", "@5C");
129     }

131     /***
132      * Converts the given file name back to internal used names. This must be
133      * done because some characters could not handled correctly from the file
134      * system.
135      * <ul>
136      * <li>&quot;@5C&quot; is converted to a backslash</li>
137      * <li>&quot;@2F&quot; is converted to a slash</li>
138      * <li>&quot;@25&quot; is converted to %</li>
139      * <li>&quot;@@&quot; is converted to @</li>
140      * </ul>
141      *
142      * @param _fileName     name from file to convert
143      * @return converted name extracted from a file name
144      */
145     public static String convertFromFileName(final String _fileName)
146     {
147         return _fileName.replaceAll("@5C", "////")
148                         .replaceAll("@2F", "/")
149                         .replaceAll("@25", "%")
150                         .replaceAll("@@", "@");
151     }

153     /***
154      * A list of string is joined to one string. Between two string the given
155      * separator is set. If quotes parameter is defined each element of the
156      * list is surrounded with quotes. Each element is converted to TCL code.
157      *
158      * @param _separator    separator between two list items
159      * @param _quotes       surround the elements of the string with quotes
160      * @param _list         list of strings
161      * @param _emptyString  string which is written if the list is empty (or
162      *                      <code>null</code> if no string for empty list is
163      *                      written)
164      * @return joined string of the list items
165      * @see #convertMql(CharSequence)
166      */
167     public static String joinTcl(final char _separator,
168                                  final boolean _quotes,
169                                  final Collection<String> _list,
170                                  final String _emptyString)
171     {
172         final StringBuilder ret = new StringBuilder();
173 
174         boolean first = true;
175         if (_list.isEmpty())  {
176             if (_emptyString != null)  {
177                 ret.append(_emptyString);
178             }
179         } else  {
180             for (final String elem : _list)  {
181                 if (!first)  {
182                     ret.append(_separator);
183                 } else  {
184                     first = false;
185                 }
186                 if (_quotes)  {
187                     ret.append('\"');
188                 }
189                 ret.append(StringUtil_mxJPO.convertTcl(elem));
190                 if (_quotes)  {
191                     ret.append('\"');
192                 }
193             }
194         }
195         return ret.toString();
196     }

325     /***
326      * The method is original implemented by the project on
327      * <a href="http://commons.apache.org/io">Apache Commons IO</a> and copied
328      * from method <code>wildcardMatch</code> within class
329      * <a href="http://commons.apache.org/io/xref/org/apache/commons/io/FilenameUtils.html">
330      * org.apache.commons.io.FilenameUtils</a>.
331      *
332      * @param _filename         file name to check
333      * @param _wildcardMatcher  wildcard matcher string
334      * @return <i>true</i> if <code>_filename</code> matches
335      *         <code>_wildcardMatcher</code>
336      */
337     public static boolean match(final String _filename,
338                                 final String _wildcardMatcher)
339     {
340         final boolean ret;
341 
342         if ((_filename == null) && (_wildcardMatcher == null))  {
343             ret = true;
344         } else if (_filename == null || _wildcardMatcher == null) {
345             ret = false;
346         } else  {
347             boolean tmpRet = false;
348 
349     //                 if (caseSensitivity == null) {
350     //                     caseSensitivity = IOCase.SENSITIVE;
351     //                 }
352     //                 filename = caseSensitivity.convertCase(filename);
353     //                 wildcardMatcher = caseSensitivity.convertCase(wildcardMatcher);
354             final String[] wcs = splitOnTokens(_wildcardMatcher);
355             boolean anyChars = false;
356             int textIdx = 0;
357             int wcsIdx = 0;
358             final Stack<int[]> backtrack = new Stack<int[]>();
359 
360             // loop around a backtrack stack, to handle complex * matching
361             do {
362                 if (backtrack.size() > 0) {
363                     final int[] array = backtrack.pop();
364                     wcsIdx = array[0];
365                     textIdx = array[1];
366                     anyChars = true;
367                 }
368 
369                 // loop whilst tokens and text left to process
370                 while (wcsIdx < wcs.length) {
371 
372                     if (wcs[wcsIdx].equals("?")) {
373                         // ? so move to next text char
374                         textIdx++;
375                         anyChars = false;
376 
377                     } else if (wcs[wcsIdx].equals("*")) {
378                         // set any chars status
379                         anyChars = true;
380                         if (wcsIdx == wcs.length - 1) {
381                             textIdx = _filename.length();
382                         }
383 
384                     } else {
385                         // matching text token
386                         if (anyChars) {
387                             // any chars then try to locate text token
388                             textIdx = _filename.indexOf(wcs[wcsIdx], textIdx);
389                             if (textIdx == -1) {
390                                 // token not found
391                                 break;
392                             }
393                             final int repeat = _filename.indexOf(wcs[wcsIdx], textIdx + 1);
394                             if (repeat >= 0) {
395                                 backtrack.push(new int[] {wcsIdx, repeat});
396                             }
397                         } else {
398                             // matching from current position
399                             if (!_filename.startsWith(wcs[wcsIdx], textIdx)) {
400                                 // couldnt match token
401                                 break;
402                             }
403                         }
404 
405                         // matched text token, move text index to end of matched token
406                         textIdx += wcs[wcsIdx].length();
407                         anyChars = false;
408                     }
409 
410                     wcsIdx++;
411                 }
412 
413                 // full match
414                 if ((wcsIdx == wcs.length) && (textIdx == _filename.length()))  {
415                     tmpRet = true;
416                     break;
417                 }
418 
419             } while (backtrack.size() > 0);
420 
421             ret = tmpRet;
422         }
423         return ret;
424     }

426     /***
427      * The method is original implemented by the project on
428      * <a href="http://commons.apache.org/io">Apache Commons IO</a> and copied
429      * from method <code>splitOnTokens</code> within class
430      * <a href="http://commons.apache.org/io/xref/org/apache/commons/io/FilenameUtils.html">
431      * org.apache.commons.io.FilenameUtils</a>.
432      * Splits a string into a number of tokens.
433      *
434      * @param _text  the text to split
435      * @return the tokens, never null
436      */
437     public static String[] splitOnTokens(final String _text)
438     {
439         final String[] ret;
440 
441          if ((_text.indexOf("?") == -1) && (_text.indexOf("*") == -1))  {
442              ret = new String[] { _text };
443          } else  {
444              final char[] array = _text.toCharArray();
445              final ArrayList<String> list = new ArrayList<String>();
446              final StringBuilder buffer = new StringBuilder();
447              for (int i = 0; i < array.length; i++) {
448                  if ((array[i] == '?') || (array[i] == '*')) {
449                      if (buffer.length() != 0) {
450                          list.add(buffer.toString());
451                          buffer.setLength(0);
452                      }
453                      if (array[i] == '?') {
454                          list.add("?");
455                      } else if (list.isEmpty() || ((i > 0) && !list.get(list.size() - 1).equals("*"))) {
456                          list.add("*");
457                      }
458                  } else  {
459                      buffer.append(array[i]);
460                  }
461              }
462              if (buffer.length() != 0)  {
463                  list.add(buffer.toString());
464              }
465              ret = list.toArray(new String[list.size()]);
466          }
467          return ret;
468      }
