198     /***
199      * Creates a new instance of the parameter cache. All default values from
200      * the parameter definitions are predefined in the parameter cache.
201      *
202      * @param _context      MX context
203      * @param _stringLog    if set to <i>true</i> log via string writer,
204      *                      <i>false</i> log via <code>System.out</code>
205      * @param _paramValues  predefined parameter values to set
206      * @throws Exception if the mapping could not be initialized or a
207      *                   predefined parameter in <code>_paramValues</code> is
208      *                   not known
209      * @see #context
210      * @see #mapBoolean
211      * @see #mapList
212      * @see #mapMap
213      * @see #mapString
214      * @see #writer
215      * @see #stringWriter
216      */
217     public ParameterCache_mxJPO(final Context _context,
218                                 final boolean _stringLog,
219                                 final Map<String,String> _paramValues)
220             throws Exception
221     {
222         this.mapping = this.initMapping(_context);
223 
224         this.context = _context;
225         this.mapBoolean = new HashMap<String,Boolean>();
226         this.mapInteger = new HashMap<String,Integer>();
227         this.mapList = new HashMap<String,Collection<String>>();
228         this.mapMap = new HashMap<String,Map<String,?>>();
229         this.mapString = new HashMap<String,String>();
230 
231         // evaluate default parameter values
232         for (final ParameterDef_mxJPO paramDef : this.mapping.getAllParameterDefs())  {
233             if (paramDef.getDefaultValue() != null)  {
234                 this.defineParameter(paramDef, paramDef.getDefaultValue());
235             }
236         }
237         // evaluate predefined parameter values
238         if (_paramValues != null)  {
239             for (final Map.Entry<String,String> paramValue : _paramValues.entrySet())  {
240                 final ParameterDef_mxJPO paramDef = this.mapping.getParameterDef(paramValue.getKey());
241                 if (paramDef == null)  {
242                     throw new Exception("unknown parameter definition '" + paramValue.getKey() + "'");
243                 }
244                 this.defineParameter(paramDef, paramValue.getValue());
245             }
246         }
247 
248         if (_stringLog)  {
249             this.stringWriter = new StringWriter();
250             this.writer = new PrintWriter(this.stringWriter);
251         } else  {
252             this.stringWriter = null;
253             this.writer = new PrintWriter(System.out);
254         }
255     }

257     /***
258      * Defines the parameter depending of the parameter definition
259      * <code>_paramDef</code> with related <code>_value</code>.
260      *
261      * @param _paramDef instance of the parameter definition
262      * @param _value    string value of the parameter to define
263      */
264     protected void defineParameter(final ParameterDef_mxJPO _paramDef,
265                                    final String _value)
266     {
267         switch (_paramDef.getType())  {
268             case BOOLEAN:
269                 this.mapBoolean.put(_paramDef.getName(),
270                                     Boolean.parseBoolean(_value));
271                 break;
272             case INTEGER:
273                 this.mapInteger.put(_paramDef.getName(),
274                                     Integer.parseInt(_value));
275                 break;
276             case LIST:
277                 this.mapList.put(_paramDef.getName(),
278                                  new ArrayList<String>(Arrays.asList(_value.split(","))));
279                 break;
280             case MAP:
281                 if ((_value != null) && !"".equals(_value))  {
282                     final Map<String,String> values = new HashMap<String,String>();
283                     for (final String entry : _value.split(","))  {
284                         final String[] entryArr = entry.split("=", 2);
285                         values.put(entryArr[0], (entryArr.length > 1) ? entryArr[1] : null);
286                     }
287                     this.mapMap.put(_paramDef.getName(), values);
288                 }
289                 break;
290             case STRING:
291                 this.mapString.put(_paramDef.getName(), _value);
292                 break;
293         }
294     }

358     /***
359      * Evaluated given parameter and defines the values for given parameter
360      * depending on the list of arguments.
361      * <ul>
362      * <li><b>boolean parameter</b>: the boolean map defines for the parameter
363      *     the converted value from the default value</li>
364      * <li><b>list parameter</b>: the next argument is added to the value list
365      *     </li>
366      * <li><b>string parameter</b>: the next argument is defined as string
367      *     value</li>
368      * </ul>
369      *
370      * @param _paramDef parameter definition
371      * @param _args     list of arguments from the command line
372      * @param _index    current index within the list of arguments
373      * @return new index of the current index within the list of arguments
374      * @see #mapBoolean
375      * @see #mapList
376      * @see #mapString
377      */
378     public int evalParameter(final ParameterDef_mxJPO _paramDef,
379                              final String[] _args,
380                              final int _index)
381     {
382         int index = _index;
383 
384         if (_paramDef.getType() == ParameterDef_mxJPO.Type.BOOLEAN)  {
385             this.mapBoolean.put(_paramDef.getName(),
386                                 !Boolean.parseBoolean(_paramDef.getDefaultValue()));
387         } else if (_paramDef.getType() == ParameterDef_mxJPO.Type.INTEGER)  {
388             this.mapInteger.put(_paramDef.getName(),
389                                 Integer.parseInt(_args[++index]));
390         } else if (_paramDef.getType() == ParameterDef_mxJPO.Type.LIST)  {
391             if (!this.mapList.containsKey(_paramDef.getName()))  {
392                 this.mapList.put(_paramDef.getName(), new ArrayList<String>());
393             }
394             this.mapList.get(_paramDef.getName()).add(_args[++index]);
395         } else if (_paramDef.getType() == ParameterDef_mxJPO.Type.STRING)  {
396             this.mapString.put(_paramDef.getName(), _args[++index]);
397         }
398 
399         return index;
400     }
