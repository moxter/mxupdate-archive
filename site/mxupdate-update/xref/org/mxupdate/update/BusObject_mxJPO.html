165     /***
166      * Constructor used to initialize the type definition enumeration.
167      *
168      * @param _typeDef  defines the related type definition enumeration
169      * @param _mxName   MX name of the administration object
170      */
171     public BusObject_mxJPO(final TypeDef_mxJPO _typeDef,
172                            final String _mxName)
173     {
174         super(_typeDef,
175               (_typeDef.hasMxBusTypeDerived() && (_mxName != null) && !_mxName.contains(BusObject_mxJPO.SPLIT_TYPE))
176                       ? new StringBuilder().append(_typeDef.getMxBusType()).append(BusObject_mxJPO.SPLIT_TYPE).append(_mxName).toString()
177                       : _mxName);
178 
179         if (_mxName != null)  {
180             if (_typeDef.hasMxBusTypeDerived())  {
181                 final String[] typeNameRev = this.getName().split(BusObject_mxJPO.SPLIT_TYPE);
182                 final String[] nameRev;
183                 if (typeNameRev.length == 2)  {
184                     this.busType = typeNameRev[0];
185                     nameRev = typeNameRev[1].split(BusObject_mxJPO.SPLIT_NAME);
186                 } else  {
187                     this.busType = _typeDef.getMxBusType();
188                     nameRev = this.getName().split(BusObject_mxJPO.SPLIT_NAME);
189                 }
190                 this.busName = nameRev[0];
191                 this.busRevision = (nameRev.length > 1) ? nameRev[1] : "";
192             } else  {
193                 final String[] nameRev = this.getName().split(BusObject_mxJPO.SPLIT_NAME);
194                 this.busType = _typeDef.getMxBusType();
195                 this.busName = nameRev[0];
196                 this.busRevision = (nameRev.length > 1) ? nameRev[1] : "";
197             }
198         } else  {
199             this.busType = _typeDef.getMxBusType();
200             this.busName = null;
201             this.busRevision = null;
202         }
203     }

217     /***
218      * Searches for all business object of current type definition and returns
219      * all MX names. The revision of the business object is appended to the
220      * name of the business object split by {@link #SPLIT_NAME}. If types which
221      * are derived from original type are used, the type is the prefix of the
222      * name with the prefix {@link #SPLIT_TYPE}.
223      *
224      * @param _paramCache   parameter cache
225      * @return set of found business object names (and revision)
226      * @throws MatrixException if the query for all business objects for given
227      *                         type failed
228      */
229     @Override()
230     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
231         throws MatrixException
232     {
233         final StringList selects = new StringList(3);
234         selects.addElement("type");
235         selects.addElement("name");
236         selects.addElement("revision");
237 
238         final Query query = new Query();
239         query.open(_paramCache.getContext());
240         query.setBusinessObjectType(this.getTypeDef().getMxBusType());
241         final BusinessObjectWithSelectList list = query.select(_paramCache.getContext(), selects);
242         query.close(_paramCache.getContext());
243 
244         final Set<String> ret = new TreeSet<String>();
245         for (final Object mapObj : list)  {
246             final BusinessObjectWithSelect map = (BusinessObjectWithSelect) mapObj;
247             final String busType = (String) map.getSelectDataList("type").get(0);
248             final String busName = (String) map.getSelectDataList("name").get(0);
249             final String busRevision = (String) map.getSelectDataList("revision").get(0);
250             final StringBuilder name = new StringBuilder();
251             if (this.getTypeDef().hasMxBusTypeDerived())  {
252                 name.append(busType).append(BusObject_mxJPO.SPLIT_TYPE);
253             }
254             name.append(busName);
255             if ((busRevision != null) && !"".equals(busRevision))  {
256                 name.append(BusObject_mxJPO.SPLIT_NAME).append(busRevision);
257             }
258             ret.add(name.toString());
259         }
260         return ret;
261     }

263     /***
264      * Checks if given MX name without prefix and suffix matches given match
265      * string. The MX name is split with {@link #SPLIT_NAME} to get
266      * the name and revision of the business object. A MX name matches if the
267      * business object name or revision matches.
268      *
269      * @param _paramCache   parameter cache
270      * @param _mxName       name (and revision) of the administration business
271      *                      object
272      * @param _match        string which must be matched
273      * @return <i>true</i> if the given MX name matches; otherwise <i>false</i>
274      */
275     @Override()
276     public boolean matchMxName(final ParameterCache_mxJPO _paramCache,
277                                final String _mxName,
278                                final String _match)
279     {
280         final String[] nameRev = _mxName.split(BusObject_mxJPO.SPLIT_NAME);
281         return (StringUtil_mxJPO.match(nameRev[0], _match)
282                 || ((nameRev.length > 1) && StringUtil_mxJPO.match(nameRev[1], _match)));
283     }

285     /***
286      * Parses all information for given administration object.
287      * Sorts the attribute values, defines the description for the TCL
288      * update script (concatenation of the revision and description).
289      *
290      * @param _paramCache   parameter cache
291      * @throws MatrixException if the parse failed
292      * @see #attrValuesSorted
293      * @see #busCurrent
294      * @see #busDescription
295      * @see #busName
296      * @see #busOid
297      * @see #busRevision
298      * @see #busStates
299      * @see #busVault
300      */
301     @Override()
302     protected void parse(final ParameterCache_mxJPO _paramCache)
303         throws MatrixException
304     {
305         final BusinessObject bus = new BusinessObject(this.getBusType(),
306                                                       this.getBusName(),
307                                                       this.getBusRevision(),
308                                                       null);
309 
310         // id, vault, description and current MX state
311         final StringList select = new StringList(4);
312         select.addElement("id");
313         select.addElement("vault");
314         select.addElement("description");
315         select.addElement("current");
316         select.addElement("policy.state");
317         final BusinessObjectWithSelect sel = bus.select(_paramCache.getContext(), select);
318         this.busOid = sel.getSelectData("id");
319         this.busVault = sel.getSelectData("vault");
320         this.busDescription = sel.getSelectData("description");
321         this.busCurrent = sel.getSelectData("current");
322         this.busStates = sel.getSelectDataList("policy.state");
323 
324         // define properties (attribute values)
325         final AttributeList attrs = bus.getAttributeValues(_paramCache.getContext(), true);
326         for (final Object attrObj : attrs)  {
327             final Attribute attr = (Attribute) attrObj;
328             if (PropertyDef_mxJPO.AUTHOR.getAttrName(_paramCache).equals(attr.getName()))  {
329                 this.setAuthor(attr.getValue());
330             } else if (PropertyDef_mxJPO.APPLICATION.getAttrName(_paramCache).equals(attr.getName()))  {
331                 this.setApplication(attr.getValue());
332             } else if (PropertyDef_mxJPO.INSTALLEDDATE.getAttrName(_paramCache).equals(attr.getName()))  {
333                 this.setInstallationDate(attr.getValue());
334             } else if (PropertyDef_mxJPO.INSTALLER.getAttrName(_paramCache).equals(attr.getName()))  {
335                 this.setInstaller(attr.getValue());
336             } else if (PropertyDef_mxJPO.VERSION.getAttrName(_paramCache).equals(attr.getName()))  {
337                 this.setVersion(attr.getValue());
338             } else if (!PropertyDef_mxJPO.FILEDATE.getAttrName(_paramCache).equals(attr.getName()))  {
339                 this.attrValuesSorted.add(new AttributeValue(attr));
340             }
341         }
342 
343         // define description
344         final StringBuilder desc = new StringBuilder();
345         if ((this.busRevision != null) && !"".equals(this.busRevision))  {
346             desc.append(this.busRevision);
347             if (this.busDescription != null)  {
348                 desc.append('\n');
349             }
350         }
351         if (this.busDescription != null)  {
352             desc.append(this.busDescription);
353         }
354         this.setDescription(desc.toString());
355 
356         // evaluate from connections
357         if (this.getTypeDef().getMxBusRelsFrom() != null)  {
358             for (final String relation : this.getTypeDef().getMxBusRelsFrom())  {
359                 this.expand4Parse(_paramCache, bus, relation, true, false);
360             }
361         }
362 
363         // evaluate to connections
364         if (this.getTypeDef().getMxBusRelsTo() != null)  {
365             for (final String relation : this.getTypeDef().getMxBusRelsTo())  {
366                 this.expand4Parse(_paramCache, bus, relation, false, true);
367             }
368         }
369 
370         // evaluate both connections
371         if (this.getTypeDef().getMxBusRelsBoth() != null)  {
372             for (final String relation : this.getTypeDef().getMxBusRelsBoth())  {
373                 // from
374                 this.expand4Parse(_paramCache, bus, relation, true, false);
375                 // to
376                 this.expand4Parse(_paramCache, bus, relation, false, true);
377             }
378         }
379     }

381     /***
382      * Expands from the business object depending on the relationship and
383      * stores the connection information. If the relationship has attributes,
384      * the related values are also evaluated.
385      *
386      * @param _paramCache   parameter cache
387      * @param _bus          the business object to expand
388      * @param _relation     name of relationship to expand
389      * @param _getFrom      must be set to <i>true</i> to expand from
390      *                      connections; otherwise <i>false</i>
391      * @param _getTo        must be set to <i>true</i> to expand to
392      *                      connections; otherwise <i>false</i>
393      * @throws MatrixException if the expand failed
394      * @see #connections
395      */
396     @SuppressWarnings("unchecked")
397     private void expand4Parse(final ParameterCache_mxJPO _paramCache,
398                               final BusinessObject _bus,
399                               final String _relation,
400                               final boolean _getFrom,
401                               final boolean _getTo)
402         throws MatrixException
403     {
404         // get attributes from relationship
405         final Map<String,Map> cache = _paramCache.defineValueMap(BusObject_mxJPO.PARAM_RELATION_ATTRS, Map.class);
406         Map<String,String> attrs = cache.get(_relation);
407         if (attrs == null)
408         {
409             attrs = new HashMap<String,String>();
410             cache.put(_relation, attrs);
411 
412             final String attrStr = MqlUtil_mxJPO.execMql(_paramCache,
413                     new StringBuilder("escape print rel \"").append(_relation)
414                             .append("\" select attribute dump '\n'"));
415             if (!"".equals(attrStr))  {
416                 for (final String attr : attrStr.split("\n"))  {
417                     attrs.put("attribute[" + attr + "]", attr);
418                 }
419             };
420         }
421 
422         final StringList busSelect = new StringList(3);
423         busSelect.addElement("type");
424         busSelect.addElement("name");
425         busSelect.addElement("revision");
426         final StringList relSelect = new StringList(1 + attrs.size());
427         relSelect.addElement("id");
428         relSelect.addAll(attrs.keySet());
429         // get from objects
430         final ExpansionWithSelect expandTo = _bus.expandSelect(_paramCache.getContext(),
431                                                               _relation,
432                                                               "*",
433                                                               busSelect,
434                                                               relSelect,
435                                                               _getFrom,
436                                                               _getTo,
437                                                               (short) 1,
438                                                               null,
439                                                               null,
440                                                               (short) 0,
441                                                               true);
442 
443         for (final Object obj : expandTo.getRelationships())  {
444             this.connections.add(new Connection(_relation,
445                                                 _getFrom ? "from" : "to",
446                                                 (RelationshipWithSelect) obj,
447                                                 attrs));
448         }
449 
450     }

452     /***
453      * Writes the information to update the business objects.
454      *
455      * @param _paramCache   parameter cache
456      * @param _out          writer instance
457      * @throws IOException if the TCL update code for the business object could
458      *                     not be written
459      */
460     @Override()
461     protected void write(final ParameterCache_mxJPO _paramCache,
462                          final Appendable _out)
463         throws IOException
464     {
465         this.writeHeader(_paramCache, _out);
466         _out.append("mql escape mod bus \"${OBJECTID}\"")
467             .append(" //\n    description \"")
468             .append(StringUtil_mxJPO.convertTcl(this.busDescription))
469             .append("\"");
470         for (final AttributeValue attr : this.attrValuesSorted)  {
471             attr.write(_out);
472         }
473         // write all connected objects
474         for (final Connection con : this.connections)  {
475             con.write(_out);
476         }
477         // write promotes to target states if required
478         final int target = this.busStates.indexOf(this.busCurrent);
479         for (int idx = 0; idx < target; idx++)  {
480             _out.append("\nmql promote bus \"${OBJECTID}\"");
481         }
482     }

518     /***
519      * The method overwrites the original method to
520      * <ul>
521      * <li>reset the description</li>
522      * <li>set the version and author attribute</li>
523      * <li>reset all not ignored attributes</li>
524      * <li>define the TCL variable &quot;OBJECTID&quot; with the object id of
525      *     the represented business object</li>
526      * </ul>
527      * The original method of the super class if called surrounded with a
528      * history off, because if the update itself is done the modified basic
529      * attribute and the version attribute of the business object is updated.
530      * <br/>
531      * The new generated MQL code is set in the front of the already defined
532      * MQL code in <code>_preMQLCode</code> and appended to the MQL statements
533      * in <code>_postMQLCode</code>.
534      *
535      * @param _paramCache       parameter cache
536      * @param _preMQLCode       MQL statements which must be called before the
537      *                          TCL code is executed
538      * @param _postMQLCode      MQL statements which must be called after the
539      *                          TCL code is executed
540      * @param _preTCLCode       TCL code which is defined before the source
541      *                          file is sourced
542      * @param _tclVariables     map of all TCL variables where the key is the
543      *                          name and the value is value of the TCL variable
544      *                          (the value is automatically converted to TCL
545      *                          syntax!)
546      * @param _sourceFile       souce file with the TCL code to update
547      * @throws Exception if the update executed within derived class failed
548      */
549     @Override()
550     protected void update(final ParameterCache_mxJPO _paramCache,
551                           final CharSequence _preMQLCode,
552                           final CharSequence _postMQLCode,
553                           final CharSequence _preTCLCode,
554                           final Map<String,String> _tclVariables,
555                           final File _sourceFile)
556         throws Exception
557     {
558         // resets the description
559         final StringBuilder preMQLCode = new StringBuilder()
560                 .append("mod bus ").append(this.busOid).append(" description \"\"");
561 
562         // reset all attributes (if they must not be ignored...)
563         final Set<String> ignoreAttrs = (this.getTypeDef().getMxBusIgnoredAttributes() == null)
564                                         ? new HashSet<String>(0)
565                                         : new HashSet<String>(this.getTypeDef().getMxBusIgnoredAttributes());
566         final Map<String,String> defaultAttrValues
567                 = _paramCache.defineValueMap(BusObject_mxJPO.PARAM_DEFAULT_ATTRS, String.class);
568         for (final AttributeValue attr : this.attrValuesSorted)  {
569             if (!ignoreAttrs.contains(attr.name))  {
570                 if (!defaultAttrValues.containsKey(attr.name))  {
571                     final String def = MqlUtil_mxJPO.execMql(_paramCache,
572                                                new StringBuilder().append("print attr '")
573                                                        .append(attr.name).append("' select default dump"));
574                     defaultAttrValues.put(attr.name, def);
575                 }
576                 preMQLCode.append(" \"").append(attr.name).append("\" \"")
577                         .append(defaultAttrValues.get(attr.name)).append('\"');
578             }
579         }
580         preMQLCode.append(";\n");
581 
582         // disconnect all objects
583         for (final Connection con : this.connections)  {
584             preMQLCode.append("disconnect connection ").append(con.conId).append(";\n");
585         }
586 
587         // append demotes if required
588         final int target = this.busStates.indexOf(this.busCurrent);
589         for (int idx = 0; idx < target; idx++)  {
590             preMQLCode.append("\ndemote bus ").append(this.busOid).append(";\n");
591         }
592 
593         // append other pre MQL code
594         preMQLCode.append(_preMQLCode);
595 
596         // post update MQL statements
597         final StringBuilder postMQLCode = new StringBuilder()
598                 .append(_postMQLCode)
599                 .append("mod bus ").append(this.busOid)
600         // define version
601                 .append(" \"").append(PropertyDef_mxJPO.VERSION.getAttrName(_paramCache))
602                         .append("\" \"").append(_tclVariables.get(PropertyDef_mxJPO.VERSION.name())).append('\"')
603         // define file date
604                 .append(" \"").append(PropertyDef_mxJPO.FILEDATE.getAttrName(_paramCache))
605                         .append("\" \"").append(_tclVariables.get(PropertyDef_mxJPO.FILEDATE.name())).append('\"');
606         // is installed date property defined?
607         if ((this.getInstallationDate() == null) || "".equals(this.getInstallationDate()))  {
608             final String date = StringUtil_mxJPO.formatInstalledDate(_paramCache, new Date());
609             _paramCache.logTrace("    - define installed date '" + date + "'");
610             postMQLCode.append(" \"").append(PropertyDef_mxJPO.INSTALLEDDATE.getAttrName(_paramCache))
611                     .append("\" \"").append(date).append('\"');
612         }
613         // exists no installer property or installer property not equal?
614         final String instVal = _tclVariables.get(PropertyDef_mxJPO.INSTALLER.name());
615         if ((this.getInstaller() == null) || !this.getInstaller().equals(instVal))  {
616             _paramCache.logTrace("    - define installer '" + instVal + "'");
617             postMQLCode.append(" \"").append(PropertyDef_mxJPO.INSTALLER.getAttrName(_paramCache))
618                     .append("\" \"").append(instVal).append('\"');
619         }
620         // exists no application property or application property not equal?
621         final String applVal = _tclVariables.get(PropertyDef_mxJPO.APPLICATION.name());
622         if ((this.getApplication() == null) || !this.getApplication().equals(applVal))  {
623             _paramCache.logTrace("    - define application '" + applVal + "'");
624             postMQLCode.append(" \"").append(PropertyDef_mxJPO.APPLICATION.getAttrName(_paramCache))
625                     .append("\" \"").append(applVal).append('\"');
626         }
627         // exists no author property or author property not equal?
628         final String authVal = _tclVariables.get(PropertyDef_mxJPO.AUTHOR.name());

788          * the numbers), the numbers are compared as integer.<br/>

799          *         attribute name is lexicographically greater than the
800          *         compared attribute name
801          */
802         public int compareTo(final AttributeValue _attribute)
803         {
804             final String name1 = this.name.replaceAll(" [0-9]*$", "");
805             final String name2 = _attribute.name.replaceAll(" [0-9]*$", "");
806             int ret = 0;
807             if (name1.equals(name2) && !name1.equals(this.name))  {

809                 final Integer num1 = Integer.parseInt(this.name.substring(index).trim());
810                 final Integer num2 = Integer.parseInt(_attribute.name.substring(index).trim());
