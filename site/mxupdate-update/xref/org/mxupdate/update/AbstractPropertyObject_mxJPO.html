165     /***
166      *
167      * @param _paramCache   parameter cache
168      * @param _out          writer instance
169      * @throws IOException if the header could not be written to the TCL update
170      *                     file
171      * @see #HEADER_APPLICATION
172      * @see #HEADER_AUTHOR
173      * @see #HEADER_INSTALLER
174      * @see #HEADER_ORIGINALNAME
175      * @see #HEADER_SYMBOLIC_NAME
176      * @see #HEADER_VERSION
177      * @see #PARAM_EXPORTAPPLICATION
178      * @see #PARAM_EXPORTAUTHOR
179      * @see #PARAM_EXPORTINSTALLER
180      * @see #PARAM_EXPORTORIGINALNAME
181      * @see #PARAM_EXPORTVERSION
182      */
183     protected void writeHeader(final ParameterCache_mxJPO _paramCache,
184                                final Appendable _out)
185             throws IOException
186     {
187         final String headerText = this.getTypeDef().getTitle();
188         _out.append("################################################################################\n")
189             .append("# ").append(headerText).append(":\n")
190             .append("# ~");
191         for (int i = 0; i < headerText.length(); i++)  {
192             _out.append("~");
193         }
194         _out.append("\n")
195             .append("# ").append(this.getName()).append("\n");
196         // symbolic name only if an administration type is defined
197         if (this.getTypeDef().getMxAdminName() != null)  {
198             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_SYMBOLIC_NAME).append(' ');
199             if (this.getSymblicNames().isEmpty())  {
200                 final StringBuilder symbName = new StringBuilder()
201                         .append(this.getTypeDef().getMxAdminName())
202                         .append("_").append(this.getName().replaceAll(" ", "").replaceAll("/", ""));
203                 _paramCache.logWarning("    - No symbolic name defined for '" + this.getName()
204                         + "'! Symbolic name '" + symbName + "' will be used!");
205                 _out.append(symbName);
206             } else if (this.getSymblicNames().size() > 1)  {
207                 final String symbName = this.getSymblicNames().iterator().next();
208                 _paramCache.logError("    - Found " + this.getSymblicNames().size()
209                         + " symbolic names! Only one is allowed! '"
210                         + symbName + "' will be used!");
211                 _paramCache.logTrace("    - Following symbolic names found:");
212                 for (final String origSymb : this.getSymblicNames())  {
213                     _paramCache.logTrace("      * '" + origSymb + "'");
214                 }
215                 _out.append(symbName);
216             } else  {
217                 _out.append(this.getSymblicNames().iterator().next());
218             }
219             _out.append('\n');
220         }
221         // original name
222         // (only if an administration type and related parameter is defined)
223         if ((this.getTypeDef().getMxAdminName() != null)
224                 && _paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTORIGINALNAME))  {
225             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_ORIGINALNAME);
226             if ((this.getOriginalName() != null) && !"".equals(this.getOriginalName()))  {
227                 _out.append(" ").append(this.getOriginalName()).append('\n');
228             } else {
229                 _out.append("\n");
230             }
231         }
232         // description
233         _out.append("#\n# DESCRIPTION:\n")
234             .append("# ~~~~~~~~~~~~\n");
235         if ((this.getDescription() != null) && !"".equals(this.getDescription()))  {
236             for (final String partDesc : this.getDescription().split("\n"))  {
237                 _out.append('#');
238                 int length = 0;
239                 for (final String desc : partDesc.split(" "))  {
240                     if (!"".equals(desc))  {
241                         length += desc.length() + 1;
242                         if (length > 79)  {
243                             _out.append("\n#");
244                             length = desc.length() + 1;
245                         }
246                         _out.append(' ').append(desc);
247                     }
248                 }
249                 _out.append("\n");
250             }
251         } else  {
252             _out.append("#\n");
253         }
254         // write header
255         if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTAUTHOR))  {
256             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_AUTHOR);
257             if ((this.getAuthor() != null) && !"".equals(this.getAuthor()))  {
258                 _out.append(" ").append(this.getAuthor()).append('\n');
259             } else {
260                 _out.append("\n");
261             }
262         }
263         // write installer
264         if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTINSTALLER))  {
265             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_INSTALLER);
266             if ((this.getInstaller() != null) && !"".equals(this.getInstaller()))  {
267                 _out.append(" ").append(this.getInstaller()).append('\n');
268             } else {
269                 _out.append("\n");
270             }
271         }
272         // write application
273         if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTAPPLICATION))  {
274             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_APPLICATION);
275             if ((this.getApplication() != null) && !"".equals(this.getApplication()))  {
276                 _out.append(" ").append(this.getApplication()).append('\n');
277             } else {
278                 _out.append("\n");
279             }
280         }
281         // write version
282         if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTVERSION))  {
283             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_VERSION);
284             if ((this.getVersion() != null) && !"".equals(this.getVersion()))  {
285                 _out.append(" ").append(this.getVersion()).append('\n');
286             } else {
287                 _out.append("\n");
288             }
289         }
290         _out.append("################################################################################\n\n");
291     }

294     /***
295      * Updates this administration (business) object if the stored information
296      * about the version is not the same as the file date. If an update is
297      * required, the file is read and the object is updated with
298      * {@link #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)}.
299      *
300      * @param _paramCache       parameter cache
301      * @param _file             file with TCL update code
302      * @param _newVersion       new version which must be set within the update
303      *                          (or <code>null</code> if the version must not
304      *                          be set).
305      * @throws Exception if the update from the derived class failed
306      * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
307      * @see #extractFromCode(StringBuilder, String, String)
308      * @see #extractSymbolicNameFromCode(ParameterCache_mxJPO,StringBuilder)
309      * @see #HEADER_APPLICATION
310      * @see #HEADER_AUTHOR
311      * @see #HEADER_INSTALLER
312      * @see #HEADER_ORIGINALNAME
313      * @see #HEADER_VERSION
314      */
315     @Override
316     public void update(final ParameterCache_mxJPO _paramCache,
317                        final File _file,
318                        final String _newVersion)
319             throws Exception
320     {
321         // parse objects
322         this.parse(_paramCache);
323 
324         // read code
325         final StringBuilder code = this.getCode(_file);
326 
327         // defines the version in the TCL variables
328         final Map<String,String> tclVariables = new HashMap<String,String>();
329         if (_newVersion == null)  {
330             tclVariables.put(PropertyDef_mxJPO.VERSION.name(),
331                               this.extractFromCode(code, AbstractPropertyObject_mxJPO.HEADER_VERSION, ""));
332         } else  {
333             tclVariables.put(PropertyDef_mxJPO.VERSION.name(), _newVersion);
334         }
335 
336         // define author
337         final String author;
338         if (_paramCache.contains(ParameterCache_mxJPO.KEY_AUTHOR))  {
339             author = _paramCache.getValueString(ParameterCache_mxJPO.KEY_AUTHOR);
340         } else  {
341             author = this.extractFromCode(code,
342                                           AbstractPropertyObject_mxJPO.HEADER_AUTHOR,
343                                           _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTAUTHOR));
344         }
345         tclVariables.put(PropertyDef_mxJPO.AUTHOR.name(), author);
346 
347         // define application
348         String appl = null;
349         if (_paramCache.contains(ParameterCache_mxJPO.KEY_APPLICATION))  {
350             appl = _paramCache.getValueString(ParameterCache_mxJPO.KEY_APPLICATION);
351         }
352         if ((appl == null) || "".equals(appl))  {
353             appl = this.extractFromCode(code,
354                                         AbstractPropertyObject_mxJPO.HEADER_APPLICATION,
355                                         _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTAPPLICATION));
356         }
357         if (appl == null)  {
358             appl = "";
359         }
360         tclVariables.put(PropertyDef_mxJPO.APPLICATION.name(), appl);
361 
362         // define installer
363         final String installer;
364         if (_paramCache.contains(ParameterCache_mxJPO.KEY_INSTALLER))  {
365             installer = _paramCache.getValueString(ParameterCache_mxJPO.KEY_INSTALLER);
366         } else  {
367             installer = this.extractFromCode(code,
368                                              AbstractPropertyObject_mxJPO.HEADER_INSTALLER,
369                                              _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTINSTALLER));
370         }
371         tclVariables.put(PropertyDef_mxJPO.INSTALLER.name(), installer);
372 
373         if (this.getTypeDef().getMxAdminName() != null)  {
374             // define symbolic name
375             tclVariables.put("SYMBOLICNAME", this.extractSymbolicNameFromCode(_paramCache, code));
376 
377             // define original name
378             final String origName;
379             if ((this.getOriginalName() != null) && !"".equals(this.getOriginalName()))  {
380                 origName = this.getOriginalName();
381             } else  {
382                 origName = this.getName();
383             }
384             tclVariables.put(PropertyDef_mxJPO.ORIGINALNAME.name(), origName);
385         }
386 
387         // define file date
388         tclVariables.put(PropertyDef_mxJPO.FILEDATE.name(),
389                          StringUtil_mxJPO.formatFileDate(_paramCache, new Date(_file.lastModified())));
390 
391         this.update(_paramCache, "", "", "", tclVariables, _file);
392     }

409     /***
410      * Extracts for given header text the related value from the source code.
411      * If no value in the update file is defined, the default value from is
412      * used.
413      *
414      * @param _code             TCL update source code
415      * @param _headerText       text used to identify the value
416      * @param _default          default value if not defined within header
417      * @return extracted string from source code
418      */
419     protected String extractFromCode(final StringBuilder _code,
420                                      final String _headerText,
421                                      final String _default)
422     {
423         final int length = _headerText.length();
424         final int start = _code.indexOf(_headerText) + length;
425         final String value;
426         if ((start > length) && (_code.charAt(start) == ' '))  {
427             final int end = _code.indexOf("\n", start);
428             if (end > 0)  {
429                 final String tmp = _code.substring(start, end).trim();
430                 if ("".equals(tmp))  {
431                     value = _default;
432                 } else  {
433                     value = tmp;
434                 }
435             } else  {
436                 value = _default;
437             }
438         } else  {
439             value = _default;
440         }
441         return value;
442     }

444     /***
445      * <p>Returns the symbolic name for current property object. If parameter
446      * {@link #PARAM_CALCSYMBOLICNAMES} is defined, the symbolic name is only
447      * calculated. Otherwise the header is checked if there is a defined
448      * symbolic name (and by used if exists). A check is done if the symbolic
449      * name starts correctly with the administration type name (in lower
450      * case!). If not, the calculated symbolic name is used. A warning is shown
451      * if the symbolic name defined in the header is not equal to the
452      * calculated symbolic name.</p>
453      * <p>The symbolic name is calculated by removing all spaces and slashes
454      * within the name of the object and add as prefix the administration type
455      * name.</p>
456      *
457      * @param _paramCache   parameter cache
458      * @param _code         TCL update source code
459      * @return extracted symbolic name from the source code header
460      * @see #PARAM_CALCSYMBOLICNAMES
461      */
462     private String extractSymbolicNameFromCode(final ParameterCache_mxJPO _paramCache,
463                                                final StringBuilder _code)
464     {
465         String codeSymbName = null;
466 
467         if (this.getTypeDef().getMxAdminName() != null)  {
468             final String symbName = this.calcDefaultSymbolicName();
469             if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_CALCSYMBOLICNAMES))  {
470                 codeSymbName = symbName;
471                 _paramCache.logDebug("    - using calculated symbolic name '" + symbName + "'");
472             } else  {
473                 codeSymbName = this.extractFromCode(_code, AbstractPropertyObject_mxJPO.HEADER_SYMBOLIC_NAME, null);
474                 if (codeSymbName == null)  {
475                     _paramCache.logError("No symbolic name defined! So '" + symbName + "' will be used.");
476                     codeSymbName = symbName;
477                 } else if (!codeSymbName.startsWith(this.getTypeDef().getMxAdminName()))  {
478                     _paramCache.logError("Symbolic name does not start correctly! So '"
479                             + symbName + "' will be used (instead of '" + codeSymbName + "').");
480                     codeSymbName = symbName;
481                 } else if (!codeSymbName.equals(symbName))  {
482                     _paramCache.logWarning("Symbolic name '" + symbName
483                             + "' should be used! But defined is '" + codeSymbName + "'.");
484                 }
485             }
486         }
487 
488         return codeSymbName;
489     }

491     /***
492      * The method updates this administration (business) object. First all MQL
493      * commands are concatenated:
494      * <ul>
495      * <li>pre MQL commands (from parameter <code>_preMQLCode</code>)</li>
496      * <li>change to TCL mode</li>
497      * <li>set all required TCL variables (from parameter
498      *     <code>_tclVariables</code>)</li>
499      * <li>append TCL update code from file (from parameter
500      *     <code>_tclCode</code>)</li>
501      * <li>change back to MQL mode</li>
502      * <li>append post MQL statements (from parameter
503      *     <code>_postMQLCode</code>)</li>
504      * </ul>
505      * At the end of the MQL statements, {@link #TEST_EXECUTED} is printed via
506      * MQL command &quot;<code>output</code> so that this output is returned
507      * from the MQL execution. The return of the MQL execution is tested if
508      * this output is returned. If not an exception is thrown because the
509      * update failed.
510      *
511      * @param _paramCache       parameter cache
512      * @param _preMQLCode       MQL statements which must be called before the
513      *                          TCL code is executed
514      * @param _postMQLCode      MQL statements which must be called after the
515      *                          TCL code is executed
516      * @param _preTCLCode       TCL code which is defined before the source
517      *                          file is sourced
518      * @param _tclVariables     map of all TCL variables where the key is the
519      *                          name and the value is value of the TCL variable
520      *                          (the value is automatically converted to TCL
521      *                          syntax!)
522      * @param _sourceFile       souce file with the TCL code to update
523      * @throws Exception if update failed
524      */
525     protected void update(final ParameterCache_mxJPO _paramCache,
526                           final CharSequence _preMQLCode,
527                           final CharSequence _postMQLCode,
528                           final CharSequence _preTCLCode,
529                           final Map<String,String> _tclVariables,
530                           final File _sourceFile)
531             throws Exception
532     {
533         final StringBuilder cmd = new StringBuilder().append(_preMQLCode);
534 
535         // append TCL mode
536         cmd.append("tcl;\n")
537            .append("eval  {\n");
538 
539         // define all TCL variables
540         for (final Map.Entry<String, String> entry : _tclVariables.entrySet())  {
541             cmd.append("set ").append(entry.getKey())
542                .append(" \"").append(StringUtil_mxJPO.convertTcl(entry.getValue())).append("\"\n");
543         }
544         // append TCL code, end of TCL mode and post MQL statements
545         // (source with the file must be replace for windows ...)
546         cmd.append(_preTCLCode)
547            .append("\nsource \"").append(_sourceFile.toString().replaceAll("////", "/")).append("\"")
548            .append("\n}\nexit;\n")
549            .append(_postMQLCode)
550            .append("output '';output '").append(AbstractPropertyObject_mxJPO.TEST_EXECUTED).append("';");
551 
552         // execute update
553         JPOCaller_mxJPO.defineInstance(_paramCache, this);
554         try  {
555             final String[] ret = MqlUtil_mxJPO.execMql(_paramCache.getContext(), cmd).split("\n");
556             if (!AbstractPropertyObject_mxJPO.TEST_EXECUTED.equals(ret[ret.length - 1]))  {
557                 throw new Exception("Execution of the update was not complete! Update Failed!");
558             }
559         } finally  {
560             JPOCaller_mxJPO.undefineInstance(_paramCache);
561         }
562     }
