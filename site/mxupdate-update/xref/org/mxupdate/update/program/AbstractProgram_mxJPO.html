59      /***
60       * All basic properties are read and initialized. Basic attributes are:
61       * <ul>
62       * <li>name</li>
63       * <li>author</li>
64       * <li>application</li>
65       * <li>installation date</li>
66       * <li>installer</li>
67       * <li>original name</li>
68       * <li>version</li>
69       * </ul>
70       *
71       * @param _paramCache   parameter cache
72       * @throws MatrixException if the &quot;basic&quot; properties could not be
73       *                         read from the program object
74       */
75      @Override
76      protected void parse(final ParameterCache_mxJPO _paramCache)
77              throws MatrixException
78      {
79          // set author depending on the properties
80          this.setAuthor(this.getPropValue(_paramCache, PropertyDef_mxJPO.AUTHOR));
81          // set application depending on the properties
82          this.setApplication(this.getPropValue(_paramCache, PropertyDef_mxJPO.APPLICATION));
83          // sets the installation date depending on the properties
84          this.setInstallationDate(this.getPropValue(_paramCache, PropertyDef_mxJPO.INSTALLEDDATE));
85          // sets the installer depending on the properties
86          this.setInstaller(this.getPropValue(_paramCache, PropertyDef_mxJPO.INSTALLER));
87          // sets the original name depending on the properties
88          this.setOriginalName(this.getPropValue(_paramCache, PropertyDef_mxJPO.ORIGINALNAME));
89          // sets the version depending on the properties
90          this.setVersion(this.getPropValue(_paramCache, PropertyDef_mxJPO.VERSION));
91      }

99      @Override
100     public void delete(final ParameterCache_mxJPO _paramCache)
101             throws Exception
102     {
103         final StringBuilder cmd = new StringBuilder()
104                 .append("delete ").append(this.getTypeDef().getMxAdminName())
105                 .append(" \"").append(this.getName()).append("\" ")
106                 .append(this.getTypeDef().getMxAdminSuffix());
107         MqlUtil_mxJPO.execMql(_paramCache, cmd);

110     /***
111      * Current program is updated. The update has following steps:
112      * <ul>
113      * <li>all not &quot;standard&quot; properties are removed</li>
114      * <li>program / JPO is updated</li>
115      * <li>&quot;standard&quot; properties are set (version, file date,
116      *     installer, original name, application and author)</li>
117      * <li>registration of the symbolic name</li>
118      * </ul>
119      *
120      * @param _paramCache   parameter cache
121      * @param _updateCode   update code (which is executed in front of)
122      * @param _newVersion   new version of the program / JPO to set
123      * @param _file         file to update (to get the file date)
124      * @throws MatrixException if update of the program / JPO failed
125      */
126     protected void update(final ParameterCache_mxJPO _paramCache,
127                           final CharSequence _updateCode,
128                           final String _newVersion,
129                           final File _file)
130             throws MatrixException
131     {
132         // append statement to reset execute user
133         final StringBuilder cmd = new StringBuilder()
134                 .append("mod prog \"").append(this.getName())
135                 .append("\" execute user \"\";\n");
136 
137         // append MQL statements to reset properties
138         final String prpStr = MqlUtil_mxJPO.execMql(_paramCache,
139                                       new StringBuilder().append("print program \"").append(this.getName())
140                                            .append("\" select property.name property.to dump ' @@@@@@'"));
141         final String[] prpArr = prpStr.toString().split("(@@@@@@)");
142         final int length = (prpArr.length + 1) / 2;
143         for (int idxName = 0, idxTo = length; idxName < length; idxName++, idxTo++)  {
144             final String name = prpArr[idxName].trim();
145             final PropertyDef_mxJPO propDef = PropertyDef_mxJPO.getEnumByPropName(_paramCache, name);
146             if (propDef == null)  {
147 // TODO: if to is defined, the remove must be specified the to ....
148                 final String to = (idxTo < length) ? prpArr[idxTo].trim() : "";
149                 cmd.append("mod prog \"").append(this.getName())
150                    .append("\" remove property \"").append(name).append("\";\n");
151             }
152         }
153 
154         // append update code
155         cmd.append(_updateCode)
156            .append("mod prog \"").append(this.getName()).append('\"');
157 
158         // define version property
159         cmd.append(" add property \"").append(PropertyDef_mxJPO.VERSION.getPropName(_paramCache)).append("\" ")
160            .append("value \"").append(_newVersion != null ? _newVersion : "").append('\"');
161         // define file date property
162         cmd.append(" add property \"").append(PropertyDef_mxJPO.FILEDATE.getPropName(_paramCache)).append("\" ")
163            .append("value \"")
164            .append(StringUtil_mxJPO.formatFileDate(_paramCache, new Date(_file.lastModified()))).append('\"');
165         // is installed date property defined?
166         if ((this.getInstallationDate() == null) || "".equals(this.getInstallationDate()))  {
167             final String date = StringUtil_mxJPO.formatInstalledDate(_paramCache, new Date());
168             _paramCache.logTrace("    - define installed date '" + date + "'");
169             cmd.append(" add property \"")
170                .append(PropertyDef_mxJPO.INSTALLEDDATE.getPropName(_paramCache)).append("\" ")
171                .append("value \"").append(date).append('\"');
172         }
173         // exists no installer property or installer property not equal?
174         final String installer;
175         if (_paramCache.contains(ParameterCache_mxJPO.KEY_INSTALLER))  {
176             installer = _paramCache.getValueString(ParameterCache_mxJPO.KEY_INSTALLER);
177         } else if ((this.getInstaller() == null) || "".equals(this.getInstaller()))  {
178             installer = _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTINSTALLER);
179         } else  {
180             installer = null;
181         }
182         if (installer != null)  {
183             _paramCache.logTrace("    - define installer '" + installer + "'");
184             cmd.append(" add property \"").append(PropertyDef_mxJPO.INSTALLER.getPropName(_paramCache)).append("\" ")
185                .append("value \"").append(installer).append('\"');
186         }
187         // is original name property defined?
188         if ((this.getOriginalName() == null) && "".equals(this.getOriginalName()))  {
189             _paramCache.logTrace("    - define original name '" + this.getName() + "'");
190             cmd.append(" add property \"").append(PropertyDef_mxJPO.ORIGINALNAME.getPropName(_paramCache)).append("\" ")
191                .append("value \"").append(this.getName()).append('\"');
192         }
193         // exists no application property or application property not equal?
194         String appl = null;
195         if (_paramCache.contains(ParameterCache_mxJPO.KEY_APPLICATION))  {
196             appl = _paramCache.getValueString(ParameterCache_mxJPO.KEY_APPLICATION);
197         }
198         if ((appl == null) || "".equals(appl))  {
199             appl = _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTAPPLICATION);
200         }
201         if (appl == null)  {
202             appl = "";
203         }
204         if (!appl.equals(this.getApplication()))  {
205             _paramCache.logTrace("    - define application '" + appl + "'");
206             cmd.append(" add property \"").append(PropertyDef_mxJPO.APPLICATION.getPropName(_paramCache)).append("\" ")
207                .append("value \"").append(appl).append('\"');
208         }
209         // exists no author property or author property not equal?
210         final String author;
211         if (_paramCache.contains(ParameterCache_mxJPO.KEY_AUTHOR))  {
212             author = _paramCache.getValueString(ParameterCache_mxJPO.KEY_AUTHOR);
213         } else if ((this.getAuthor() == null) || "".equals(this.getAuthor()))  {
214             author = _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTAUTHOR);
215         } else  {
216             author = null;
217         }
218         if (author != null)  {
219             _paramCache.logTrace("    - define author '" + author + "'");
220             cmd.append(" add property \"").append(PropertyDef_mxJPO.AUTHOR.getPropName(_paramCache)).append("\" ")
221                .append("value \"").append(author).append('\"');
222         }
223         cmd.append(";\n");
224 
225         // read symbolic names
226         this.readSymbolicNames(_paramCache);
227 
228         // append registration of symbolic names
229         this.appendSymbolicNameRegistration(_paramCache,
230                                             this.calcDefaultSymbolicName(_paramCache),
231                                             cmd);
232 
233         // and execute alls
234         MqlUtil_mxJPO.execMql(_paramCache, cmd);
235     }
