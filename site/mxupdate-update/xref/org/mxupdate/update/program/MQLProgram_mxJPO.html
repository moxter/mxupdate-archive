106     /***
107      * Searches for all programs which are not JPOs and returns this list.
108      *
109      * @param _paramCache   parameter cache
110      * @return set of all program names (which are not JPOs)
111      * @throws MatrixException if the &quot;<code>list program</code>&quot;
112      *                         failed which is used to evaluate the JPO names
113      */
114     @Override()
115     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
116         throws MatrixException
117     {
118         final StringBuilder cmd = new StringBuilder()
119                 .append("list program * select name isjavaprogram dump \"\t\"");
120         final Set<String> ret = new TreeSet<String>();
121         for (final String name : MqlUtil_mxJPO.execMql(_paramCache, cmd).split("\n"))  {
122             if (!"".equals(name))  {
123                 final String[] nameArr = name.split("\t");
124                 if (!"TRUE".equals(nameArr[1]))  {
125                     ret.add(nameArr[0]);
126                 }
127             }
128         }
129         return ret;
130     }

149     /***
150      * Writes the code from the MQL program to given writer instance.
151      *
152      * @param _paramCache   parameter cache
153      * @param _out          writer instance
154      * @throws MatrixException  if the print of the code of the program failed
155      * @throws IOException      if the source code could not be written to the
156      *                          writer instance
157      */
158     @Override()
159     protected void write(final ParameterCache_mxJPO _paramCache,
160                          final Appendable _out)
161         throws IOException, MatrixException
162     {
163         final String markStartStr = _paramCache.getValueString(MQLProgram_mxJPO.PARAM_MARKSTART).trim();
164         final String markEndStr = _paramCache.getValueString(MQLProgram_mxJPO.PARAM_MARKEND).trim();
165 
166         // get line prefix
167         final String linePrefix = this.getLinePrefix(_paramCache, this.getName());
168 
169         // append to marker the line prefixes
170         final String markStart = this.makeLinePrefix(linePrefix, markStartStr);
171         final String markEnd = this.makeLinePrefix(linePrefix, markEndStr);
172 
173         this.writeUpdateCode(_paramCache, _out, markStart, markEnd, linePrefix);
174 
175         // append original code (without old TCL update code)
176         final int start = this.getCode().indexOf(markStart);
177         final int end = this.getCode().indexOf(markEnd);
178         if ((start >= 0) && (end > 0))  {
179             _out.append(this.getCode().substring(0, start).trim())
180                 .append(this.getCode().substring(end + markEnd.length()).trim());
181         } else  {
182             _out.append(this.getCode());
183         }
184     }

186     /***
187      * Depending on the extension of given <code>_fileName</code> the defined
188      * line prefix defined with parameter {@link #PARAM_EXTENSION} is returned.
189      *
190      * @param _paramCache   parameter cache
191      * @param _fileName     name of the file with the extension
192      * @return related of the extension the found extension
193      * @see #PARAM_EXTENSION
194      */
195     protected String getLinePrefix(final ParameterCache_mxJPO _paramCache,
196                                    final String _fileName)
197     {
198         final Map<String,String> extensions = _paramCache.<String>getValueMap(MQLProgram_mxJPO.PARAM_EXTENSION);
199         final int fileExtIdx = _fileName.lastIndexOf('.');
200         final String fileExtension = (fileExtIdx >= 0) ? _fileName.substring(fileExtIdx) : null;
201         return extensions.get(fileExtension);
202     }

220     /***
221      * The method overwrites the original method to append the MQL statements
222      * in the <code>_preMQLCode</code> to update this MQL program. Following
223      * steps are done within update:
224      * <ul>
225      * <li>An existing execute user is removed.</li>
226      * <li>Execution of the JPO is immediate.</li>
227      * <li>The description is set to an empty string.</li>
228      * <li>The MQL program is set to not hidden.</li>
229      * <li>The MQL program does not need the context of a business object.</li>
230      * <li>The MQL program is not downloadable.</li>
231      * <li>The input / output of the MQL program is not piped.</li>
232      * <li>The MQL program is not pooled (used only for TCL programs).</li>
233      * <li>The program is updated with the content of <code>_sourceFile</code>.
234      *     </li>
235      * <li>If the file name includes a '@' the file content is copied to a new
236      *     file and updated the new file (because of a bug in MX that a MQL
237      *     program could not be updated with @ in file names).</li>
238      * <li>Embedded TCL update code is searched depending on the file extension
239      *     and the TCL update needed parameter {@link #PARAM_NEEDED}. If for a
240      *     file extension a line prefix is defined, the line prefix is removed
241      *     from the TCL update code (and also the mark texts defined with
242      *     {@link #PARAM_MARKSTART} and {@link #PARAM_MARKEND} depends on the
243      *     line prefix).</li>
244      * </ul>
245      *
246      * @param _paramCache       parameter cache
247      * @param _preMQLCode       MQL statements which must be called before the
248      *                          TCL code is executed
249      * @param _postMQLCode      MQL statements which must be called after the
250      *                          TCL code is executed
251      * @param _preTCLCode       TCL code which is defined before the source
252      *                          file is sourced
253      * @param _tclVariables     map of all TCL variables where the key is the
254      *                          name and the value is value of the TCL variable
255      *                          (the value is automatically converted to TCL
256      *                          syntax!)
257      * @param _sourceFile       souce file with the TCL code to update
258      * @throws Exception if the update from derived class failed
259      * @see #PARAM_EXTENSION
260      * @see #PARAM_MARKEND
261      * @see #PARAM_MARKSTART
262      * @see #PARAM_NEEDED
263      */
264     @Override()
265     protected void update(final ParameterCache_mxJPO _paramCache,
266                           final CharSequence _preMQLCode,
267                           final CharSequence _postMQLCode,
268                           final CharSequence _preTCLCode,
269                           final Map<String,String> _tclVariables,
270                           final File _sourceFile)
271         throws Exception
272     {
273         // get parameters
274         final String markStartStr = _paramCache.getValueString(MQLProgram_mxJPO.PARAM_MARKSTART).trim();
275         final String markEndStr = _paramCache.getValueString(MQLProgram_mxJPO.PARAM_MARKEND).trim();
276         final boolean exec = _paramCache.getValueBoolean(MQLProgram_mxJPO.PARAM_NEEDED);
277 
278         // get line prefix and length of line prefix
279         final String linePrefix = this.getLinePrefix(_paramCache, _sourceFile.getName());
280 
281         // append to marker the line prefixes
282         final String markStart = this.makeLinePrefix(linePrefix, markStartStr);
283         final String markEnd = this.makeLinePrefix(linePrefix, markEndStr);
284 
285         //
286         File tempFile = null;
287         try  {
288             if (_sourceFile.getPath().contains("@"))  {
289                 tempFile = File.createTempFile("MXUPDATE", "TMP");
290 
291                 FileWriter out = null;
292                 try  {
293                     out = new FileWriter(tempFile);
294                     out.write(this.getCode());
295                 } finally  {
296                     if (out != null)  {
297                         out.close();
298                     }
299                 }
300             }
301 
302             final StringBuilder preMQLCode = new StringBuilder();
303             final StringBuilder preTCLCode = new StringBuilder();
304 
305             // update code; reset execute user
306             preMQLCode.append("escape mod prog \"").append(StringUtil_mxJPO.convertMql(this.getName()))
307                       .append("\" execute user \"\" execute immediate !needsbusinessobject !downloadable !pipe !pooled description \"\" !hidden file \"");
308             if (tempFile != null)  {
309                 preMQLCode.append(StringUtil_mxJPO.convertMql(tempFile.getPath()));
310             } else  {
311                 preMQLCode.append(StringUtil_mxJPO.convertMql(_sourceFile.getPath()));
312             }
313             preMQLCode.append("\";\n");
314 
315             // append TCL code of file
316             preTCLCode.append(this.extractTclUpdateCode(_paramCache,
317                                                         exec,
318                                                         this.getCode(_sourceFile),
319                                                         markStart,
320                                                         markEnd,
321                                                         linePrefix));
322 
323             // append already existing pre MQL code
324             preMQLCode.append(";\n")
325                       .append(_preMQLCode);
326 
327             // append procedure to order fields of the form
328             preTCLCode.append('\n')
329                       .append(_preTCLCode);
330 
331             // and update
332             super.update(_paramCache, preMQLCode, _postMQLCode, preTCLCode, _tclVariables, null);
333         } finally  {
334             if (tempFile != null)  {
335                 tempFile.delete();
336             }
337         }
338     }
