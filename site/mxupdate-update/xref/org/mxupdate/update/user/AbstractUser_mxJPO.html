466     /***
467      * <p>The method overwrites the original method to append the MQL
468      * statements in the <code>_preMQLCode</code> to reset this user. Following
469      * steps are done:
470      * <ul>
471      * <li>remove assigned {@link #site}</li>
472      * <li>remove all {@link #cues}</li>
473      * <li>remove all {@link #filters}</li>
474      * <li>remove all {@link #queries}</li>
475      * <li>remove all {@link #tables}</li>
476      * <li>remove all {@link #tips}</li>
477      * <li>remove all {@link #toolSets tool sets}</li>
478      * <li>remove all {@link #views}</li>
479      * </ul></p>
480      * <p>The workspace specific objects are only removed if the name of this
481      * user object does not match to the matches defined with the parameter
482      * {@link #PARAM_IGNORE_WSO_USERS}.</p>
483      *
484      * @param _paramCache       parameter cache
485      * @param _preMQLCode       MQL statements which must be called before the
486      *                          TCL code is executed
487      * @param _postMQLCode      MQL statements which must be called after the
488      *                          TCL code is executed
489      * @param _preTCLCode       TCL code which is defined before the source
490      *                          file is sourced
491      * @param _tclVariables     map of all TCL variables where the key is the
492      *                          name and the value is value of the TCL variable
493      *                          (the value is automatically converted to TCL
494      *                          syntax!)
495      * @param _sourceFile       souce file with the TCL code to update
496      * @throws Exception if the update from derived class failed
497      * @see #ignoreWorkspaceObjects(ParameterCache_mxJPO)
498      */
499     @Override()
500     protected void update(final ParameterCache_mxJPO _paramCache,
501                           final CharSequence _preMQLCode,
502                           final CharSequence _postMQLCode,
503                           final CharSequence _preTCLCode,
504                           final Map<String,String> _tclVariables,
505                           final File _sourceFile)
506         throws Exception
507     {
508         final StringBuilder preMQLCode = new StringBuilder();
509 
510         // remove hidden flag
511         if (this.isHidden())  {
512             preMQLCode.append("escape mod ").append(this.getTypeDef().getMxAdminName())
513                       .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
514                       .append(" !hidden;\n");
515         }
516 
517         // remove site...
518         if (this.site != null)  {
519             preMQLCode.append("escape mod ").append(this.getTypeDef().getMxAdminName())
520                       .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
521                       .append(" site \"\";\n");
522         }
523 
524         if (!this.ignoreWorkspaceObjects(_paramCache))  {
525             // remove all assigned cues
526             for (final Cue_mxJPO cue : this.cues.values())  {
527                 preMQLCode.append("escape delete cue \"").append(StringUtil_mxJPO.convertMql(cue.getName()))
528                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
529             }
530             // remove all assigned filters
531             for (final Filter_mxJPO filter : this.filters.values())  {
532                 preMQLCode.append("escape delete filter \"").append(StringUtil_mxJPO.convertMql(filter.getName()))
533                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
534             }
535             // remove all assigned queries
536             for (final Query_mxJPO query : this.queries.values())  {
537                 preMQLCode.append("escape delete query \"").append(StringUtil_mxJPO.convertMql(query.getName()))
538                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
539             }
540             // remove all assigned tables
541             for (final Table_mxJPO table : this.tables.values())  {
542                 preMQLCode.append("escape delete table \"").append(StringUtil_mxJPO.convertMql(table.getName()))
543                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
544             }
545             // remove all assigned tips
546             for (final Tip_mxJPO tip : this.tips.values())  {
547                 preMQLCode.append("escape delete tip \"").append(StringUtil_mxJPO.convertMql(tip.getName()))
548                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
549             }
550             // remove all assigned tool sets
551             for (final ToolSet_mxJPO toolSet : this.toolSets.values())  {
552                 preMQLCode.append("escape delete toolset \"").append(StringUtil_mxJPO.convertMql(toolSet.getName()))
553                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
554             }
555             // remove all assigned views
556             for (final View_mxJPO view : this.views.values())  {
557                 preMQLCode.append("escape delete view \"").append(StringUtil_mxJPO.convertMql(view.getName()))
558                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
559             }
560         }
561 
562         // append already existing pre MQL code
563         preMQLCode.append(_preMQLCode);
564 
565         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
566     }

568     /***
569      * Calculates if workspace objects for this user are not handled. This is
570      * done by checking if the name of this user matches one of the match lists
571      * defined with parameter {@link #PARAM_IGNORE_WSO_USERS}. In this case the
572      * the workspace objects are ignored for this user.
573      *
574      * @param _paramCache       parameter cache
575      * @return <i>true</i> if the handling of workspace objects for this user
576      *         is ignored
577      * @see #PARAM_IGNORE_WSO_USERS
578      */
579     protected boolean ignoreWorkspaceObjects(final ParameterCache_mxJPO _paramCache)
580     {
581         final Collection<String> ignoreMatches = _paramCache.getValueList(AbstractUser_mxJPO.PARAM_IGNORE_WSO_USERS);
582         boolean ignore = false;
583         if (ignoreMatches != null)  {
584             for (final String ignoreMatch : ignoreMatches)  {
585                 if (StringUtil_mxJPO.match(this.getName(), ignoreMatch))  {
586                     ignore = true;
587                     break;
588                 }
589             }
590         }
591 
592         return ignore;
593     }
