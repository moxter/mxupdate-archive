21  package org.mxupdate.update.user.workspace;
22  
23  import java.io.IOException;
24  
25  import org.mxupdate.update.user.AbstractUser_mxJPO;
26  import org.mxupdate.update.util.ParameterCache_mxJPO;
27  import org.mxupdate.update.util.StringUtil_mxJPO;
28  
29  /***
30   * User specific class to store the workspace object for one cue.
31   *
32   * @author The MxUpdate Team
33   * @version $Id: Cue_mxJPO.java 556 2009-10-10 18:04:25Z tim.moxter $
34   */
35  public class Cue_mxJPO
36      extends AbstractVisualQueryWorkspaceObject_mxJPO
37  {
38      /***
39       * Order in which the cue is applied in relation to other cues: before,
40       * with or after other cues (-1 / 0 / 1).
41       *
42       * @see #parse(String, String)
43       * @see #write(ParameterCache_mxJPO, Appendable)
44       */
45      private String order;
46  
47      /***
48       * Foreground color.
49       *
50       * @see #parse(String, String)
51       * @see #write(ParameterCache_mxJPO, Appendable)
52       */
53      private String foregroundColor;
54  
55      /***
56       * Highlight color.
57       *
58       * @see #parse(String, String)
59       * @see #write(ParameterCache_mxJPO, Appendable)
60       */
61      private String highlightColor;
62  
63      /***
64       * Name of the font.
65       *
66       * @see #parse(String, String)
67       * @see #write(ParameterCache_mxJPO, Appendable)
68       */
69      private String fontName;
70  
71      /***
72       * Line style.
73       *
74       * @see #parse(String, String)
75       * @see #write(ParameterCache_mxJPO, Appendable)
76       */
77      private String lineStyle;
78  
79      /***
80       * Default constructor.
81       *
82       * @param _user         related user for this workspace object
83       */
84      public Cue_mxJPO(final AbstractUser_mxJPO _user)
85      {
86          super(_user, "cue");
87      }
88  
89      /***
90       * <p>Parses all cue specific URL values. This includes:
91       * <ul>
92       * <li>{@link #order}</li>
93       * <li>{@link #foregroundColor foreground color}</li>
94       * <li>{@link #highlightColor highlight color}</li>
95       * <li>{@link #fontName name of the font}</li>
96       * <li>{@link #lineStyle line style}</li>
97       * </ul></p>
98       *
99       * @param _url      URL to parse
100      * @param _content  content of the URL to parse
101      */
102     @Override()
103     public void parse(final String _url,
104                       final String _content)
105     {
106         if ("/zlevel".equals(_url))  {
107             this.order = _content;
108         } else if ("/foregroundColor".equals(_url))  {
109             this.foregroundColor = _content;
110         } else if ("/highlightColor".equals(_url))  {
111             this.highlightColor = _content;
112         } else if ("/fontName".equals(_url))  {
113             this.fontName = _content;
114         } else if ("/lineStyle".equals(_url))  {
115             this.lineStyle = _content;
116         } else  {
117             super.parse(_url, _content);
118         }
119     }
120 
121     /***
122      * <p>Writes all cue specific values to the TCL update file
123      * <code>_out</code>. This includes:
124      * <ul>
125      * <li>{@link #order}</li>
126      * <li>{@link #foregroundColor foreground color}</li>
127      * <li>{@link #highlightColor highlight color}</li>
128      * <li>{@link #fontName name of the font}</li>
129      * <li>{@link #lineStyle line style}</li>
130      * </ul></p>
131      *
132      * @param _paramCache   parameter cache
133      * @param _out          appendable instance to the TCL update file
134      * @throws IOException if the TCL update code could not written
135      */
136     @Override()
137     public void write(final ParameterCache_mxJPO _paramCache,
138                       final Appendable _out)
139         throws IOException
140     {
141         super.write(_paramCache, _out);
142         if (this.order != null)  {
143             _out.append(" //\n    order \"").append(StringUtil_mxJPO.convertTcl(this.order)).append("\"");
144         }
145         if (this.foregroundColor != null)  {
146             _out.append(" //\n    color \"").append(StringUtil_mxJPO.convertTcl(this.foregroundColor)).append("\"");
147         }
148         if (this.highlightColor != null)  {
149             _out.append(" //\n    highlight \"").append(StringUtil_mxJPO.convertTcl(this.highlightColor)).append("\"");
150         }
151         if (this.fontName != null)  {
152             _out.append(" //\n    font \"").append(StringUtil_mxJPO.convertTcl(this.fontName)).append("\"");
153         }
154         if (this.lineStyle != null)  {
155             _out.append(" //\n    linestyle \"").append(StringUtil_mxJPO.convertTcl(this.lineStyle)).append("\"");
156         }
157     }
