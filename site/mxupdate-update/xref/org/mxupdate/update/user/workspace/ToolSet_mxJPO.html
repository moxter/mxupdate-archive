25  import org.mxupdate.update.user.AbstractUser_mxJPO;
26  import org.mxupdate.update.util.AdminProperty_mxJPO;

30  /***
31   * User specific class to store the workspace object for one tool set.
32   *
33   * @author The MxUpdate Team
34   * @version $Id: ToolSet_mxJPO.java 559 2009-10-11 19:16:59Z tim.moxter $
35   */
36  public class ToolSet_mxJPO
37      extends AbstractVisualWorkspaceObject_mxJPO
38  {
39      /***
40       * Default constructor.
41       *
42       * @param _user         related user for this workspace object
43       */
44      public ToolSet_mxJPO(final AbstractUser_mxJPO _user)
45      {
46          super(_user, "toolset");
47      }
48  
49      /***
50       * <p>Writes all tool set specific values to the TCL update file
51       * <code>_out</code>. This includes:
52       * <ul>
53       * <li>all member programs defined via the properties with the name
54       *     &quot;%Member&quot;</li>
55       * </ul></p>
56       *
57       * @param _paramCache   parameter cache
58       * @param _out          appendable instance to the TCL update file
59       * @throws IOException if the TCL update code could not written
60       */
61      @Override()
62      public void write(final ParameterCache_mxJPO _paramCache,
63                        final Appendable _out)
64          throws IOException
65      {
66          super.write(_paramCache, _out);
67  
68          // methods
69          for (final AdminProperty_mxJPO prop : this.getProperties())  {
70              if (prop.isSetting() && "%Member".equals(prop.getName()))  {
71                  _out.append(" //\n    program \"").append(StringUtil_mxJPO.convertTcl(prop.getRefAdminName())).append("\"");
72              }
73          }
74      }
75  }
