279     /***
280      * Returns a set of found person names. First for all administration
281      * persons are searched. Then, all persons are removed from the matching
282      * set, for whom a related business person object exists. This new set is
283      * returned.
284      *
285      * @param _paramCache   parameter cache
286      * @return set of person names for which no person business object exists
287      * @throws MatrixException if the query for person objects failed
288      */
289     @Override()
290     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
291         throws MatrixException
292     {
293         final Set<String> persons = super.getMxNames(_paramCache);
294 
295         // evaluate for all business person object names
296         final StringList selects = new StringList();
297         selects.addElement("name");
298         final Query query = new Query();
299         query.open(_paramCache.getContext());
300         query.setBusinessObjectType(this.getTypeDef().getMxBusType());
301         final BusinessObjectWithSelectList list = query.select(_paramCache.getContext(), selects);
302         query.close(_paramCache.getContext());
303 
304         // remove the found persons which related business object
305         for (final Object mapObj : list)  {
306             final BusinessObjectWithSelect map = (BusinessObjectWithSelect) mapObj;
307             final String busName = (String) map.getSelectDataList("name").get(0);
308             persons.remove(busName);
309         }
310 
311         return persons;
312     }

314     /***
315      * <p>Parses all person specific URL values. This includes:
316      * <ul>
317      * <li>{@link #groups assigned groups}</li>
318      * <li>{@link #roles assigned roles}</li>
319      * <li>{@link #access}</li>
320      * <li>{@link #adminAccess business administration access}</li>
321      * <li>is person {@link #isInactive inactive}</li>
322      * <li>flag if the person is an {@link #isApplicationUser application user}
323      *     </li>
324      * <li>flag if the person is a {@link #isFullUser full user}</li>
325      * <li>flag if the person is a
326      *     {@link #isBusinessAdministrator business administrator}</li>
327      * <li>flag if the person is a
328      *     {@link #isSystemAdministrator system administrator}</li>
329      * <li>flag if the person is a {@link #isTrusted trusted} person</li>
330      * <li>person wants {@link #wantsEmail email}</li>
331      * <li>person wants {@link #wantsIconMail icon mail}</li>
332      * <li>{@link #address}</li>
333      * <li>{@link #email email address}</li>
334      * <li>{@link #fax fax number}</li>
335      * <li>{@link #fullName full name}</li>
336      * <li>{@link #vault default vault}</li>
337      * <li>{@link #phone phone number}</li>
338      * <li>{@link #passwordNeverExpires password never expires flag}</li>
339      * </ul></p>
340      * <p>If an <code>_url</code> is included in {@link #IGNORED_URLS}, this
341      * URL is ignored.</p>
342      *
343      * @param _url      URL to parse
344      * @param _content  content depending on the URL
345      * @see #IGNORED_URLS
346      */
347     @Override()
348     protected void parse(final String _url,
349                          final String _content)
350     {
351         if (!PersonAdmin_mxJPO.IGNORED_URLS.contains(_url))  {
352             if ("/assignmentList/assignment/groupRef".equals(_url))  {
353                 this.groups.add(_content);
354             } else if ("/assignmentList/assignment/roleRef".equals(_url))  {
355                 this.roles.add(_content);
356 
357             } else if (_url.startsWith("/access"))  {
358                 this.access.add(_url.substring(8).replaceAll("Access$", ""));
359 
360             } else if (_url.startsWith("/adminAccess"))  {
361                 this.adminAccess.add(_url.substring(13).replaceAll("Access$", ""));
362 
363             } else if ("/defaultApplication/applicationRef".equals(_url))  {
364                 this.defaultApplication = _content;
365 
366             } else if ("/inactive".equals(_url))  {
367                 this.isInactive = true;
368             } else if ("/applicationsOnly".equals(_url))  {
369                 this.isApplicationUser = true;
370             } else if ("/fullUser".equals(_url))  {
371                 this.isFullUser = true;
372             } else if ("/businessAdministrator".equals(_url))  {
373                 this.isBusinessAdministrator = true;
374             } else if ("/systemAdministrator".equals(_url))  {
375                 this.isSystemAdministrator = true;
376             } else if ("/trusted".equals(_url))  {
377                 this.isTrusted = true;
378 
379             } else if ("/wantsEmail".equals(_url))  {
380                 this.wantsEmail = true;
381             } else if ("/wantsIconMail".equals(_url))  {
382                 this.wantsIconMail = true;
383 
384             } else if ("/address".equals(_url))  {
385                 this.address = _content;
386             } else if ("/email".equals(_url))  {
387                 this.email = _content;
388             } else if ("/fax".equals(_url))  {
389                 this.fax = _content;
390             } else if ("/fullName".equals(_url))  {
391                 this.fullName = _content;
392             } else if ("/homeVault/vaultRef".equals(_url))  {
393                 this.vault = _content;
394             } else if ("/phone".equals(_url))  {
395                 this.phone = _content;
396 
397             } else if ("/passwordNeverExpires".equals(_url))  {
398                 this.passwordNeverExpires = true;
399 
400             } else  {
401                 super.parse(_url, _content);
402             }
403         }
404     }

422     /***
423      * Writes specific information about the cached administration person to
424      * the given TCL update file <code>_out</code>. The included information is
425      * <ul>
426      * <li>{@link #passwordNeverExpires password never expires flag} (if
427      *     parameter {@link #PARAM_IGNORE_PSWD_NEVER_EXPIRES} is not set)</li>
428      * <li>{@link #access}</li>
429      * <li>{@link #adminAccess business administration access}</li>
430      * <li>person wants {@link #wantsEmail email} (if parameter
431      *     {@link #PARAM_IGNORE_WANTS_EMAIL} is not set)</li>
432      * <li>person wants {@link #wantsIconMail icon mail} (if parameter
433      *     {@link #PARAM_IGNORE_WANTS_ICON_MAIL} is not set)</li>
434      * <li>{@link #address}</li>
435      * <li>{@link #email email address}</li>
436      * <li>{@link #fax fax number}</li>
437      * <li>{@link #fullName full name}</li>
438      * <li>{@link #phone phone number}</li>
439      * <li>{@link #vault default vault}</li>
440      * <li>{@link #defaultApplication default application} (if defined)</li>
441      * <li>assigned {@link #groups}</li>
442      * <li>assigned {@link #roles}</li>
443      * </ul>
444      *
445      * @param _paramCache   parameter cache
446      * @param _out          appendable instance to the TCL update file
447      * @throws IOException if the TCL update code could not written
448      * @see #PARAM_IGNORE_PSWD_NEVER_EXPIRES
449      * @see #PARAM_IGNORE_WANTS_EMAIL
450      */
451     @Override()
452     protected void writeObject(final ParameterCache_mxJPO _paramCache,
453                                final Appendable _out)
454         throws IOException
455     {
456         super.writeObject(_paramCache, _out);
457 
458         // password never expires flag if not matched
459         final Collection<String> ignorePswdNeverExpires = _paramCache.getValueList(PersonAdmin_mxJPO.PARAM_IGNORE_PSWD_NEVER_EXPIRES);
460         if ((ignorePswdNeverExpires == null) || !StringUtil_mxJPO.match(this.getName(), ignorePswdNeverExpires))  {
461             _out.append(" //\n    ").append(this.passwordNeverExpires ? "" : "!").append("neverexpires");
462         }
463         _out.append(" //\n    access \"").append(StringUtil_mxJPO.joinTcl(',', false, this.access, "none")).append("\"")
464             .append(" //\n    admin \"").append(StringUtil_mxJPO.joinTcl(',', false, this.adminAccess, "none")).append("\"");
465         // wants email only if not matched
466         final Collection<String> ignoreWantsEmail = _paramCache.getValueList(PersonAdmin_mxJPO.PARAM_IGNORE_WANTS_EMAIL);
467         if ((ignoreWantsEmail == null) || !StringUtil_mxJPO.match(this.getName(), ignoreWantsEmail))  {
468             _out.append(" //\n    ").append(this.wantsEmail ? "enable" : "disable").append(" email");
469         }
470         // wants icon mail only if not matched
471         final Collection<String> ignoreWantsIconMail = _paramCache.getValueList(PersonAdmin_mxJPO.PARAM_IGNORE_WANTS_ICON_MAIL);
472         if ((ignoreWantsIconMail == null) || !StringUtil_mxJPO.match(this.getName(), ignoreWantsIconMail))  {
473             _out.append(" //\n    ").append(this.wantsIconMail ? "enable" : "disable").append(" iconmail");
474         }
475         _out.append(" //\n    address \"").append(StringUtil_mxJPO.convertTcl(this.address)).append("\"")
476             .append(" //\n    email \"").append(StringUtil_mxJPO.convertTcl(this.email)).append("\"")
477             .append(" //\n    fax \"").append(StringUtil_mxJPO.convertTcl(this.fax)).append("\"")
478             .append(" //\n    fullname \"").append(StringUtil_mxJPO.convertTcl(this.fullName)).append("\"")
479             .append(" //\n    phone \"").append(StringUtil_mxJPO.convertTcl(this.phone)).append("\"")
480             .append(" //\n    vault \"").append(StringUtil_mxJPO.convertTcl(this.vault)).append("\"");
481         if (this.defaultApplication != null)  {
482             _out.append(" //\n    application \"")
483                 .append(StringUtil_mxJPO.convertTcl(this.defaultApplication)).append("\"");
484         }
485         for (final String group : this.groups)  {
486             _out.append(" //\n    assign group \"")
487                 .append(StringUtil_mxJPO.convertTcl(group))
488                 .append("\"");
489         }
490         for (final String role : this.roles)  {
491             _out.append(" //\n    assign role \"")
492                 .append(StringUtil_mxJPO.convertTcl(role))
493                 .append("\"");
494         }
495     }

497     /***
498      * Appends at the end of the TCL update file the change of the person
499      * &quot;type&quot;. The person &quot;type&quot; defines, if the person is
500      * e.g. active, or trusted etc..
501      *
502      * @param _paramCache   parameter cache
503      * @param _out          appendable instance to the TCL update file
504      * @throws IOException if the extension could not be written
505      */
506     @Override()
507     protected void writeEnd(final ParameterCache_mxJPO _paramCache,
508                             final Appendable _out)
509             throws IOException
510     {
511         super.writeEnd(_paramCache, _out);
512         _out.append("\nmql mod person \"${NAME}\" type ");
513         if (!this.isApplicationUser)  {
514             _out.append("not");
515         }
516         _out.append("application,");
517         if (!this.isFullUser)  {
518             _out.append("not");
519         }
520         _out.append("full,");
521         if (!this.isBusinessAdministrator)  {
522             _out.append("not");
523         }
524         _out.append("business,");
525         if (!this.isInactive)  {
526             _out.append("not");
527         }
528         _out.append("inactive,");
529         if (!this.isTrusted)  {
530             _out.append("not");
531         }
532         _out.append("trusted,");
533         if (!this.isSystemAdministrator)  {
534             _out.append("not");
535         }
536         _out.append("system");
537     }

539     /***
540      * The method overwrites the original method to
541      * <ul>
542      * <li>reset the comment (description)</li>
543      * <li>set the version and author attribute</li>
544      * <li>reset all not ignored attributes</li>
545      * <li>define the TCL variable &quot;OBJECTID&quot; with the object id of
546      *     the represented business object</li>
547      * <li>sets the {@link #passwordNeverExpires password never expires flag}
548      *     to <i>false</i> (means that the password expires); depends on
549      *     parameter {@link #PARAM_IGNORE_PSWD_NEVER_EXPIRES}</li>
550      * <li>disables that the person wants email; depends on
551      *     parameter {@link #PARAM_IGNORE_WANTS_EMAIL}</li>
552      * <li>enables that the person wants icon mail; depends on
553      *     parameter {@link #PARAM_IGNORE_WANTS_ICON_MAIL}</li>
554      * </ul>
555      * The original method of the super class if called surrounded with a
556      * history off, because if the update itself is done the modified basic
557      * attribute and the version attribute of the business object is updated.
558      * <br/>
559      * The new generated MQL code is set in the front of the already defined
560      * MQL code in <code>_preMQLCode</code> and appended to the MQL statements
561      * in <code>_postMQLCode</code>.
562      *
563      * @param _paramCache       parameter cache
564      * @param _preMQLCode       MQL statements which must be called before the
565      *                          TCL code is executed
566      * @param _postMQLCode      MQL statements which must be called after the
567      *                          TCL code is executed
568      * @param _preTCLCode       TCL code which is defined before the source
569      *                          file is sourced
570      * @param _tclVariables     map of all TCL variables where the key is the
571      *                          name and the value is value of the TCL variable
572      *                          (the value is automatically converted to TCL
573      *                          syntax!)
574      * @param _sourceFile       souce file with the TCL code to update
575      * @throws Exception if the update from derived class failed
576      * @see #PARAM_IGNORE_PSWD_NEVER_EXPIRES
577      * @see #PARAM_IGNORE_WANTS_EMAIL
578      * @see #PARAM_IGNORE_WANTS_ICON_MAIL
579      */
580     @Override()
581     protected void update(final ParameterCache_mxJPO _paramCache,
582                           final CharSequence _preMQLCode,
583                           final CharSequence _postMQLCode,
584                           final CharSequence _preTCLCode,
585                           final Map<String,String> _tclVariables,
586                           final File _sourceFile)
587         throws Exception
588     {
589         // append other pre MQL code
590         final StringBuilder preMQLCode = new StringBuilder()
591                 .append(_preMQLCode)
592                 .append("escape mod person \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
593                         .append("comment '' ")
594                         .append("access none ")
595                         .append("admin none ")
596                         .append("address '' ")
597                         .append("email '' ")
598                         .append("fax '' ")
599                         .append("fullname '' ")
600                         .append("phone '' ")
601                         .append("remove assign all");
602         // reset wants email if not ignored
603         final Collection<String> ignoreWantsEmail = _paramCache.getValueList(PersonAdmin_mxJPO.PARAM_IGNORE_WANTS_EMAIL);
604         if ((ignoreWantsEmail == null) || !StringUtil_mxJPO.match(this.getName(), ignoreWantsEmail))  {
605             preMQLCode.append(" disable email");
