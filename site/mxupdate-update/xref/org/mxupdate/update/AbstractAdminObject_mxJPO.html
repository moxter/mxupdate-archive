147     /***
148      * Reads the symbolic names for current admin objects and stores them in
149      * {@link #symbolicNames}.
150      *
151      * @param _paramCache       parameter cache
152      * @throws MatrixException if the symbolic names could not be read
153      * @see #symbolicNames
154      */
155     protected void readSymbolicNames(final ParameterCache_mxJPO _paramCache)
156         throws MatrixException
157     {
158         // context must be checked if used within automatic tests
159         if ((this.getTypeDef().getMxAdminName() != null) && (_paramCache.getContext() != null))  {
160             final String symbProg = _paramCache.getValueString(AbstractAdminObject_mxJPO.PARAM_SYMB_NAME_PROG);
161             final String symbProgIdxOf = new StringBuilder()
162                     .append(" on program ").append(symbProg).append(' ').toString();
163             final StringBuilder cmd = new StringBuilder()
164                     .append("escape list property on program \"")
165                                 .append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
166                         .append(this.getTypeDef().getMxAdminName())
167                         .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
168                         .append(this.getTypeDef().getMxAdminSuffix());
169             for (final String symbName : MqlUtil_mxJPO.execMql(_paramCache, cmd).split("\n"))  {
170                 if (!"".equals(symbName))  {
171                     this.symbolicNames.add(symbName.substring(0, symbName.indexOf(symbProgIdxOf)));
172                 }
173             }
174         }
175     }

177     /***
178      * Appends the escaped MQL code to register given symbolic name
179      * <code>_symbName</code> depending on current defined symbolic names
180      * {@link #symbolicNames}. If not registered, a new registration is done.
181      * If wrong symbolic names are defined, they are removed.
182      *
183      * @param _paramCache   parameter cache
184      * @param _symbName     symbolic name which must be set
185      * @param _mqlCode      string builder where the MQL command must be
186      *                      appended
187      * @see #symbolicNames
188      */
189     protected void appendSymbolicNameRegistration(final ParameterCache_mxJPO _paramCache,
190                                                   final String _symbName,
191                                                   final StringBuilder _mqlCode)
192     {
193         if (this.getTypeDef().getMxAdminName() != null)  {
194             final String symbProg = _paramCache.getValueString(AbstractAdminObject_mxJPO.PARAM_SYMB_NAME_PROG);
195             if (!this.symbolicNames.contains(_symbName))  {
196                 _paramCache.logTrace("    - register symbolic name '" + _symbName + "'");
197                 _mqlCode.append("escape add property \"").append(StringUtil_mxJPO.convertMql(_symbName)).append("\" ")
198                         .append(" on program \"").append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
199                         .append(this.getTypeDef().getMxAdminName())
200                         .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
201                         .append(this.getTypeDef().getMxAdminSuffix())
202                         .append(";\n");
203             }
204             for (final String exSymbName : this.symbolicNames)  {
205                 if (!_symbName.equals(exSymbName))  {
206                     _paramCache.logTrace("    - remove symbolic name '" + exSymbName + "'");
207                     _mqlCode.append("escape delete property \"")
208                                     .append(StringUtil_mxJPO.convertMql(exSymbName)).append("\" ")
209                             .append(" on program \"").append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
210                             .append(this.getTypeDef().getMxAdminName())
211                             .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
212                             .append(this.getTypeDef().getMxAdminSuffix())
213                             .append(";\n");
214                 }
215             }
216         }
217     }

266     /***
267      * Creates a XML representation of the Object to export, parses them and
268      * executes the post preparation {@link #prepare(ParameterCache_mxJPO)}.
269      *
270      * @param _paramCache   parameter cache
271      * @throws MatrixException  if the export of the admin object failed
272      * @throws SAXException     if the exported XML document could not be
273      *                          parsed
274      * @throws IOException      should not happen; only if the input source of
275      *                          the string reader which embeds the XML document
276      *                          failed
277      * @see #getExportMQL()                 used to get the MQL command to get
278      *                                      a XML representation
279      * @see PadSaxHandler                   SAX handler to parse the XML file
280      * @see #parse(String, String)          parser called within the SAX
281      *                                      handler
282      * @see #prepare(ParameterCache_mxJPO)  called post preparation method
283      */
284     @Override()
285     protected void parse(final ParameterCache_mxJPO _paramCache)
286         throws MatrixException, SAXException, IOException
287     {
288         // create XML reader
289         final XMLReader reader = XMLReaderFactory.createXMLReader();
290         // register Sax Content Handler
291         final PadSaxHandler handler = new PadSaxHandler();
292         reader.setContentHandler(handler);
293         reader.setDTDHandler(handler);
294         reader.setEntityResolver(handler);
295         // parse the XML string of the export
296         final InputSource inputSource = new InputSource(new StringReader(this.execXMLExport(_paramCache)));
297         inputSource.setEncoding("UTF8");
298         reader.parse(inputSource);
299         // prepare post preparation
300         this.prepare(_paramCache);
301     }

335     /***
336      * Parsed administration object related XML tags. This includes:
337      * <ul>
338      * <li>description</li>
339      * <li>is the object {@link #hidden}</li>
340      * <li>properties</li>
341      * </ul>
342      *
343      * @param _url      URL within the XML
344      * @param _content  value of the URL
345      * @see #IGNORED_URLS
346      */
347     protected void parse(final String _url,
348                          final String _content)
349     {
350         if (!AbstractAdminObject_mxJPO.IGNORED_URLS.contains(_url))  {
351             if ("/adminProperties/description".equals(_url))  {
352                 this.setDescription(_content);
353             } else if ("/adminProperties/hidden".equals(_url))  {
354                 this.hidden = true;
355 
356             } else if ("/adminProperties/propertyList/property".equals(_url))  {
357                 this.propertiesStack.add(new AdminProperty_mxJPO());
358             } else if (_url.startsWith("/adminProperties/propertyList/property"))  {
359                 if (!this.propertiesStack.peek().parse(_url.substring(38), _content))  {
360                     System.err.println("unknown parsing property url: "+_url+"("+_content+")");
361                 }
362             } else  {
363                 System.err.println("unknown parsing url: "+_url+"("+_content+")");
364             }
365         }
366     }

368     /***
369      * Sorted the properties, sets the author and version depending on the
370      * properties and reads the symbolic names.
371      *
372      * @param _paramCache   parameter cache
373      * @throws MatrixException if the symbolic names could not be extracted
374      * @see #propertiesStack
375      * @see #propertiesMap
376      */
377     protected void prepare(final ParameterCache_mxJPO _paramCache)
378         throws MatrixException
379     {
380         // sort the properties
381         for (final AdminProperty_mxJPO property : this.propertiesStack)  {
382             final StringBuilder key = new StringBuilder().append(property.getName());
383             if ((property.getRefAdminName() != null) && (property.getRefAdminType() != null))  {
384                 key.append("::").append(property.getRefAdminType())
385                    .append("::").append(property.getRefAdminName());
386             }
387             this.propertiesMap.put(key.toString(), property);
388         }
389         // set author depending on the properties
390         final AdminProperty_mxJPO author = this.propertiesMap.get(PropertyDef_mxJPO.AUTHOR.getPropName(_paramCache));
391         if (author != null)  {
392             this.setAuthor(author.getValue());
393         }
394         // set application depending on the properties
395         final AdminProperty_mxJPO appl = this.propertiesMap.get(PropertyDef_mxJPO.APPLICATION.getPropName(_paramCache));
396         if (appl != null)  {
397             this.setApplication(appl.getValue());
398         }
399         // sets the installation date depending on the properties
400         final AdminProperty_mxJPO installationDate
401                 = this.propertiesMap.get(PropertyDef_mxJPO.INSTALLEDDATE.getPropName(_paramCache));
402         if (installationDate != null)  {
403             this.setInstallationDate(installationDate.getValue());
404         }
405         // sets the installer depending on the properties
406         final AdminProperty_mxJPO installer
407                 = this.propertiesMap.get(PropertyDef_mxJPO.INSTALLER.getPropName(_paramCache));
408         if (installer != null)  {
409             this.setInstaller(installer.getValue());
410         }
411         // sets the original name depending on the properties
412         final AdminProperty_mxJPO origName
413                 = this.propertiesMap.get(PropertyDef_mxJPO.ORIGINALNAME.getPropName(_paramCache));
414         if (origName != null)  {
415             this.setOriginalName(origName.getValue());
416         }
417         // sets the version depending on the properties
418         final AdminProperty_mxJPO version = this.propertiesMap.get(PropertyDef_mxJPO.VERSION.getPropName(_paramCache));
419         if (version != null)  {
420             this.setVersion(version.getValue());
421         }
422 
423         // reads symbolic names of the administration objects
424         this.readSymbolicNames(_paramCache);
425     }

431      * @throws IOException  if the write of the TCL update to the writer

439     {

444         if (!"".equals(this.getTypeDef().getMxAdminSuffix()))  {
445             _out.append(" ").append(this.getTypeDef().getMxAdminSuffix());
446         }
447         _out.append(" //\n    description \"").append(StringUtil_mxJPO.convertTcl(this.getDescription())).append("\"");
448         this.writeObject(_paramCache, _out);
449         this.writeProperties(_paramCache, _out);
450         this.writeEnd(_paramCache, _out);
451     }

479     /***
480      * Writes the MQL code to add all none standard
481      * {@link #propertiesMap properties} to the TCL update file.
482      *
483      * @param _paramCache   parameter cache
484      * @param _out          appendable instance to the TCL update file
485      * @throws IOException if the write to the TCL update file failed
486      */
487     protected void writeProperties(final ParameterCache_mxJPO _paramCache,
488                                    final Appendable _out)
489         throws IOException
490     {
491         for (final AdminProperty_mxJPO prop : this.propertiesMap.values())  {
492             if ((PropertyDef_mxJPO.getEnumByPropName(_paramCache, prop.getName()) == null) && !prop.isSetting())  {
493                 _out.append("\nmql escape add property \"").append(StringUtil_mxJPO.convertTcl(prop.getName())).append("\"")
494                     .append(" //\n    on ")
495                     .append(this.getTypeDef().getMxAdminName())
496                     .append(" \"${NAME}\"");
497                 if (!"".equals(this.getTypeDef().getMxAdminSuffix()))  {
498                     _out.append(' ').append(this.getTypeDef().getMxAdminSuffix());
499                 }
500                 if ((prop.getRefAdminName()) != null && (prop.getRefAdminType() != null))  {
501                     _out.append("  //\n    to ").append(prop.getRefAdminType())
502                         .append(" \"").append(StringUtil_mxJPO.convertTcl(prop.getRefAdminName())).append("\"");
503                     // if target is a table, a system is required!
504                     if ("table".equals(prop.getRefAdminType()))  {
505                         _out.append(" system");
506                     }
507                 }
508                 if (prop.getValue() != null)  {
509                     _out.append(" //\n    value \"").append(StringUtil_mxJPO.convertTcl(prop.getValue())).append("\"");
510                 }
511             }
512         }
513     }

549     /***
550      * The method overwrites the original method append MQL code to
551      * <ul>
552      * <li>remove all existing {@link #propertiesMap properties}</li>
553      * <li>define the TCL variable for the name</li>
554      * <li>define property &quot;version&quot;</li>
555      * <li>define property &quot;file date&quot;</li>
556      * <li>define &quot;installed&quot; date property if not defined</li>
557      * <li>define &quot;installer&quot; property if not defined</li>
558      * <li>define &quot;original name&quot; property if not defined</li>
559      * <li>define &quot;application&quot; property if not set or not equal to
560      *     defined value from user</li>
561      * <li>define &quot;author&quot; property if not set or not equal to value
562      *     from TCL update code (or to default value if not defined in TCL
563      *     update code)</li>
564      * <li>register administration object (if not already done) and remove all
565      *     symbolic names which are not correct</li>
566      * </ul>
567      * Then the update method of the super class is called.
568      *
569      * @param _paramCache       parameter cache
570      * @param _preMQLCode       MQL statements which must be called before the
571      *                          TCL code is executed
572      * @param _postMQLCode      MQL statements which must be called after the
573      *                          TCL code is executed
574      * @param _preTCLCode       TCL code which is defined before the source
575      *                          file is sourced
576      * @param _tclVariables     map of all TCL variables where the key is the
577      *                          name and the value is value of the TCL variable
578      *                          (the value is automatically converted to TCL
579      *                          syntax!)
580      * @param _sourceFile       souce file with the TCL code to update
581      * @throws Exception if the update from the derived class failed
582      */
583     @Override()
584     protected void update(final ParameterCache_mxJPO _paramCache,
585                           final CharSequence _preMQLCode,
586                           final CharSequence _postMQLCode,
587                           final CharSequence _preTCLCode,
588                           final Map<String,String> _tclVariables,
589                           final File _sourceFile)
590         throws Exception
591     {
592         // remove all properties
593         final StringBuilder preMQLCode = new StringBuilder();
594         for (final AdminProperty_mxJPO prop : this.propertiesMap.values())  {
595             // % must be ignored because this means settings
596             if ((PropertyDef_mxJPO.getEnumByPropName(_paramCache, prop.getName()) == null) && !prop.isSetting())  {
597                 // must be done via modify because of properties without names
