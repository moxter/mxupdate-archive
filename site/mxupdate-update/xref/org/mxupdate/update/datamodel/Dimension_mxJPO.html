55       * Called TCL procedure within the TCL update to parse the new dimension
56       * definition. The TCL procedure calls method
57       * {@link #jpoCallExecute(ParameterCache_mxJPO, String...)} with the new
58       * dimension definition. All quot's are replaced by <code>@0@0@</code> and
59       * all apostroph's are replaced by <code>@1@1@</code>.
60       *
61       * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
62       * @see #jpoCallExecute(ParameterCache_mxJPO, String...)
63       */
64      private static final String TCL_PROCEDURE
65              = "proc updateDimension {_sDimension _lsArgs}  {\n"
66                  + "regsub -all {'} $_lsArgs {@0@0@} sArg\n"
67                  + "regsub -all {//\"} $sArg {@1@1@} sArg\n"
68                  + "regsub -all {//////[} $sArg {[} sArg\n"
69                  + "regsub -all {//////]} $sArg {]} sArg\n"
70                  + "mql exec prog org.mxupdate.update.util.JPOCaller dimension ${_sDimension} \"${sArg}\"\n"
71              + "}\n";

149     /***
150      * Parses all dimension specific expression URLs.
151      *
152      * @param _url      URL to parse
153      * @param _content  content of the URL to parse
154      */
155     @Override
156     protected void parse(final String _url,
157                          final String _content)
158     {
159         if (!Dimension_mxJPO.IGNORED_URLS.contains(_url))  {
160             if ("/unitList/unit".equals(_url))  {
161                 if (this.currentUnit != null)  {
162                     this.units.add(this.currentUnit);
163                 }
164                 this.currentUnit = new Unit();
165             } else if ("/unitList/unit/adminProperties/name".equals(_url))  {
166                 this.currentUnit.name = _content;
167             } else if ("/unitList/unit/adminProperties/description".equals(_url))  {
168                 this.currentUnit.description = _content;
169             } else if ("/unitList/unit/adminProperties/propertyList/property".equals(_url))  {
170                 if (this.currentUnit.currentUnitProperty != null)  {
171                     this.currentUnit.properties.add(this.currentUnit.currentUnitProperty);
172                 }
173                 this.currentUnit.currentUnitProperty = new AdminProperty_mxJPO();
174             } else if (_url.startsWith("/unitList/unit/adminProperties/propertyList/property"))  {
175                 if (!this.currentUnit.currentUnitProperty.parse(_url.substring(52), _content))  {
176                     super.parse(_url, _content);
177                 }
178             } else if ("/unitList/unit/unitDefault".equals(_url))  {
179                 this.currentUnit.defaultUnit = true;
180             } else if ("/unitList/unit/unitLabel".equals(_url))  {
181                 this.currentUnit.label = _content;
182             } else if ("/unitList/unit/unitMultiplier".equals(_url))  {
183                 this.currentUnit.multiplier = Double.parseDouble(_content);
184             } else if ("/unitList/unit/unitOffset".equals(_url))  {
185                 this.currentUnit.offset = Double.parseDouble(_content);
186             } else  {
187                 super.parse(_url, _content);
188             }
189         }
190     }

192     /***
193      * Prepares unit relation information for dimensions. While parsing current
194      * values are stored separately and must be prepared for usage and some
195      * unit information is stored in properties and must be extracted:
196      * <ul>
197      * <li>If a parsed {@link #currentUnit} exists this unit is added to
198      *     {@link #units}.</li>
199      * <li>If a unit has a parsed {@link Unit#currentUnitProperty} this
200      *     property is added to the {@link Unit#properties}.</li>
201      * <li>If the {@link Unit#properties} includes settings they are extracted
202      *     into {@link Unit#settings}.</li>
203      * <li>If the {@link Unit#properties} includes a grouped system information
204      *     the values are written into {@link Unit#systemInfos}.</li>
205      * </ul>
206      *
207      * @param _paramCache   parameter cache
208      * @throws MatrixException if preparation failed
209      */
210     @Override
211     protected void prepare(final ParameterCache_mxJPO _paramCache)
212         throws MatrixException
213     {
214         if (this.currentUnit != null)  {
215             this.units.add(this.currentUnit);
216             this.currentUnit = null;
217         }
218         for (final Unit unit : this.units)  {
219             if (unit.currentUnitProperty != null)  {
220                 unit.properties.add(unit.currentUnitProperty);
221                 unit.currentUnitProperty = null;
222             }
223             for (final AdminProperty_mxJPO prop : new HashSet<AdminProperty_mxJPO>(unit.properties))  {
224                 // extract settings
225                 if (prop.isSetting())  {
226                     unit.settings.put(prop.getName().substring(1), prop.getValue());
227                     unit.properties.remove(prop);
228                 } else if ("unit".equals(prop.getRefAdminType()))  {
229                     Set<String> units = unit.systemInfos.get(prop.getName());
230                     if (units == null)  {
231                         units = new TreeSet<String>();
232                         unit.systemInfos.put(prop.getName(), units);
233                     }
234                     units.add(prop.getRefAdminName());
235                     unit.properties.remove(prop);
236                 }
237             }
238         }
239         super.prepare(_paramCache);
240     }

242     /***
243      * Writes the TCL update files for dimensions. The original method is
244      * overwritten because a dimension could not be only updated. A compare
245      * must be done in front or otherwise some data is lost.
246      *
247      * @param _paramCache   parameter cache
248      * @param _out          appendable instance to the TCL update file
249      * @throws IOException if the extension could not be written
250      * @see #units
251      */
252     @Override
253     protected void write(final ParameterCache_mxJPO _paramCache,
254                          final Appendable _out)
255             throws IOException
256     {
257         // write header
258         this.writeHeader(_paramCache, _out);
259 
260         // write dimension description
261         _out.append("updateDimension \"${NAME}\"  {\n")
262             .append("  description \"").append(StringUtil_mxJPO.convertTcl(this.getDescription())).append("\"\n")
263             .append("  hidden \"").append(Boolean.toString(this.isHidden())).append("\"\n");
264         for (final Unit unit : this.units)  {
265             _out.append("  unit \"").append(StringUtil_mxJPO.convertTcl(unit.name)).append("\" {\n");
266             if (unit.defaultUnit)  {
267                 _out.append("    default true\n");
268             }
269             _out.append("    description \"")
270                             .append(StringUtil_mxJPO.convertTcl(unit.description)).append("\"\n")
271                     .append("    label \"").append(StringUtil_mxJPO.convertTcl(unit.label)).append("\"\n")
272                     .append("    multiplier ").append(Double.toString(unit.multiplier)).append("\n")
273                     .append("    offset ").append(Double.toString(unit.offset)).append("\n");
274             // system information
275             for (final Map.Entry<String, Set<String>> systemInfo : unit.systemInfos.entrySet())  {
276                 for (final String unitName : systemInfo.getValue())  {
277                     _out.append("    system \"").append(StringUtil_mxJPO.convertTcl(systemInfo.getKey()))
278                         .append("\" to unit \"").append(StringUtil_mxJPO.convertTcl(unitName))
279                         .append("\"\n");
280                 }
281             }
282             // settings
283             for (final Map.Entry<String, String> setting : unit.settings.entrySet())  {
284                 _out.append("    setting \"").append(StringUtil_mxJPO.convertTcl(setting.getKey()))
285                     .append("\" \"").append(StringUtil_mxJPO.convertTcl(setting.getValue()))
286                     .append("\"\n");
287             }
288             // properties
289             for (final AdminProperty_mxJPO prop : unit.properties)  {
290                 _out.append("    property \"").append(StringUtil_mxJPO.convertTcl(prop.getName()))
291                     .append("\"");
292                 if ((prop.getRefAdminName() != null) && (prop.getRefAdminType() != null))  {
293                     _out.append(" to ").append(prop.getRefAdminType())
294                         .append(" \"")
295                         .append(StringUtil_mxJPO.convertTcl(prop.getRefAdminName()))
296                         .append("\"");
297                 }
298                 if (prop.getValue() != null)  {
299                     _out.append(" value \"")
300                         .append(StringUtil_mxJPO.convertTcl(prop.getValue()))
301                         .append("\"");
302                 }
303                 _out.append('\n');
304             }
305             _out.append("  }\n");
306         }
307         _out.append("}");
308 
309         // append properties
310         this.writeProperties(_paramCache, _out);
311     }

363     /***
364      * The method is called within the update of an administration object. The
365      * method is called directly within the update.
366      * <ul>
367      * <li>All <code>@0@0@</code> are replaced by quot's and all
368      *     <code>@1@1@</code> are replaced by apostroph's.</li>
369      * <li>The new dimension definition is parsed.</li>
370      * <li>A delta MQL script generated to update the dimension to the new
371      *     target definition.</li>
372      * <li>The created delta MQL script is executed.</li>
373      * </ul>
374      *
375      * @param _paramCache   parameter cache
376      * @param _args         arguments from the TCL procedure
377      * @throws Exception if update failed, {@link UpdateException_mxJPO} if
378      *                   a unit could not be removed, a default unit is
379      *                   changed, a modified or offset of a unit is changed
380      * @see #TCL_PROCEDURE
381      */
382     @Override
383     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
384                                final String... _args)
385             throws Exception
386     {
387         if (!"dimension".equals(_args[0]))  {
388             super.jpoCallExecute(_paramCache, _args);
389         } else if (!this.getName().equals(_args[1])) {
390             throw new Exception("Dimension '" + _args[1]
391                     + "' is wanted to updated, but have dimension '" + _args[1] + "'!");
392         } else  {
393             final String code = _args[2].replaceAll("@0@0@", "'")
394                                         .replaceAll("@1@1@", "//\"");
395 
396             final DimensionDefParser_mxJPO parser = new DimensionDefParser_mxJPO(new StringReader(code));
397             final Dimension_mxJPO dimension = parser.dimension(_paramCache, this.getTypeDef(), this.getName());
398 
399             final StringBuilder cmd = new StringBuilder()
400                     .append("escape mod dimension \"")
401                     .append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ");
402 
403             // basic information
404             Dimension_mxJPO.calcValueDelta(cmd, "description", dimension.getDescription(), this.getDescription());
405             // hidden flag, because hidden flag must be set with special syntax
406             if (this.isHidden() != dimension.isHidden())  {
407                 if (!dimension.isHidden())  {
408                     cmd.append('!');
409                 }
410                 cmd.append("hidden ");
411             }
412 
413             // prepare maps of units depending on the unit name (as key)
414             final Map<String,Unit> curUnits = new HashMap<String,Unit>();
415             for (final Unit unit : this.units)  {
416                 curUnits.put(unit.name, unit);
417             }
418             final Map<String,Unit> tarUnits = new HashMap<String,Unit>();
419             for (final Unit unit : dimension.units)  {
420                 tarUnits.put(unit.name, unit);
421             }
422 
423             // remove units which not needed anymore (or throw exception)
424             for (final Map.Entry<String,Unit> curUnit : curUnits.entrySet())  {
425                 if (!tarUnits.containsKey(curUnit.getKey()))  {
426                     if (!_paramCache.getValueBoolean(Dimension_mxJPO.PARAM_ALLOW_REMOVE_UNIT))  {
427                         throw new UpdateException_mxJPO(UpdateException_mxJPO.Error.DIMENSION_UPDATE_REMOVEUNIT);
428                     }
429                     cmd.append("remove unit \"").append(StringUtil_mxJPO.convertMql(curUnit.getKey())).append("\" ");
430                 }
431             }
432 
433             final StringBuilder postCmd = new StringBuilder();
434             for (final Map.Entry<String,Unit> tarUnit : tarUnits.entrySet())  {
435                 if (!curUnits.containsKey(tarUnit.getKey()))  {
436                     _paramCache.logDebug("    add unit " + tarUnit.getKey());
437                     tarUnit.getValue().appendDelta(_paramCache, null, cmd, postCmd);
438                 }
439             }
440 
441             for (final Unit tarUnit : tarUnits.values())  {
442                 if (curUnits.containsKey(tarUnit.name))  {
443                     tarUnit.appendDelta(_paramCache, curUnits.get(tarUnit.name), cmd, postCmd);
444                 }
445             }
446 
447             cmd.append(postCmd).append(";");
448             MqlUtil_mxJPO.execMql(_paramCache, cmd);
449         }
450     }
451     /***
452      * Calculates the delta between the new and the old value. If a delta
453      * exists, the kind with the new delta is added to the string builder.
454      *
455      * @param _out      appendable instance where the delta must be append
456      * @param _kind     kind of the delta
457      * @param _newVal   new target value
458      * @param _curVal   current value in the database
459      */
460     protected static void calcValueDelta(final StringBuilder _out,
461                                          final String _kind,
462                                          final String _newVal,
463                                          final String _curVal)
464     {
465         final String curVal = (_curVal == null) ? "" : _curVal;
466         final String newVal = (_newVal == null) ? "" : _newVal;
467 
468         if (!curVal.equals(newVal))  {
469             _out.append(_kind).append(" \"").append(StringUtil_mxJPO.convertMql(newVal)).append("\" ");
470         }
471     }

535         private AdminProperty_mxJPO currentUnitProperty;

537         /***
538          * Holds all unit specific properties for a single unit. The properties

540          * dimension is parsed the properties holds also the {@link #settings},
541          * and {@link #systemInfos}.
542          *
543          * @see Dimension_mxJPO#parse(String, String)
544          * @see Dimension_mxJPO#prepare(ParameterCache_mxJPO)
545          */
546         private final Set<AdminProperty_mxJPO> properties = new TreeSet<AdminProperty_mxJPO>();
547 
548         /***
549          * Compares this dimension with <code>_toCompare</code> dimension. The
550          * algorithm checks {@link #name} for both dimension and is defined as:
551          * <ul>
552          * <li>If both {@link #name} are <code>null</code> then a &quot;0&quot;
553          *     is returned.</li>
554          * <li>If only the {@link #name} of this unit is <code>null</code> then
555          *     a &quot;-1&quot; is returned.</li>
556          * <li>If only the {@link #name} of the <code>_toCompare</code> unit is
557          *     <code>null</code> then a &quot;1&quot; is returned.</li>
558          * <li>Otherwise the value of a string compare between both
559          *     {@link #name}s is returned.</li>
560          * </ul>
561          *
562          * @param _toCompare    unit instance to which this unit must be
563          *                      compared to
564          * @return &quot;0&quot; if both units are equal; &quot;1&quot; if
565          *         greater; otherwise &quot;-1&quot;
566          */
567         public int compareTo(final Unit _toCompare)
568         {
569             return (this.name == null)
570                     ? (_toCompare.name == null)
571                         ? 0
572                         : 1
573                     : (_toCompare.name == null)
574                         ? -1
575                         : this.name.compareTo(_toCompare.name);
576         }

578         /***
579          * <p>Appends the MQL delta commands to updated current unit definition
580          * <code>_current</code> to this new target unit definition.
581          * <p> FYI: For each setting which is modified a <code>modify
582          * unit</code> must be defined as prefix or otherwise only the last
583          * setting is correct (issue in MX kernel).</p>
584          *
585          * @param _paramCache   parameter cache
586          * @param _current      current unit definition in MX (or
587          *                      <code>null</code> if currently not existing)
588          * @param _cmd          string builder used to append MQL commands
589          * @param _postCmd      string builder used to append the system
590          *                      information (should be defined at least)
591          * @throws UpdateException_mxJPO if the multiplier or offset is changed
592          */
593         protected void appendDelta(final ParameterCache_mxJPO _paramCache,
594                                    final Unit _current,
595                                    final StringBuilder _cmd,
596                                    final StringBuilder _postCmd)
597             throws UpdateException_mxJPO
598         {
599             final StringBuilder modUnitCmd = new StringBuilder()
