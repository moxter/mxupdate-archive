170     @Override
171     protected void writeEnd(final ParameterCache_mxJPO _paramCache,
172                             final Appendable _out)
173             throws IOException
174     {
175         _out.append("\n\ntestAttributes -").append(this.getTypeDef().getMxAdminName())
176             .append(" \"${NAME}\" -attributes [list //\n");
177         for (final String attr : this.attributes)  {
178             _out.append("    \"").append(StringUtil_mxJPO.convertTcl(attr)).append("\" //\n");
179         }
180         _out.append("]");

218     /***
219      * The method is called within the update of an administration object. The
220      * method is called directly within the update and checks which attributes
221      * are missed in the new definition and adds missing attributes to the
222      * administration object. If an attribute is not defined anymore but
223      * assigned in MX, an exception is thrown.
224      *
225      * @param _paramCache   parameter cache
226      * @param _args         arguments from the TCL procedure
227      * @throws Exception if an unknown parameter is defined, the given name of
228      *                   the administration object is not the same or an
229      *                   attribute is assigned to an administration object
230      *                   within MX but not defined anymore
231      */
232     @Override
233     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
234                                final String... _args)
235             throws Exception
236     {
237         int idx = 1;
238         String name = null;
239         String attrStr = null;
240 
241         final Set<String> ignoreAttrs = new HashSet<String>();
242         final Set<String> removeAttrs = new HashSet<String>();
243         while (idx < _args.length)  {
244             final String arg = _args[idx];
245             if ("-attributes".equals(arg))  {
246                 attrStr = _args[++idx];
247             } else if ("-ignoreattr".equals(arg))  {
248                 ignoreAttrs.add(_args[++idx]);
249             } else if ("-interface".equals(arg))  {
250                 name = _args[++idx];
251                 final Collection<String> tmp1
252                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_INTERFACE);
253                 if (tmp1 != null)  {
254                     ignoreAttrs.addAll(tmp1);
255                 }
256                 final Collection<String> tmp2
257                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_INTERFACE);
258                 if (tmp2 != null)  {
259                     removeAttrs.addAll(tmp2);
260                 }
261             } else if ("-relationship".equals(arg))  {
262                 name = _args[++idx];
263                 final Collection<String> tmp1
264                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_RELATIONSHIP);
265                 if (tmp1 != null)  {
266                     ignoreAttrs.addAll(tmp1);
267                 }
268                 final Collection<String> tmp2
269                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_RELATIONSHIP);
270                 if (tmp2 != null)  {
271                     removeAttrs.addAll(tmp2);
272                 }
273             } else if ("-removeattr".equals(arg))  {
274                 removeAttrs.add(_args[++idx]);
275             } else if ("-type".equals(arg))  {
276                 name = _args[++idx];
277                 final Collection<String> tmp1
278                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_TYPE);
279                 if (tmp1 != null)  {
280                     ignoreAttrs.addAll(tmp1);
281                 }
282                 final Collection<String> tmp2
283                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_TYPE);
284                 if (tmp2 != null)  {
285                     removeAttrs.addAll(tmp2);
286                 }
287             } else  {
288                 throw new Exception("unknown parameter \"" + arg + "\"");
289             }
290             idx++;
291         }
292 
293         // check for equal administration name
294         if (!this.getName().equals(name))  {
295             throw new Exception(this.getTypeDef().getLogging()
296                     + " '" + this.getName() + "' was called to"
297                     + " update via update script, but "
298                     + this.getTypeDef().getLogging() + " '" + name + "' was"
299                     + " called in the procedure...");
300         }
301 
302         // get all attributes
303 // TODO: if a { or } is in an attribute name, the list is not created correctly
304         final Pattern pattern = Pattern.compile("(//{[^//{//}]*//} )|([^ //{//}]* )");
305         final Matcher matcher = pattern.matcher(attrStr + " ");
306         final Set<String> newAttrs = new TreeSet<String>();
307         while (matcher.find())  {
308             final String attrName = matcher.group().trim().replaceAll("(^//{)|(//}$)", "");
309             if (!"".equals(attrName))  {
310                 newAttrs.add(attrName);
311             }
312         }
313 
314         // now check for not defined but existing attributes
315         for (final String attr : this.attributes)  {
316             if (!newAttrs.contains(attr))  {
317                 boolean ignore = false;
318                 for (final String ignoreAttr : ignoreAttrs)  {
319                     if (StringUtil_mxJPO.match(attr, ignoreAttr))  {
320                         ignore = true;
321                         _paramCache.logDebug("    - attribute '" + attr + "' is not defined but will be ignored");
322                         break;
323                     }
324                 }
325                 if (!ignore)  {
326                     boolean remove = false;
327                     for (final String removeAttr : removeAttrs)  {
328                         if (StringUtil_mxJPO.match(attr, removeAttr))  {
329                             remove = true;
330                             _paramCache.logDebug("    - attribute '" + attr + "' is not defined and will be removed");
331                             MqlUtil_mxJPO.execMql(_paramCache,
332                                     new StringBuilder()
333                                         .append("escape mod ").append(this.getTypeDef().getMxAdminName())
334                                         .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\"")
335                                         .append(" remove attribute \"")
336                                                 .append(StringUtil_mxJPO.convertMql(attr)).append("\""));
337                             break;
338                         }
339                     }
340                     if (!remove)  {
341                         throw new Exception("Attribute '" + attr + "' is defined to be deleted"
342                                 + " in " + this.getTypeDef().getLogging() + " '"
343                                 + this.getName() + "', but not allowed!");
344                     }
345                 }
346             }
347         }
348 
349         // and check for not existing attributes but needed
350         for (final String attr : newAttrs)  {
351             if (!this.attributes.contains(attr))  {
352                 _paramCache.logDebug("    - add attribute '" + attr + "'");
353                 MqlUtil_mxJPO.execMql(_paramCache,
354                         new StringBuilder()
355                             .append("escape mod ").append(this.getTypeDef().getMxAdminName())
356                             .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\"")
357                             .append(" add attribute \"").append(StringUtil_mxJPO.convertMql(attr)).append("\""));
358             }
359         }
360     }
