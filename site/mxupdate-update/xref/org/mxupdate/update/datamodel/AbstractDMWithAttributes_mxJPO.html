239     /***
240      * The method is called within the update of an administration object. The
241      * method is called directly within the update and checks which attributes
242      * are missed in the new definition and adds missing attributes to the
243      * administration object. If an attribute is not defined anymore but
244      * assigned in MX, an exception is thrown.
245      *
246      * @param _paramCache   parameter cache
247      * @param _args         arguments from the TCL procedure
248      * @throws Exception if an unknown parameter is defined, the given name of
249      *                   the administration object is not the same or an
250      *                   attribute is assigned to an administration object
251      *                   within MX but not defined anymore
252      */
253     @Override
254     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
255                                final String... _args)
256             throws Exception
257     {
258         int idx = 1;
259         String name = null;
260         String attrStr = null;
261 
262         final Set<String> ignoreAttrs = new HashSet<String>();
263         final Set<String> removeAttrs = new HashSet<String>();
264         while (idx < _args.length)  {
265             final String arg = _args[idx];
266             if ("-attributes".equals(arg))  {
267                 attrStr = _args[++idx];
268             } else if ("-ignoreattr".equals(arg))  {
269                 ignoreAttrs.add(this.jpoCallExecuteConvert(_args[++idx]));
270             } else if ("-interface".equals(arg))  {
271                 name = this.jpoCallExecuteConvert(_args[++idx]);
272                 final Collection<String> tmp1
273                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_INTERFACE);
274                 if (tmp1 != null)  {
275                     ignoreAttrs.addAll(tmp1);
276                 }
277                 final Collection<String> tmp2
278                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_INTERFACE);
279                 if (tmp2 != null)  {
280                     removeAttrs.addAll(tmp2);
281                 }
282             } else if ("-relationship".equals(arg))  {
283                 name = this.jpoCallExecuteConvert(_args[++idx]);
284                 final Collection<String> tmp1
285                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_RELATIONSHIP);
286                 if (tmp1 != null)  {
287                     ignoreAttrs.addAll(tmp1);
288                 }
289                 final Collection<String> tmp2
290                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_RELATIONSHIP);
291                 if (tmp2 != null)  {
292                     removeAttrs.addAll(tmp2);
293                 }
294             } else if ("-removeattr".equals(arg))  {
295                 removeAttrs.add(this.jpoCallExecuteConvert(_args[++idx]));
296             } else if ("-type".equals(arg))  {
297                 name = this.jpoCallExecuteConvert(_args[++idx]);
298                 final Collection<String> tmp1
299                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_TYPE);
300                 if (tmp1 != null)  {
301                     ignoreAttrs.addAll(tmp1);
302                 }
303                 final Collection<String> tmp2
304                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_TYPE);
305                 if (tmp2 != null)  {
306                     removeAttrs.addAll(tmp2);
307                 }
308             } else  {
309                 throw new Exception("unknown parameter \"" + arg + "\"");
310             }
311             idx++;
312         }
313 
314         // check for equal administration name
315         if (!this.getName().equals(name))  {
316             throw new Exception(this.getTypeDef().getLogging()
317                     + " '" + this.getName() + "' was called to"
318                     + " update via update script, but "
319                     + this.getTypeDef().getLogging() + " '" + name + "' was"
320                     + " called in the procedure...");
321         }
322 
323         // get all attributes
324 // TODO: if a { or } is in an attribute name, the list is not created correctly
325 /*        final Pattern pattern = Pattern.compile("(//{[^//{//}]*//} )|([^ //{//}]* )");
326         final Matcher matcher = pattern.matcher(attrStr + " ");
327 */
328         final Set<String> newAttrs = new TreeSet<String>();
329         for (final String attrName : attrStr.split(" "))  {
330             if (!"".equals(attrName))  {
331                 newAttrs.add(this.jpoCallExecuteConvert(attrName));
332             }
333         }
334 
335         // now check for not defined but existing attributes
336         for (final String attr : this.attributes)  {
337             if (!newAttrs.contains(attr))  {
338                 boolean ignore = false;
339                 for (final String ignoreAttr : ignoreAttrs)  {
340                     if (StringUtil_mxJPO.match(attr, ignoreAttr))  {
341                         ignore = true;
342                         _paramCache.logDebug("    - attribute '" + attr + "' is not defined but will be ignored");
343                         break;
344                     }
345                 }
346                 if (!ignore)  {
347                     boolean remove = false;
348                     for (final String removeAttr : removeAttrs)  {
349                         if (StringUtil_mxJPO.match(attr, removeAttr))  {
350                             remove = true;
351                             _paramCache.logDebug("    - attribute '" + attr + "' is not defined and will be removed");
352                             MqlUtil_mxJPO.execMql(_paramCache,
353                                     new StringBuilder()
354                                         .append("escape mod ").append(this.getTypeDef().getMxAdminName())
355                                         .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\"")
356                                         .append(" remove attribute \"")
357                                                 .append(StringUtil_mxJPO.convertMql(attr)).append("\""));
358                             break;
359                         }
360                     }
361                     if (!remove)  {
362                         throw new Exception("Attribute '" + attr + "' is defined to be deleted"
363                                 + " in " + this.getTypeDef().getLogging() + " '"
364                                 + this.getName() + "', but not allowed!");
365                     }
366                 }
367             }
368         }
369 
370         // and check for not existing attributes but needed
371         for (final String attr : newAttrs)  {
372             if (!this.attributes.contains(attr))  {
373                 _paramCache.logDebug("    - add attribute '" + attr + "'");
374                 MqlUtil_mxJPO.execMql(_paramCache,
375                         new StringBuilder()
376                             .append("escape mod ").append(this.getTypeDef().getMxAdminName())
377                             .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\"")
378                             .append(" add attribute \"").append(StringUtil_mxJPO.convertMql(attr)).append("\""));
379             }
380         }
381     }

383     /***
384      *
385      * @param _value    value to convert
386      * @return
387      * @see #TCL_PROCEDURE
388      */
389     protected String jpoCallExecuteConvert(final String _value)
390     {
391         return _value.replaceAll("@20", " ")
392                      .replaceAll("@22", "//\"")
393                      .replaceAll("@27", "'")
394                      .replaceAll("@7B", "{")
395                      .replaceAll("@7D", "}")
396                      .replaceAll("@@", "@");
397     }
