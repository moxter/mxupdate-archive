187     /***
188      * Searches for all attribute objects depending on the attribute type.
189      *
190      * @param _paramCache   parameter cache
191      * @return set of MX names of all attributes of attribute type
192      *         {@link #attrTypeList}
193      * @throws MatrixException if the query for attribute objects failed
194      * @see #SELECT_ATTRS
195      * @see #attrTypeList
196      */
197     @Override
198     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
199             throws MatrixException
200     {
201         final Set<String> ret = new TreeSet<String>();
202         final int length = this.attrTypeList.length();
203         for (final String name : MqlUtil_mxJPO.execMql(_paramCache,
204                                                        AbstractAttribute_mxJPO.SELECT_ATTRS).split("\n"))  {
205             if (!"".equals(name))  {
206                 if (name.startsWith(this.attrTypeList))  {
207                     ret.add(name.substring(length));
208                 }
209             }
210         }
211         return ret;
212     }

214     /***
215      * The method parses the attribute specific XML urls. This includes
216      * information about:
217      * <ul>
218      * <li>assigned rules (see {@link #rules})</li>
219      * <li>default value (see {@link #defaultValue})</li>
220      * <li>is the attribute multiline (see {@link #multiline})</li>
221      * <li>attribute type (see {@link #type})</li>
222      * <li>defined ranges (see {@link #ranges}, {@link #rangeProgramRef} and
223      *     {@link #rangeProgramInputArguments}</li>
224      * </ul>
225      *
226      * @param _url      URL to parse
227      * @param _content  content of the URL to parse
228      */
229     @Override
230     protected void parse(final String _url,
231                          final String _content)
232     {
233 
234         if ("/accessRuleRef".equals(_url))  {
235             this.rules.add(_content);
236         } else if ("/defaultValue".equals(_url))  {
237             this.defaultValue = _content;
238         } else if ("/dimensionRef".equals(_url))  {
239             this.dimension = _content;
240         } else if ("/multiline".equals(_url))  {
241             this.multiline = true;
242         } else if ("/primitiveType".equals(_url))  {
243             this.type = _content;
244 
245         } else if ("/rangeList".equals(_url))  {
246             // to be ignored ...
247         } else if ("/rangeList/range".equals(_url))  {
248             this.ranges.add(new Range());
249         } else if ("/rangeList/range/rangeType".equals(_url))  {
250             this.ranges.peek().type = _content;
251         } else if ("/rangeList/range/rangeValue".equals(_url))  {
252             this.ranges.peek().value1 = _content;
253         } else if ("/rangeList/range/includingValue".equals(_url))  {
254             this.ranges.peek().include1 = true;
255         } else if ("/rangeList/range/rangeSecondValue".equals(_url))  {
256             this.ranges.peek().value2 = _content;
257         } else if ("/rangeList/range/includingSecondValue".equals(_url))  {
258             this.ranges.peek().include2 = true;
259 
260         } else if ("/rangeProgram".equals(_url))  {
261             // to be ignored ...
262         } else if ("/rangeProgram/programRef".equals(_url))  {
263             this.rangeProgramRef = _content;
264         } else if ("/rangeProgram/inputArguments".equals(_url))  {
265             this.rangeProgramInputArguments = _content;
266 
267         } else  {
268             super.parse(_url, _content);
269         }
270     }

273      * The ranges in {@link #ranges} are sorted into {@link #rangesSorted}.

278     @Override
279     protected void prepare(final ParameterCache_mxJPO _paramCache)

282         for (final Range range : this.ranges)  {
283             this.rangesSorted.add(range);
284         }
285         super.prepare(_paramCache);
286     }

288     /***
289      * Writes specific information about the cached attribute to the given
290      * writer instance.
291      *
292      * @param _paramCache   parameter cache
293      * @param _out          appendable instance to the TCL update file
294      * @throws IOException if the TCL update code could not be written to the
295      *                     writer instance
296      */
297     @Override
298     protected void writeObject(final ParameterCache_mxJPO _paramCache,
299                                final Appendable _out)
300             throws IOException
301     {
302         _out.append(" //\n    ").append(this.isHidden() ? "hidden" : "!hidden");
303         if ("string".equalsIgnoreCase(this.type))  {
304             _out.append(" //\n    ").append(this.multiline ? "" : "!").append("multiline");
305         }
306         for (final String rule : this.rules)  {
307             _out.append(" //\n    add rule \"").append(StringUtil_mxJPO.convertTcl(rule)).append("\"");
308         }
309         if (this.defaultValue != null)  {
310             _out.append(" //\n    default \"").append(StringUtil_mxJPO.convertTcl(this.defaultValue)).append("\"");
311         } else  {
312             _out.append(" //\n    default \"\"");
313         }
314         // append triggers
315         this.writeTriggers(_out);
316         // append ranges
317         for (final Range range : this.rangesSorted)  {
318             range.write(_out);
319         }
320     }

365     /***
366      * The method overwrites the original method to append the MQL statements
367      * in the <code>_preMQLCode</code> to reset this attribute. This includes:
368      * <ul>
369      * <li>set to not hidden</li>
370      * <li>reset description and default value</li>
371      * <li>remove all ranges</li>
372      * <li>define the TCL procedure {@link #TCL_PROCEDURE} to set the
373      *     dimension</li>
374      * <li>if the dimension is not updated within the TCL update file (tested
375      *     with {@link #dimensionUpdated}) method
376      *     {@link #updateDimension(ParameterCache_mxJPO, String)} is called
377      *     </li>
378      * </ul>
379      *
380      * @param _paramCache       parameter cache
381      * @param _preMQLCode       MQL statements which must be called before the
382      *                          TCL code is executed
383      * @param _postMQLCode      MQL statements which must be called after the
384      *                          TCL code is executed
385      * @param _preTCLCode       TCL code which is defined before the source
386      *                          file is sourced
387      * @param _tclVariables     map of all TCL variables where the key is the
388      *                          name and the value is value of the TCL variable
389      *                          (the value is automatically converted to TCL
390      *                          syntax!)
391      * @param _sourceFile       souce file with the TCL code to update
392      * @throws Exception if the update from derived class failed
393      * @see Range#remove4Update(Appendable)
394      * @see #updateDimension(ParameterCache_mxJPO, String)
395      * @see #dimensionUpdated
396      * @see #TCL_PROCEDURE
397      */
398     @Override
399     protected void update(final ParameterCache_mxJPO _paramCache,
400                           final CharSequence _preMQLCode,
401                           final CharSequence _postMQLCode,
402                           final CharSequence _preTCLCode,
403                           final Map<String,String> _tclVariables,
404                           final File _sourceFile)
405             throws Exception
406     {
407         // remove all properties
408         final StringBuilder preMQLCode = new StringBuilder()
409                 .append("mod ").append(this.getTypeDef().getMxAdminName())
410                 .append(" \"").append(this.getName()).append('\"')
411                 .append(" !hidden description \"\" default \"\"");
412         // remove rules
413         for (final String rule : this.rules)  {
414             preMQLCode.append(" remove rule \"").append(rule).append('\"');
415         }
416         // remove ranges
417         for (final Range range : this.rangesSorted)  {
418             range.remove4Update(preMQLCode);
419         }
420         // append already existing pre MQL code
421         preMQLCode.append(";\n")
422                   .append(_preMQLCode);
423         // add TCL code for the procedure
424         final StringBuilder preTCLCode = new StringBuilder()
425                 .append(AbstractAttribute_mxJPO.TCL_PROCEDURE)
426                 .append(_preTCLCode);
427 
428         super.update(_paramCache, preMQLCode, _postMQLCode, preTCLCode, _tclVariables, _sourceFile);
429 
430         // is the dimension already updated?
431         if (!this.dimensionUpdated)  {
432             this.updateDimension(_paramCache, "");
433         }
434     }

436     /***
437      * The method is called from the TCL update code to define the dimension
438      * for this attribute. If the correct use case is defined method
439      * {@link #updateDimension(ParameterCache_mxJPO, String)} is called.
440      *
441      * @param _paramCache   parameter cache
442      * @param _args         first index defines the use case (must be
443      *                      &quot;defineAttrDimension&quot; that the dimension
444      *                      of the attribute is updated); second index the name
445      *                      of the attribute to update; third index the new
446      *                      dimension to set
447      * @throws Exception if the update of the dimension failed or for all other
448      *                   use cases from super JPO call
449      * @see #updateDimension(ParameterCache_mxJPO, String)
450      * @see #dimensionUpdated
451      */
452     @Override
453     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
454                                final String... _args)
455             throws Exception
456     {
457         // check if dimension is defined
458         if ("defineAttrDimension".equals(_args[0]))  {
459             // check that attribute names are equal
460             if (!this.getName().equals(_args[1]))  {
461                 throw new Exception("dimension for wrong attribute '"
462                         + _args[1] + "' is set (attribute '" + this.getName()
463                         + "' is updated!)");
464             }
465 
466             this.updateDimension(_paramCache, _args[2]);
467             this.dimensionUpdated = true;
468         } else  {
469             super.jpoCallExecute(_paramCache, _args);
470         }
471     }
472 
473     /***
474      * Updates the dimension of this attribute. If a dimension is already
475      * defined for this attribute and it is not the same attribute, an
476      * exception is thrown (because information could be lost).
477      *
478      * @param _paramCache   parameter cache
479      * @param _dimension    new dimension to set
480      * @throws Exception if update failed or the dimension of an update should
481      *                   be changed (and some information could be lost)
482      * @see #jpoCallExecute(ParameterCache_mxJPO, String...)
483      * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
484      */
485     protected void updateDimension(final ParameterCache_mxJPO _paramCache,
486                                    final String _dimension)
487             throws Exception
488     {
489         if ((this.dimension == null) || "".equals(this.dimension))  {
490             if (!"".equals(_dimension))  {
491                 _paramCache.logDebug("    - set dimension '" + _dimension + "'");
492                 MqlUtil_mxJPO.execMql(_paramCache,
493                         new StringBuilder().append("escape mod attribute \"")
494                         .append(StringUtil_mxJPO.convertMql(this.getName()))
495                         .append("\" add dimension \"")
496                         .append(StringUtil_mxJPO.convertMql(_dimension))
497                         .append('\"'));
498             }
499         } else if (!this.dimension.equals(_dimension))  {
500             throw new Exception("If dimension '" + this.dimension
501                     + "' is changed to new dimension '" + _dimension
502                     + "' some information could be lost!");
503         }
504     }

506     /***
507      * Class holding range values of this attribute.
508      */
509     private class Range
510             implements Comparable<AbstractAttribute_mxJPO.Range>
511     {
512         /***
513          * Holds the range type. Typically following range types are known:
514          * <ul>
515          * <li>between<li>
516          * <li>equal<li>
517          * <li>greaterthan<li>
518          * <li>greaterthanequal<li>
519          * <li>lessthan<li>
520          * <li>lessthanequal<li>
521          * <li>match<li>
522          * <li>notequal<li>
523          * <li>notmatch<li>
524          * <li>notsmatch<li>
525          * <li>smatch<li>
526          * </ul>
527          */
528         String type = null;
529 
530         /***
531          * Hold the first range value.
532          */
533         String value1 = null;
534 
535         /***
536          * Holds the second range value.
537          */
538         String value2 = null;
539 
540         /***
541          * Include first value (used for range type &quot;between&quot;).
542          */
543         Boolean include1 = false;
544 
545         /***
546          * Include second value (used for range type &quot;between&quot;).
547          */
548         Boolean include2 = false;
549 
550         /***
551          * Write this range value to the writer instance.
552          * Following range types are converted:
553          * <ul>
554          * <li>&quot;equal&quot; to &quot;=&quot;</li>
555          * <li>&quot;greaterthan&quot; to &quot;&gt;&quot;</li>
556          * <li>&quot;greaterthanequal&quot; to &quot;&gt;=&quot;</li>
557          * <li>&quot;lessthan&quot; to &quot;&lt;&quot;</li>
558          * <li>&quot;lessthanequal&quot; to &quot;&lt;=&quot;</li>
559          * <li>&quot;notequal&quot; to &quot;!=&quot;</li>
560          * <li>&quot;notmatch&quot; to &quot;!match&quot;</li>
561          * <li>&quot;notsmatch&quot; to &quot;!smatch&quot;</li>
562          * </ul>
563          * If the range type is a program, the name of the program and the
564          * input arguments are defined directly on the attribute in
565          * {@link AbstractAttribute_mxJPO#rangeProgramRef} and
566          * {@link AbstractAttribute_mxJPO#rangeProgramInputArguments}.
567          *
568          * @param _out  writer instance
569          * @throws IOException if write to the writer instance is not possible
570          */
571         private void write(final Appendable _out)
572                 throws IOException
573         {
574             _out.append(" //\n    add range ");
575             // if the range is a program it is a 'global' attribute info
576             if ("programRange".equals(this.type))  {
577                 _out.append("program \"")
578                     .append(StringUtil_mxJPO.convertTcl(AbstractAttribute_mxJPO.this.rangeProgramRef))
579                     .append('\"');
580                 if (AbstractAttribute_mxJPO.this.rangeProgramInputArguments != null)  {
581                     _out.append(" input \"")
582                         .append(StringUtil_mxJPO.convertTcl(AbstractAttribute_mxJPO.this.rangeProgramInputArguments))
583                         .append('\"');
584                 }
585             } else  {
586                 if ("equal".equals(this.type))  {
587                     _out.append('=');
588                 } else if ("greaterthan".equals(this.type))  {
589                     _out.append(">");
590                 } else if ("greaterthanequal".equals(this.type))  {
591                     _out.append(">=");
592                 } else if ("lessthan".equals(this.type))  {
593                     _out.append("<");
594                 } else if ("lessthanequal".equals(this.type))  {
595                     _out.append("<=");
596                 } else if ("notequal".equals(this.type))  {
597                     _out.append("!=");
598                 } else if ("notmatch".equals(this.type))  {
599                     _out.append("!match");
600                 } else if ("notsmatch".equals(this.type))  {
601                     _out.append("!smatch");
602                 } else  {
603                     _out.append(this.type);
604                 }
605                 _out.append(" \"").append(StringUtil_mxJPO.convertTcl(this.value1)).append("\"");
606                 if ("between".equals(this.type))  {
607                     if (this.include1)  {
608                         _out.append(" inclusive");
609                     } else  {
610                         _out.append(" exclusive");
611                     }
612                     _out.append(" \"").append(StringUtil_mxJPO.convertTcl(this.value2)).append("\"");
613                     if (this.include2)  {
614                         _out.append(" inclusive");
615                     } else  {
616                         _out.append(" exclusive");
617                     }
618                 }
619             }
620         }
621 
622         /***
623          * Appends the MQL code to remove this range depending on the type of
624          * range in {@link #type}. If it is a program the related name of
625          * program is defined as attribute variable in
626          * {@link AbstractAttribute_mxJPO#rangeProgramRef}.
627          *
628          * @param _out      appendable instance where the MQL code is appended
629          *                  to remove this range
630          * @throws IOException if the MQL code could not appended
631          */
632         private void remove4Update(final Appendable _out)
633                 throws IOException
634         {
635             _out.append(" remove range ");
636 
637             if ("programRange".equals(this.type))  {
638                 _out.append("program \"").append(AbstractAttribute_mxJPO.this.rangeProgramRef).append('\"');
639             } else  {
640                 _out.append(this.type)
641                     .append(" \"").append(this.value1).append('\"');
642                 if ("between".equals(this.type))  {
643                     if (this.include1)  {
644                         _out.append(" inclusive");
645                     } else  {
646                         _out.append(" exclusive");
647                     }
648                     _out.append(" \"").append(this.value2).append("\"");
649                     if (this.include2)  {
650                         _out.append(" inclusive");
651                     } else  {
652                         _out.append(" exclusive");
