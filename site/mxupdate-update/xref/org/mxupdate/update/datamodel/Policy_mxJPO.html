57      /***
58       * Called TCL procedure within the TCL update to parse the new policy
59       * definition. The TCL procedure calls method
60       * {@link #jpoCallExecute(ParameterCache_mxJPO, String...)} with the new
61       * policy definition. All quot's are replaced by <code>@0@0@</code> and all
62       * apostroph's are replaced by <code>@1@1@</code>.
63       *
64       * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
65       * @see #jpoCallExecute(ParameterCache_mxJPO, String...)
66       */
67      private static final String TCL_PROCEDURE
68              = "proc updatePolicy {_sPolicy _lsArgs}  {\n"
69                  + "regsub -all {'} $_lsArgs {@0@0@} sArg\n"
70                  + "regsub -all {//\"} $sArg {@1@1@} sArg\n"
71                  + "regsub -all {//////[} $sArg {[} sArg\n"
72                  + "regsub -all {//////]} $sArg {]} sArg\n"
73                  + "mql exec prog org.mxupdate.update.util.JPOCaller policy $_sPolicy \"${sArg}\"\n"
74              + "}\n";

273     /***
274      * Calls the state prepare method. All policies properties starting with
275      * <code>state_</code> are checked if they are defining a symbolic name of
276      * a state. If yes the related symbolic name of the state is updated and
277      * removed from the property list.
278      *
279      * @param _paramCache   parameter cache
280      * @throws MatrixException if the prepare within super class failed
281      */
282     @Override
283     protected void prepare(final ParameterCache_mxJPO _paramCache)
284             throws MatrixException
285     {
286         for (final State state : this.states)  {
287             state.prepare();
288         }
289         super.prepare(_paramCache);
290         for (final AdminProperty_mxJPO property : new HashSet<AdminProperty_mxJPO>(this.getPropertiesMap().values()))  {
291             if ((property.getName() != null) && property.getName().startsWith("state_"))  {
292                 for (final State state : this.states)  {
293                     if (state.name.equals(property.getValue()))  {
294                         state.symbolicNames.add(property.getName());
295                         this.getPropertiesMap().remove(property.getName());
296                     }
297                 }
298             }
299         }
300     }

302     /***
303      * Writes the update script for this policy.
304      *
305      * @param _paramCache   parameter cache
306      * @param _out          writer instance
307      * @throws IOException if the TCL update code could not be written
308      */
309     @Override
310     protected void write(final ParameterCache_mxJPO _paramCache,
311                          final Appendable _out)
312             throws IOException
313     {
314         this.writeHeader(_paramCache, _out);
315         _out.append("updatePolicy \"${NAME}\" {");
316         final String suffix = this.getTypeDef().getMxAdminSuffix();
317         if (!"".equals(suffix))  {
318             _out.append(" ").append(suffix);
319         }
320         _out.append("\n  description \"").append(StringUtil_mxJPO.convertTcl(this.getDescription())).append("\"");
321         // types
322         if (this.allTypes)  {
323             _out.append("\n  type all");
324         } else  {
325             _out.append("\n  type {")
326                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.types, null))
327                 .append("}");
328         }
329         // formats
330         if (this.allFormats)  {
331             _out.append("\n  format all");
332         } else  {
333             _out.append("\n  format {")
334                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.formats, null))
335                 .append("}");
336         }
337         _out.append("\n  defaultformat \"").append(StringUtil_mxJPO.convertTcl(this.defaultFormat)).append('\"')
338             .append("\n  sequence \"").append(StringUtil_mxJPO.convertTcl(this.sequence)).append('\"')
339             .append("\n  store \"").append(StringUtil_mxJPO.convertTcl(this.store)).append('\"')
340             .append("\n  hidden \"").append(Boolean.toString(this.isHidden())).append("\"");
341         // all states
342         for (final State state : this.states)  {
343             state.writeObject(_out);
344         }
345         _out.append("\n}");
346         this.writeProperties(_paramCache, _out);
347     }

399     /***
400      * The method is called within the update of an administration object. The
401      * method is called directly within the update.
402      * <ul>
403      * <li>All <code>@0@0@</code> are replaced by quot's and all
404      *     <code>@1@1@</code> are replaced by apostroph's.</li>
405      * <li>The new policy definition is parsed.</li>
406      * <li>A delta MQL script generated to update the policy to the new target
407      *     definition.</li>
408      * <li>All symbolic names for states are defined (as property on the
409      *     policy).</li>
410      * <li>The delta MQL script is executed.</li>
411      * </ul>
412      *
413      * @param _paramCache   parameter cache
414      * @param _args         arguments from the TCL procedure
415      * @throws Exception if a state is not defined anymore or the policy could
416      *                   not be updated
417      * @see #TCL_PROCEDURE
418      */
419     @Override
420     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
421                                final String... _args)
422             throws Exception
423     {
424         final String code = _args[2].replaceAll("@0@0@", "'")
425                                     .replaceAll("@1@1@", "//\"");
426 
427         final PolicyDefParser_mxJPO parser = new PolicyDefParser_mxJPO(new StringReader(code));
428         final Policy_mxJPO policy = parser.policy(_paramCache, this.getTypeDef(), _args[1]);
429 
430         final StringBuilder cmd = new StringBuilder()
431                 .append("mod policy \"").append(this.getName()).append("\" ");
432 
433         // basic information
434         this.calcValueDelta(cmd, "description", policy.getDescription(), this.getDescription());
435         this.calcListDelta(cmd, "type", policy.types, this.types);
436         this.calcListDelta(cmd, "format", policy.formats, this.formats);
437         this.calcValueDelta(cmd, "defaultformat", policy.defaultFormat, this.defaultFormat);
438         this.calcValueDelta(cmd, "sequence", policy.sequence, this.sequence);
439         // hidden flag, because hidden flag must be set with special syntax
440         if (this.isHidden() != policy.isHidden())  {
441             if (!policy.isHidden())  {
442                 cmd.append('!');
443             }
444             cmd.append("hidden ");
445         }
446         // because the store of a policy could not be removed....
447         if ((policy.store != null) && !"".equals(policy.store))  {
448             this.calcValueDelta(cmd, "store", policy.store, this.store);
449         }
450 
451         // states....
452         // (first add new states because of references in branches)
453         final Iterator<State> curStateIter = this.states.iterator();
454         final Iterator<State> newStateIter = policy.states.iterator();
455         final Map<State,State> stateDeltaMap = new HashMap<State,State>();
456         while (curStateIter.hasNext() && newStateIter.hasNext())  {
457             final State curState = curStateIter.next();
458             State newState = newStateIter.next();
459             while (!curState.name.equals(newState.name) && newStateIter.hasNext())  {
460                 cmd.append("add state \"").append(StringUtil_mxJPO.convertMql(newState.name))
461                    .append("\" before \"").append(StringUtil_mxJPO.convertMql(curState.name)).append("\" ");
462                 _paramCache.logDebug("    - insert new state '" + newState.name + "' before '" + curState.name + "'");
463                 stateDeltaMap.put(newState, null);
464                 newState = newStateIter.next();
465             }
466             if (curState.name.equals(newState.name))  {
467                 stateDeltaMap.put(newState, curState);
468             }
469         }
470         while (newStateIter.hasNext())  {
471             final State newState = newStateIter.next();
472             cmd.append("add state \"").append(StringUtil_mxJPO.convertMql(newState.name)).append("\" ");
473             _paramCache.logDebug("    - add new state '" + newState.name + "'");
474             stateDeltaMap.put(newState, null);
475         }
476         // check for already existing state, but not defined anymore!
477         if (curStateIter.hasNext())  {
478 throw new Exception("some states are not defined anymore!");
479         }
480 
481         // now update state information itself
482         cmd.append(';')
483            .append("mod policy \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ");
484         for (final Map.Entry<State, State> entry : stateDeltaMap.entrySet())  {
485             cmd.append("state \"").append(StringUtil_mxJPO.convertMql(entry.getKey().name)).append("\" ");
486             entry.getKey().calcDelta(cmd, entry.getValue());
487         }
488 
489         // set symbolic names for all policy states
490         cmd.append(';')
491            .append("mod policy \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ");
492         for (final State state : policy.states)  {
493             for (final String symbolicName : state.symbolicNames)  {
494                 cmd.append(" add property \"").append(StringUtil_mxJPO.convertMql(symbolicName))
495                    .append("\" value \"").append(StringUtil_mxJPO.convertMql(state.name)).append('\"');
496             }
497         }
498 
499         final boolean isMqlEscapeOn = MqlUtil_mxJPO.isEscapeOn(_paramCache);
500         try  {
501             MqlUtil_mxJPO.setEscapeOn(_paramCache);
502             MqlUtil_mxJPO.execMql(_paramCache, cmd);
503         } finally  {
504             if (!isMqlEscapeOn)  {
505                 MqlUtil_mxJPO.setEscapeOff(_paramCache);
506             }
507         }
508     }

510     /***
511      * Calculates the delta between the new and the old value. If a delta
512      * exists, the kind with the new delta is added to the string builder.
513      *
514      * @param _out      appendable instance where the delta must be append
515      * @param _kind     kind of the delta
516      * @param _newVal   new target value
517      * @param _curVal   current value in the database
518      * @throws IOException if the delta could not appended
519      */
520     protected void calcValueDelta(final Appendable _out,
521                                   final String _kind,
522                                   final String _newVal,
523                                   final String _curVal)
524             throws IOException
525     {
526         final String curVal = (_curVal == null) ? "" : _curVal;
527         final String newVal = (_newVal == null) ? "" : _newVal;
528 
529         if (!curVal.equals(newVal))  {
530             _out.append(_kind).append(" \"").append(StringUtil_mxJPO.convertMql(newVal)).append("\" ");
531         }
532     }

534     /***
535      *
536      * @param _out      appendable instance where the delta must be append
537      * @param _kind     kind of the delta
538      * @param _new      new target values
539      * @param _current  current values in MX
540      * @throws IOException if the delta could not appended
541      */
542     protected void calcListDelta(final Appendable _out,
543                                  final String _kind,
544                                  final Set<String> _new,
545                                  final Set<String> _current)
546             throws IOException
547     {
548         boolean equal = (_current.size() == _new.size());
549         if (equal)  {
550             for (final String format : _current)  {
551                 if (!_new.contains(format))  {
552                     equal = false;
553                     break;
554                 }
555             }
556         }
557         if (!equal)  {
558             for (final String format : _current)  {
559                 if (!_new.contains(format))  {
560                     _out.append("remove ").append(_kind).append(" \"").append(format).append("\" ");
561                 }
562             }
563             for (final String format : _new)  {
564                 if (!_current.contains(format))  {
565                     _out.append("add ").append(_kind).append(" \"").append(format).append("\" ");
566                 }
567             }
568         }
569     }

774         /***
775          *
776          * @param _out      appendable instance where the delta must be append
777          * @param _oldState old state to compare to (or <code>null</code> if
778          *                  this state is new)
779          * @throws IOException if the delta could not appended
780          */
781         protected void calcDelta(final Appendable _out,
782                                  final State _oldState)
783                 throws IOException
784         {
785             // basics
786             _out.append("promote ").append(String.valueOf(this.autoPromotion)).append(' ')
787                 .append("revision ").append(String.valueOf(this.revisionable)).append(' ')
788                 .append("checkouthistory ").append(String.valueOf(this.checkoutHistory)).append(' ')
789                 .append("version ").append(String.valueOf(this.versionable)).append(' ')
790                 .append("action \"").append(StringUtil_mxJPO.convertMql(this.actionProgram)).append("\" ")
791                 .append("input \"").append(StringUtil_mxJPO.convertMql(this.actionInput)).append("\" ")
792                 .append("check \"").append(StringUtil_mxJPO.convertMql(this.checkProgram)).append("\" ")
793                 .append("input \"").append(StringUtil_mxJPO.convertMql(this.checkInput)).append("\" ");
794             // route message
795             _out.append("route message \"").append(StringUtil_mxJPO.convertMql(this.routeMessage)).append("\" ");
796             for (final String routeUser : this.routeUsers)  {
797                 if ((_oldState == null) || !_oldState.routeUsers.contains(routeUser))  {
798                     _out.append("add route \"").append(StringUtil_mxJPO.convertMql(routeUser)).append("\" ");
799                 }
800             }
801             // owner access
802             _out.append("owner ")
803                 .append(StringUtil_mxJPO.joinMql(',', false, this.ownerAccess, "none"))
804                 .append(' ')
805             // public access
806                 .append("public ")
807                 .append(StringUtil_mxJPO.joinMql(',', false, this.publicAccess, "none"))
808                 .append(' ');
809             // user access
810             final Set<String> newUsers = new HashSet<String>();
811             for (final UserAccess userAccess : this.userAccess)  {
812                 newUsers.add(userAccess.userRef);
813             }
814             final Set<String> oldUser = new HashSet<String>();
815             if (_oldState != null)  {
816                 for (final UserAccess userAccess : _oldState.userAccess)  {
817                     if (newUsers.contains(userAccess.userRef))  {
818                         oldUser.add(userAccess.userRef);
819                     } else  {
820                         _out.append("remove user \"")
821                             .append(StringUtil_mxJPO.convertMql(userAccess.userRef))
822                             .append("\" all ");
823                     }
824                 }
825             }
826             for (final UserAccess userAccess : this.userAccess)  {
827                 if ((_oldState != null) && !oldUser.contains(userAccess.userRef))  {
828                     _out.append("add ");
829                 }
830                 _out.append("user \"").append(userAccess.userRef).append("\" ")
831                     .append(StringUtil_mxJPO.joinMql(',', false, userAccess.access, "none"))
832                     .append(" ")
833                     .append("filter \"")
834                     .append(StringUtil_mxJPO.convertMql(userAccess.expressionFilter))
835                     .append("\" ");
836             }
837             // triggers
838             if (_oldState != null)  {
839                 for (final Trigger trigger : _oldState.triggers.values())  {
840                     if (!this.triggers.containsKey(trigger.name))  {
841                         _out.append("remove trigger ")
842                             .append(trigger.getEventType())
843                             .append(' ')
844                             .append(trigger.getKind())
845                             .append(' ');
846                     }
847                 }
848             }
849             for (final Trigger trigger : this.triggers.values())  {
850                 _out.append("add trigger ").append(trigger.getEventType()).append(' ').append(trigger.getKind())
851                     .append(" \"").append(StringUtil_mxJPO.convertMql(trigger.program)).append("\"")
852                     .append(" input \"").append(StringUtil_mxJPO.convertMql(trigger.arguments)).append("\" ");
853             }
854             // signatures
855             final Set<String> newSigs = new HashSet<String>();
856             for (final Signature signature : this.signatures)  {
857                 newSigs.add(signature.name);
858             }
859             final Map<String,Signature> oldSigs = new HashMap<String,Signature>();
860             if (_oldState != null)  {
861                 for (final Signature signature : _oldState.signatures)  {
862                     if (newSigs.contains(signature.name))  {
863                         oldSigs.put(signature.name, signature);
864                     } else  {
865                         _out.append("remove signature \"")
866                             .append(StringUtil_mxJPO.convertMql(signature.name))
867                             .append("\" ");
868                     }
869                 }
870             }
871             for (final Signature signature : this.signatures)  {
872                 final Signature oldSig;
873                 if ((_oldState != null) && !oldSigs.containsKey(signature.name))  {
874                     _out.append("add ");
875                     oldSig = null;
876                 } else  {
877                     oldSig = oldSigs.get(signature.name);
878                 }
879                 _out.append("signature \"").append(StringUtil_mxJPO.convertMql(signature.name)).append("\" ");
880                 signature.calcDelta(_out, oldSig);
881             }
882         }
