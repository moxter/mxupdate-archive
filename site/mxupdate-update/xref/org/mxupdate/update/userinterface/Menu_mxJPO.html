49      /***
50       * Flag to store the information that the menu is a tree menu.
51       *
52       * @see #prepare(ParameterCache_mxJPO)
53       * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
54       * @see #writeEnd(ParameterCache_mxJPO, Appendable)

56      private boolean treeMenu = false;

58      /***
59       * Holds all children of this menu instance.
60       *
61       * @see MenuChild

63      private final Stack<MenuChild> childs = new Stack<MenuChild>();

66       * Constructor used to initialize the type definition enumeration.

69       * @param _mxName   MX name of the administration object

71      public Menu_mxJPO(final TypeDef_mxJPO _typeDef,

74          super(_typeDef, _mxName);

77      /***
78       * Parses all menu related URL's. This means the names of all referenced
79       * children menus and commands are parsed and stored in {@link #childs}.
80       *
81       * @param _url      URL to parse
82       * @param _content  related content of the URL
83       */
84      @Override
85      protected void parse(final String _url,
86                           final String _content)
87      {
88          if ("/commandRefList".equals(_url))  {
89              // to be ignored ...
90          } else if ("/commandRefList/commandRef".equals(_url))  {
91              final MenuChild child = new MenuChild();
92              child.type = "command";
93              this.childs.add(child);
94          } else if ("/commandRefList/commandRef/name".equals(_url))  {
95              this.childs.peek().name = _content;
96          } else if ("/commandRefList/commandRef/order".equals(_url))  {
97              this.childs.peek().order = Integer.parseInt(_content);
98  
99          } else if ("/menuRefList".equals(_url))  {
100             // to be ignored ...
101         } else if ("/menuRefList/menuRef".equals(_url))  {
102             final MenuChild child = new MenuChild();
103             child.type = "menu";
104             this.childs.add(child);
105         } else if ("/menuRefList/menuRef/name".equals(_url))  {
106             this.childs.peek().name = _content;
107         } else if ("/menuRefList/menuRef/order".equals(_url))  {
108             this.childs.peek().order = Integer.parseInt(_content);
109 
110         } else  {
111             super.parse(_url, _content);
112         }
113     }

115     /***
116      * @param _paramCache   parameter cache
117      * @throws MatrixException if the preparation from derived class failed or
118      *                         the check for tree menu used in
119      *                         {@link #treeMenu} failed
120      * @see #treeMenu
121      */
122     @Override
123     protected void prepare(final ParameterCache_mxJPO _paramCache)
124             throws MatrixException
125     {
126         final StringBuilder cmd = new StringBuilder()
127                 .append("print menu \"").append(this.getName()).append("\" select parent[Tree] dump");
128         if ("TRUE".equalsIgnoreCase(MqlUtil_mxJPO.execMql(_paramCache.getContext(), cmd)))  {
129             this.treeMenu = true;
130         }
131         super.prepare(_paramCache);
132     }

134     /***
135      * @param _paramCache   parameter cache
136      * @param _out          appendable instance to the TCL update file
137      * @throws IOException if the TCL update code could not be written
138      */
139     @Override
140     protected void writeObject(final ParameterCache_mxJPO _paramCache,
141                                final Appendable _out)
142             throws IOException
143     {
144         super.writeObject(_paramCache, _out);
145 
146         // order childs
147         final Map<Integer,MenuChild> tmpChilds = new TreeMap<Integer,MenuChild>();
148         for (final MenuChild child : this.childs)  {
149             tmpChilds.put(child.order, child);
150         }
151 
152         // output childs
153         for (final MenuChild child : tmpChilds.values())  {
154             _out.append(" //\n    add ").append(child.type).append(" \"")
155                 .append(StringUtil_mxJPO.convertTcl(child.name)).append("\"");
156         }
157 
158     }

160     /***
161      * @param _paramCache   parameter cache
162      * @param _out          appendable instance to the TCL update file
163      * @throws IOException if the TCL update code could not be written
164      */
165     @Override
166     protected void writeEnd(final ParameterCache_mxJPO _paramCache,
167                             final Appendable _out)
168             throws IOException
169     {
170         if (this.treeMenu)  {
171             _out.append("\n\nmql mod menu \"Tree\" //\n    add menu \"${NAME}\"");
172         }
173     }

175     /***
176      * The method overwrites the original method to append the MQL statements
177      * in the <code>_preMQLCode</code> to reset this menu. Following steps are
178      * done:
179      * <ul>
180      * <li>remove all child commands / menus</li>
181      * <li>remove definition as tree menu</li>
182      * </ul>
183      * The description etc. is reseted in the command super class.
184      *
185      * @param _paramCache       parameter cache
186      * @param _preMQLCode       MQL statements which must be called before the
187      *                          TCL code is executed
188      * @param _postMQLCode      MQL statements which must be called after the
189      *                          TCL code is executed
190      * @param _preTCLCode       TCL code which is defined before the source
191      *                          file is sourced
192      * @param _tclVariables     map of all TCL variables where the key is the
193      *                          name and the value is value of the TCL variable
194      *                          (the value is automatically converted to TCL
195      *                          syntax!)
196      * @param _sourceFile       souce file with the TCL code to update
197      * @throws Exception if the update from derived class failed
198      */
199     @Override
200     protected void update(final ParameterCache_mxJPO _paramCache,
201                           final CharSequence _preMQLCode,
202                           final CharSequence _postMQLCode,
203                           final CharSequence _preTCLCode,
204                           final Map<String,String> _tclVariables,
205                           final File _sourceFile)
206             throws Exception
207     {
208         // remove child commands / menus
209         final StringBuilder preMQLCode = new StringBuilder()
210                 .append("mod ").append(this.getTypeDef().getMxAdminName())
211                         .append(" \"").append(this.getName()).append('\"');
212         for (final MenuChild child : this.childs)  {
213             preMQLCode.append(" remove ").append(child.type)
214                       .append(" \"").append(child.name).append("\"");
215         }
216         preMQLCode.append(";\n");
217 
218         // remove information about tree menu...
219         if (this.treeMenu)  {
220             preMQLCode.append("mod menu Tree remove menu \"").append(this.getName()).append("\";\n");
221         }
222 
223         // append already existing pre MQL code
224         preMQLCode.append(_preMQLCode);
225 
226         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
227     }
