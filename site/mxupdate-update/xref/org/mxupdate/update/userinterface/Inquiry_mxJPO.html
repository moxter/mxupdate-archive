36  /***
37   * The class is used to export and import / update inquiry configuration items.
38   *
39   * @author The MxUpdate Team

42  public class Inquiry_mxJPO
43      extends AbstractAdminObject_mxJPO
44  {
45      /***
46       * Defines the parameter for the comment in front of the separator between
47       * TCL update code and the inquiry code.
48       *
49       * @see #writeEnd(ParameterCache_mxJPO, Appendable)
50       */
51      private static final String PARAM_SEPARATOR_COMMENT = "UIInquirySeparatorComment";
52  
53      /***
54       * Defines the parameter for the separator between TCL update code and the
55       * inquiry.
56       *
57       * @see #writeEnd(ParameterCache_mxJPO, Appendable)
58       * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
59       */
60      private static final String PARAM_SEPARATOR_TEXT = "UIInquirySeparatorText";
61

63       * Code for the inquiry.
64       *
65       * @see #parse(String, String)
66       * @see #writeEnd(ParameterCache_mxJPO, Appendable)

71       * Format for the inquiry.

98      /***
99       * Parses the code, format and pattern of an inquiry.
100      *
101      * @param _url      URL to parse
102      * @param _content  related content of the URL to parse
103      * @see #code
104      * @see #format
105      * @see #pattern
106      */
107     @Override()
108     protected void parse(final String _url,
109                          final String _content)
110     {
111         if ("/code".equals(_url))  {
112             this.code = _content;
113         } else if ("/fmt".equals(_url))  {
114             this.format = _content;
115         } else if ("/pattern".equals(_url))  {
116             this.pattern = _content;
117         } else  {
118             super.parse(_url, _content);
119         }
120     }

122     /***
123      * Writes the inquiry specific information to the TCL update write. The
124      * properties are identified as settings if they starts with a
125      * &quot;%&quot;. This settings are also written.
126      *
127      * @param _paramCache   parameter cache
128      * @param _out          appendable instance to the TCL update file
129      * @throws IOException if the TCL update code for the menu could not be
130      *                     written
131      * @see #pattern
132      * @see #format
133      * @see #getPropertiesMap()
134      */
135     @Override()
136     protected void writeObject(final ParameterCache_mxJPO _paramCache,
137                                final Appendable _out)
138         throws IOException
139     {
140         _out.append(" //\n    pattern \"").append(StringUtil_mxJPO.convertTcl(this.pattern)).append("\"")
141             .append(" //\n    format \"").append(StringUtil_mxJPO.convertTcl(this.format)).append("\"")
142             .append(" //\n    file [file join \"${FILE}\"]");
143         for (final AdminProperty_mxJPO prop : this.getPropertiesMap().values())  {
144             if (prop.isSetting())  {
145                 _out.append(" //\n    add argument \"")
146                     .append(StringUtil_mxJPO.convertTcl(prop.getName().substring(1))).append("\"")
147                     .append(" \"").append(StringUtil_mxJPO.convertTcl(prop.getValue())).append("\"");
148             }
149         }
150     }
151 
152     /***
153      * At the end of the TCL update file the inquiry code must be appended.
154      *
155      * @param _paramCache   parameter cache
156      * @param _out          appendable instance to the TCL update file
157      * @throws IOException if the extension could not be written
158      * @see #code
159      * @see #PARAM_SEPARATOR_COMMENT
160      * @see #PARAM_SEPARATOR_TEXT
161      */
162     @Override()
163     protected void writeEnd(final ParameterCache_mxJPO _paramCache,
164                             final Appendable _out)
165         throws IOException
166     {
167         _out.append("\n\n").append(_paramCache.getValueString(Inquiry_mxJPO.PARAM_SEPARATOR_COMMENT))
168             .append('\n').append(_paramCache.getValueString(Inquiry_mxJPO.PARAM_SEPARATOR_TEXT)).append("\n\n");
169         if (this.code != null)  {
170             _out.append(this.code);
171         }
172     }

174     /***
175      * Updates this inquiry. Because the TCL source code of an inquiry includes
176      * also the inquiry code itself, this inquiry code must be separated and
177      * written in a temporary file. This temporary file is used while the
178      * update is running (defined via TCL variable <code>FILE</code>). After
179      * the update, the temporary file is removed (because not needed anymore).
180      * Also the MQL statements to reset this inquiry are appended to the
181      * statements in <code>_preMQLCode</code> to:
182      * <ul>
183      * <li>reset the description, pattern, format and code</li>
184      * <li>remove all arguments</li>
185      * </ul>
186      *
187      * @param _paramCache       parameter cache
188      * @param _preMQLCode       MQL statements which must be called before the
189      *                          TCL code is executed
190      * @param _postMQLCode      MQL statements which must be called after the
191      *                          TCL code is executed
192      * @param _preTCLCode       TCL code which is defined before the source
193      *                          file is sourced
194      * @param _tclVariables     map of all TCL variables where the key is the
195      *                          name and the value is value of the TCL variable
196      *                          (the value is automatically converted to TCL
197      *                          syntax!)
198      * @param _sourceFile       souce file with the TCL code to update
199      * @throws Exception if the update from derived class failed
200      * @see #PARAM_SEPARATOR_TEXT
201      */
202     @Override()
203     protected void update(final ParameterCache_mxJPO _paramCache,
204                           final CharSequence _preMQLCode,
205                           final CharSequence _postMQLCode,
206                           final CharSequence _preTCLCode,
207                           final Map<String,String> _tclVariables,
208                           final File _sourceFile)
209         throws Exception
210     {
211         // reset HRef, description, alt, label and height
212         final StringBuilder preMQLCode = new StringBuilder()
213                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
214                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
215                 .append(" description \"\" pattern \"\" format \"\" code \"\"");
216 
217         // reset arguments
218         for (final AdminProperty_mxJPO prop : this.getPropertiesMap().values())  {
219             if (prop.isSetting())  {
220                 preMQLCode.append(" remove argument \"")
221                           .append(StringUtil_mxJPO.convertMql(prop.getName().substring(1))).append('\"');
222             }
223         }
224 
225         // append already existing pre MQL code
226         preMQLCode.append(";\n")
227                   .append(_preMQLCode);
228 
229         // separate the inquiry code and the TCL code
230         final String sep = _paramCache.getValueString(Inquiry_mxJPO.PARAM_SEPARATOR_TEXT);
231         final StringBuilder orgCode = this.getCode(_sourceFile);
232         final int idx = orgCode.lastIndexOf(sep);
233         final CharSequence code = (idx >= 0)
234                                   ? orgCode.subSequence(0, idx)
235                                   : orgCode;
236         final CharSequence inqu = (idx >= 0)
237                                   ? orgCode.subSequence(idx + sep.length() + 1,
238                                                         orgCode.length())
239                                   : "";
240 
241         final File tmpTclFile = File.createTempFile("TMP_", ".tcl");
242         try  {
243             final File tmpInqFile = File.createTempFile("TMP_", ".inquiry");
244             try  {
245                 // write TCL code
246                 final Writer outTCL = new FileWriter(tmpTclFile);
247                 try {
248                     outTCL.append(code.toString().trim());
249                 } finally  {
250                     outTCL.close();
251                 }
252 
253                 // write inquiry code
254                 final Writer outInq = new FileWriter(tmpInqFile);
255                 try {
256                     outInq.append(inqu.toString().trim());
257                 } finally  {
258                     outInq.close();
259                 }
260 
261                 // define TCL variable for the file
262                 final Map<String,String> tclVariables = new HashMap<String,String>();
263                 tclVariables.putAll(_tclVariables);
264                 tclVariables.put("FILE", tmpInqFile.getPath());
265 
266                 // and update
267                 super.update(_paramCache, preMQLCode, _postMQLCode, code, tclVariables, tmpTclFile);
268             } finally  {
269                 tmpInqFile.delete();
270             }
271         } finally  {
272             tmpTclFile.delete();
273         }
274     }
