110     /***
111      * Parses all channel specific values. This includes:
112      * <ul>
113      * <li>command references in {@link #commandRefs}</li>
114      * <li>{@link #height}</li>
115      * <li>{@link #href}</li>
116      * <li>{@link #label}</li>
117      * </ul>
118      *
119      * @param _url      URL to parse
120      * @param _content  related content of the URL to parse
121      */
122     @Override
123     protected void parse(final String _url,
124                          final String _content)
125     {
126         if ("/alt".equals(_url))  {
127             this.alt = _content;
128 
129         } else if ("/commandRefList".equals(_url))  {
130             // to be ignored ...
131         } else if ("/commandRefList/commandRef".equals(_url))  {
132             this.commandRefs.add(new CommandRef());
133         } else if ("/commandRefList/commandRef/name".equals(_url))  {
134             this.commandRefs.peek().name = _content;
135         } else if ("/commandRefList/commandRef/order".equals(_url))  {
136             this.commandRefs.peek().order = Integer.parseInt(_content);
137 
138         } else if ("/height".equals(_url))  {
139             this.height = Integer.parseInt(_content);
140         } else if ("/href".equals(_url))  {
141             this.href = _content;
142         } else if ("/label".equals(_url))  {
143             this.label = _content;
144         } else  {
145             super.parse(_url, _content);
146         }
147     }

149     /***

152      * @param _paramCache   parameter cache

154      * @see #commandRefs        stack of not ordered commands references
155      * @see #orderCmds          instance of ordered commands references

157     @Override
158     protected void prepare(final ParameterCache_mxJPO _paramCache)
159             throws MatrixException
160     {
161         // order referenced commands
162         for (final CommandRef cmdRef : this.commandRefs)  {
163             this.orderCmds.put(cmdRef.order, cmdRef);
164         }
165         super.prepare(_paramCache);
166     }

168     /***
169      * Writes the label, href, alt, height, settings and the referenced
170      * commands as MQL code to the writer instance.
171      *
172      * @param _paramCache   parameter cache
173      * @param _out          appendable instance to the TCL update file
174      * @throws IOException if the TCL update code for the channel could not be
175      *                     written
176      */
177     @Override
178     protected void writeObject(final ParameterCache_mxJPO _paramCache,
179                                final Appendable _out)
180             throws IOException
181     {
182         _out.append(" //\n    label \"").append(StringUtil_mxJPO.convertTcl(this.label)).append("\"");
183         if (this.href != null)  {
184             _out.append(" //\n    href \"").append(StringUtil_mxJPO.convertTcl(this.href)).append("\"");
185         }
186         if (this.alt != null)  {
187             _out.append(" //\n    alt \"").append(StringUtil_mxJPO.convertTcl(this.alt)).append("\"");
188         }
189         if (this.height != null)  {
190             _out.append(" //\n    height \"").append(this.height.toString()).append("\"");
191         }
192         // settings
193         for (final AdminProperty_mxJPO prop : this.getPropertiesMap().values())  {
194             if (prop.isSetting())  {
195                 _out.append(" //\n    add setting \"")
196                     .append(StringUtil_mxJPO.convertTcl(prop.getName().substring(1))).append("\"")
197                     .append(" \"").append(StringUtil_mxJPO.convertTcl(prop.getValue())).append("\"");
198             }
199         }
200         // referenced commands
201         for (final CommandRef cmdRef : this.orderCmds.values())  {
202             _out.append(" //\n    place \"").append(StringUtil_mxJPO.convertTcl(cmdRef.name)).append("\" after \"\"");
203         }
204     }

206     /***
207      * The method overwrites the original method to append the MQL statements
208      * in the <code>_preMQLCode</code> to reset this channel. Following steps
209      * are done:
210      * <ul>
211      * <li>reset HRef, description, alt and label</li>
212      * <li>set height to 0</li>
213      * <li>remove all settings and commands</li>
214      * </ul>
215      *
216      * @param _paramCache       parameter cache
217      * @param _preMQLCode       MQL statements which must be called before the
218      *                          TCL code is executed
219      * @param _postMQLCode      MQL statements which must be called after the
220      *                          TCL code is executed
221      * @param _preTCLCode       TCL code which is defined before the source
222      *                          file is sourced
223      * @param _tclVariables     map of all TCL variables where the key is the
224      *                          name and the value is value of the TCL variable
225      *                          (the value is automatically converted to TCL
226      *                          syntax!)
227      * @param _sourceFile       souce file with the TCL code to update
228      * @throws Exception if the update from derived class failed
229      */
230     @Override
231     protected void update(final ParameterCache_mxJPO _paramCache,
232                           final CharSequence _preMQLCode,
233                           final CharSequence _postMQLCode,
234                           final CharSequence _preTCLCode,
235                           final Map<String,String> _tclVariables,
236                           final File _sourceFile)
237             throws Exception
238     {
239         // reset HRef, description, alt, label and height
240         final StringBuilder preMQLCode = new StringBuilder()
241                 .append("mod ").append(this.getTypeDef().getMxAdminName())
242                 .append(" \"").append(this.getName()).append('\"')
243                 .append(" href \"\" description \"\" alt \"\" label \"\" height 0");
244 
245         // reset settings
246         for (final AdminProperty_mxJPO prop : this.getPropertiesMap().values())  {
247             if (prop.isSetting())  {
248                 preMQLCode.append(" remove setting \"").append(prop.getName().substring(1)).append('\"');
249             }
250         }
251 
252         // remove commands
253         for (final CommandRef cmdRef : this.orderCmds.values())  {
254             preMQLCode.append(" remove command \"").append(cmdRef.name).append('\"');
255         }
256 
257         // append already existing pre MQL code
258         preMQLCode.append(";\n")
259                   .append(_preMQLCode);
260 
261         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
262     }
