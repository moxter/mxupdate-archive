165     /***
166      * Export given administration (business) object with given name into given
167      * path. The name of the file where is written through is evaluated within
168      * this export method.
169      *
170      * @param _paramCache   parameter cache
171      * @param _path         path to write through (if required also including
172      *                      depending file path defined from the information
173      *                      annotation)
174      * @throws MatrixException  if some MQL statement failed
175      * @throws SAXException     if the XML export of the object could not
176      *                          parsed (for admin objects)
177      * @throws IOException      if the TCL update code could not be written
178      */
179     public void export(final ParameterCache_mxJPO _paramCache,
180                        final File _path)
181         throws MatrixException, SAXException, IOException
182     {
183         this.parse(_paramCache);
184         final File file = new File(_path, this.getFileName());
185         if (!file.getParentFile().exists())  {
186             file.getParentFile().mkdirs();
187         }
188         final Writer out = new FileWriter(file);
189         this.write(_paramCache, out);
190         out.flush();
191         out.close();
192     }

264     /***
265      * Extracts the MX name from given file name if the file prefix and suffix
266      * matches. If the file prefix and suffix not matches a <code>null</code>
267      * is returned.
268      *
269      * @param _paramCache   parameter cache
270      * @param _file         file for which the MX name is searched
271      * @return MX name or <code>null</code> if the file is not an update file
272      *         for current type definition
273      * @throws UpdateException_mxJPO if the configuration item name could not
274      *                               be extracted from the file name
275      */
276     public String extractMxName(final ParameterCache_mxJPO _paramCache,
277                                 final File _file)
278         throws UpdateException_mxJPO
279     {
280         final String suffix = this.getTypeDef().getFileSuffix();
281         final int suffixLength = (suffix != null) ? suffix.length() : 0;
282         final String prefix = this.getTypeDef().getFilePrefix();
283         final int prefixLength = (prefix != null) ? prefix.length() : 0;
284 
285         final String fileName = _file.getName();
286         final String mxName;
287         if (((prefix == null) || fileName.startsWith(prefix)) && ((suffix == null) || fileName.endsWith(suffix)))  {
288             mxName = StringUtil_mxJPO.convertFromFileName(fileName.substring(0, fileName.length() - suffixLength)
289                                                  .substring(prefixLength));
290         } else  {
291             mxName = null;
292         }
293         return mxName;
294     }

345     /***
346      * Reads for given file the code and returns them.
347      *
348      * @param _file     file to read the code
349      * @return read code of the file
350      * @throws IOException if the file could not be opened or read
351      */
352     protected StringBuilder getCode(final File _file)
353         throws IOException
354     {
355         // read code
356         final StringBuilder code = new StringBuilder();
357         final BufferedReader reader = new BufferedReader(new FileReader(_file));
358         String line = reader.readLine();
359         while (line != null)  {
360             code.append(line).append('\n');
361             line = reader.readLine();
362         }
363         reader.close();
364 
365         return code;
366     }

545     /***
546      * <p>The calculates and returns the default symbolic name. A typical
547      * symbolic name has as prefix the admin type name, then an underscore and
548      * at least the name of the admin object. All not allowed special
549      * characters are replaced by &quot;nothing&quot; (zero length string).</p>
550      * <p>The characters to replace are defined as parameter with name
551      * {@link #PARAM_CALC_SYMB_NAME_REG_EXP}.</p>
552      *
553      * @param _paramCache   parameter cache
554      * @return calculated default symbolic name for administration objects (if
555      *         the administration object is a business object <code>null</code>
556      *         is returned)
557      * @see #PARAM_CALC_SYMB_NAME_REG_EXP
558      */
559     protected String calcDefaultSymbolicName(final ParameterCache_mxJPO _paramCache)
560     {
561         final String regExp = _paramCache.getValueString(AbstractObject_mxJPO.PARAM_CALC_SYMB_NAME_REG_EXP);
562         return (this.getTypeDef().getMxAdminName() == null)
563                ? null
564                : new StringBuilder()
565                         .append(this.getTypeDef().getMxAdminName())
566                         .append("_")
567                         .append(this.getName().replaceAll(regExp, ""))
568                         .toString();
569     }

583     /***
584      * Returns the file name for this MxUpdate administration object. The file
585      * name is a concatenation of the defined file prefix within the
586      * information annotation , the name of the MX object and the file suffix
587      * within the information annotation. All special characters are converted
588      * automatically from {@link StringUtil_mxJPO#convertToFileName(String)}.
589      *
590      * @return file name of this administration (business) object
591      * @see #export(ParameterCache_mxJPO, File)
592      */
593     public String getFileName()
594     {
595         final StringBuilder ret = new StringBuilder();
596         if (this.getTypeDef().getFilePrefix() != null)  {
597             ret.append(this.getTypeDef().getFilePrefix());
598         }
599         ret.append(this.getName());
600         if (this.getTypeDef().getFileSuffix() != null)  {
601             ret.append(this.getTypeDef().getFileSuffix());
602         }
603         return StringUtil_mxJPO.convertToFileName(ret.toString());
604     }
