189     /***
190      * Export given administration (business) object with given name into given
191      * path. The name of the file where is written through is evaluated within
192      * this export method.
193      *
194      * @param _paramCache   parameter cache
195      * @param _path         path to write through (if required also including
196      *                      depending file path defined from the information
197      *                      annotation)
198      * @throws MatrixException  if some MQL statement failed
199      * @throws SAXException     if the XML export of the object could not
200      *                          parsed (for admin objects)
201      * @throws IOException      if the TCL update code could not be written
202      */
203     public void export(final ParameterCache_mxJPO _paramCache,
204                        final File _path)
205             throws MatrixException, SAXException, IOException
206     {
207         this.parse(_paramCache);
208         final File file = new File(_path, this.getFileName());
209         if (!file.getParentFile().exists())  {
210             file.getParentFile().mkdirs();
211         }
212         final Writer out = new FileWriter(file);
213         this.write(_paramCache, out);
214         out.flush();
215         out.close();
216     }

288     /***
289      * Extracts the MX name from given file name if the file prefix and suffix
290      * matches. If the file prefix and suffix not matches a <code>null</code>
291      * is returned.
292      *
293      * @param _paramCache   parameter cache
294      * @param _file         file for which the MX name is searched
295      * @return MX name or <code>null</code> if the file is not an update file
296      *         for current type definition
297      * @throws UpdateException_mxJPO if the configuration item name could not
298      *                               be extracted from the file name
299      */
300     public String extractMxName(final ParameterCache_mxJPO _paramCache,
301                                 final File _file)
302         throws UpdateException_mxJPO
303     {
304         final String suffix = this.getTypeDef().getFileSuffix();
305         final int suffixLength = (suffix != null) ? suffix.length() : 0;
306         final String prefix = this.getTypeDef().getFilePrefix();
307         final int prefixLength = (prefix != null) ? prefix.length() : 0;
308 
309         final String fileName = _file.getName();
310         final String mxName;
311         if (((prefix == null) || fileName.startsWith(prefix)) && ((suffix == null) || fileName.endsWith(suffix)))  {
312             mxName = StringUtil_mxJPO.convertFromFileName(fileName.substring(0, fileName.length() - suffixLength)
313                                                  .substring(prefixLength));
314         } else  {
315             mxName = null;
316         }
317         return mxName;
318     }

369     /***
370      * Reads for given file the code and returns them.
371      *
372      * @param _file     file to read the code
373      * @return read code of the file
374      * @throws IOException if the file could not be opened or read
375      */
376     protected StringBuilder getCode(final File _file)
377             throws IOException
378     {
379         // read code
380         final StringBuilder code = new StringBuilder();
381         final BufferedReader reader = new BufferedReader(new FileReader(_file));
382         String line = reader.readLine();
383         while (line != null)  {
384             code.append(line).append('\n');
385             line = reader.readLine();
386         }
387         reader.close();
388 
389         return code;
390     }

392     /***
393      * Returns the stored value within Matrix for administration object
394      * with given property name. For performance reason the method uses
395      * &quot;print&quot; commands, because a complete XML parse including a
396      * complete export takes longer time.
397      *
398      * @param _paramCache   parameter cache
399      * @param _prop         property for which the value is searched
400      * @return value for given property
401      * @throws MatrixException if the property value could not be extracted
402      */
403     public String getPropValue(final ParameterCache_mxJPO _paramCache,
404                                final PropertyDef_mxJPO _prop)
405             throws MatrixException
406     {
407         final String curVersion;
408         // check for existing administration type...
409         if (this.getTypeDef().getMxAdminName() != null)  {
410             final String tmp = MqlUtil_mxJPO.execMql(_paramCache, new StringBuilder()
411                     .append("print ").append(this.getTypeDef().getMxAdminName())
412                     .append(" \"").append(this.getName()).append("\" ")
413                     .append(this.getTypeDef().getMxAdminSuffix())
414                     .append(" select property[").append(_prop.getPropName(_paramCache)).append("] dump"));
415             final int length = 7 + _prop.getPropName(_paramCache).length();
416             curVersion = (tmp.length() >= length)
417                          ? tmp.substring(length)
418                          : "";
419         // otherwise we have a business object....
420         } else  {
421             final String[] nameRev = this.getName().split("________");
422             curVersion = MqlUtil_mxJPO.execMql(_paramCache, new StringBuilder()
423                     .append("print bus \"")
424                     .append(this.getTypeDef().getMxBusType())
425                     .append("\" \"").append(nameRev[0])
426                     .append("\" \"").append((nameRev.length > 1) ? nameRev[1] : "")
427                     .append("\" select attribute[").append(_prop.getAttrName(_paramCache)).append("] dump"));
428         }
429 
430         return curVersion;
431     }

600     /***
601      * Reads the symbolic names for current admin objects and stores them in
602      * {@link #symbolicNames}.
603      *
604      * @param _paramCache       parameter cache
605      * @throws MatrixException if the symbolic names could not be read
606      * @see #symbolicNames
607      */
608     protected void readSymbolicNames(final ParameterCache_mxJPO _paramCache)
609         throws MatrixException
610     {
611         // context must be checked if used within automatic tests
612         if ((this.getTypeDef().getMxAdminName() != null) && (_paramCache.getContext() != null))  {
613             final String symbProg = _paramCache.getValueString(AbstractObject_mxJPO.PARAM_SYMB_NAME_PROG);
614             final String symbProgIdxOf = new StringBuilder()
615                     .append(" on program ").append(symbProg).append(' ').toString();
616             final StringBuilder cmd = new StringBuilder()
617                     .append("escape list property on program \"")
618                                 .append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
619                         .append(this.getTypeDef().getMxAdminName())
620                         .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
621                         .append(this.getTypeDef().getMxAdminSuffix());
622             for (final String symbName : MqlUtil_mxJPO.execMql(_paramCache, cmd).split("\n"))  {
623                 if (!"".equals(symbName))  {
624                     this.symbolicNames.add(symbName.substring(0, symbName.indexOf(symbProgIdxOf)));
625                 }
626             }
627         }
628     }

630     /***
631      * Appends the escaped MQL code to register given symbolic name
632      * <code>_symbName</code> depending on current defined symbolic names
633      * {@link #symbolicNames}. If not registered, a new registration is done.
634      * If wrong symbolic names are defined, they are removed.
635      *
636      * @param _paramCache   parameter cache
637      * @param _symbName     symbolic name which must be set
638      * @param _mqlCode      string builder where the MQL command must be
639      *                      appended
640      * @see #symbolicNames
641      */
642     protected void appendSymbolicNameRegistration(final ParameterCache_mxJPO _paramCache,
643                                                   final String _symbName,
644                                                   final StringBuilder _mqlCode)
645     {
646         if (this.getTypeDef().getMxAdminName() != null)  {
647             final String symbProg = _paramCache.getValueString(AbstractObject_mxJPO.PARAM_SYMB_NAME_PROG);
648             if (!this.symbolicNames.contains(_symbName))  {
649                 _paramCache.logTrace("    - register symbolic name '" + _symbName + "'");
650                 _mqlCode.append("escape add property \"").append(StringUtil_mxJPO.convertMql(_symbName)).append("\" ")
651                         .append(" on program \"").append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
652                         .append(this.getTypeDef().getMxAdminName())
653                         .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
654                         .append(this.getTypeDef().getMxAdminSuffix())
655                         .append(";\n");
656             }
657             for (final String exSymbName : this.symbolicNames)  {
658                 if (!_symbName.equals(exSymbName))  {
659                     _paramCache.logTrace("    - remove symbolic name '" + exSymbName + "'");
660                     _mqlCode.append("escape delete property \"")
661                                     .append(StringUtil_mxJPO.convertMql(exSymbName)).append("\" ")
662                             .append(" on program \"").append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
663                             .append(this.getTypeDef().getMxAdminName())
664                             .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
665                             .append(this.getTypeDef().getMxAdminSuffix())
666                             .append(";\n");
667                 }
668             }
669         }
670     }

672     /***
673      * <p>The calculates and returns the default symbolic name. A typical
674      * symbolic name has as prefix the admin type name, then an underscore and
675      * at least the name of the admin object. All not allowed special
676      * characters are replaced by &quot;nothing&quot; (zero length string).</p>
677      * <p>The characters to replace are defined as parameter with name
678      * {@link #PARAM_CALC_SYMB_NAME_REG_EXP}.</p>
679      *
680      * @param _paramCache   parameter cache
681      * @return calculated default symbolic name for administration objects (if
682      *         the administration object is a business object <code>null</code>
683      *         is returned)
684      * @see #PARAM_CALC_SYMB_NAME_REG_EXPs
685      */
686     protected String calcDefaultSymbolicName(final ParameterCache_mxJPO _paramCache)
687     {
688         final String regExp = _paramCache.getValueString(AbstractObject_mxJPO.PARAM_CALC_SYMB_NAME_REG_EXP);
689         return (this.getTypeDef().getMxAdminName() == null)
690                ? null
691                : new StringBuilder()
692                         .append(this.getTypeDef().getMxAdminName())
693                         .append("_")
694                         .append(this.getName().replaceAll(regExp, ""))
695                         .toString();
696     }

710     /***
711      * Returns the file name for this MxUpdate administration object. The file
712      * name is a concatenation of the defined file prefix within the
713      * information annotation , the name of the MX object and the file suffix
714      * within the information annotation. All special characters are converted
