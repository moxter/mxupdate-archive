159     /***
160      * Method used as entry from the MQL interface to install / update the data
161      * model for MxUpdate. The data model is installed / updated in this order:
162      * <ul>
163      * <li>install trigger group type</li>
164      * <li>install trigger group policy</li>
165      * <li>install trigger group relationship</li>
166      * <li>update attributes in
167      *     {@link #updateAttributes(ParameterCache_mxJPO, String, String, String, String, String, String)}</li>
168      * <li>update used business types by appending required MxUpdate attributes
169      *     in {@link #updateBusTypes(ParameterCache_mxJPO)}</li>
170      * <li>define plug-in properties in
171      *     {@link #makePluginProperty(ParameterCache_mxJPO, File)}</li>
172      * <li>register the MxUdpate Update programs with their symbolic names in
173      *     {@link #registerPrograms(ParameterCache_mxJPO)}</li>
174      * <li>register MxUpdate as application in
175      *     {@link #registerMxUpdate(ParameterCache_mxJPO, String, String)}</li>
176      * </ul>
177      *
178      * @param _context      MX context for this request
179      * @param _args         first value defines the source installation path
180      *                      (in development it is the resources sub directory),
181      *                      second value the version of MxUpdate which must be
182      *                      installed
183      * @throws Exception if installation failed
184      * @see #installTriggerGroupType(ParameterCache_mxJPO, File, String, String, String, String, String, String)
185      * @see #installTriggerGroupPolicy(ParameterCache_mxJPO, String, String, String, String, String, String)
186      * @see #installTriggerGroupRelation(ParameterCache_mxJPO, String, String, String, String, String, String)
187      * @see #updateAttributes(ParameterCache_mxJPO, String, String, String, String, String, String)
188      * @see #updateBusTypes(ParameterCache_mxJPO)
189      * @see #registerPrograms(ParameterCache_mxJPO)
190      * @see #registerMxUpdate(ParameterCache_mxJPO, String, String)
191      * @see #makePluginProperty(ParameterCache_mxJPO, File)
192      */
193     public void mxMain(final Context _context,
194                        final String... _args)
195             throws Exception
196     {
197         final File path = new File(_args[0]);
198         final String applVersion = _args[1];
199 
200         // initialize mapping
201         final ParameterCache_mxJPO paramCache = new ParameterCache_mxJPO(_context, false);
202 
203         final String applName = paramCache.getValueString(InstallDataModel_mxJPO.PARAM_APPLNAME);
204         final String authorName = paramCache.getValueString(InstallDataModel_mxJPO.PARAM_AUTHOR);
205         final String installerName = paramCache.getValueString(InstallDataModel_mxJPO.PARAM_INSTALLER);
206         final String fileDate = StringUtil_mxJPO.formatFileDate(paramCache, new Date());
207         final String installedDate = StringUtil_mxJPO.formatInstalledDate(paramCache, new Date());
208 
209         this.installTriggerGroupType(paramCache, path,
210                 applName, applVersion,
211                 authorName, installerName,
212                 fileDate, installedDate);
213         this.installTriggerGroupPolicy(paramCache,
214                 applName, applVersion,
215                 authorName, installerName,
216                 fileDate, installedDate);
217         this.installTriggerGroupRelation(paramCache,
218                 applName, applVersion,
219                 authorName, installerName,
220                 fileDate, installedDate);
221         this.updateAttributes(paramCache,
222                 applName, applVersion,
223                 authorName, installerName,
224                 fileDate, installedDate);
225         this.updateBusTypes(paramCache);
226         this.makePluginProperty(paramCache, path);
227         this.registerPrograms(paramCache);
228         this.registerMxUpdate(paramCache, applName, applVersion);
229     }

231     /***
232      * All programs evaluated with {@link #LIST_MXUPDATE_PROGRAMS} are checked
233      * if they must be registered with a symbolic name. If they are not already
234      * registered, the registration of the symbolic names of the programs are
235      * done.
236      *
237      * @param _paramCache   parameter cache
238      * @throws Exception if registration of the symbolic names for all MxUpdate
239      *                   Update programs failed
240      * @see #LIST_MXUPDATE_PROGRAMS
241      */
242     protected void registerPrograms(final ParameterCache_mxJPO _paramCache)
243         throws Exception
244     {
245         final TypeDef_mxJPO jpoTypeDef
246                 = _paramCache.getMapping().getTypeDef(InstallDataModel_mxJPO.TYPEDEF_JPO);
247         final TypeDef_mxJPO mqlTypeDef
248                 = _paramCache.getMapping().getTypeDef(InstallDataModel_mxJPO.TYPEDEF_MQL);
249 
250         final String progs = MqlUtil_mxJPO.execMql(_paramCache.getContext(),
251                                                    InstallDataModel_mxJPO.LIST_MXUPDATE_PROGRAMS);
252         for (final String progLine : progs.split("\n"))  {
253             final String[] progLineArr = progLine.split("@");
254             final String progName = progLineArr[0];
255             // do we have a JPO?
256             if ("true".equalsIgnoreCase(progLineArr[1]))  {
257                 this.registerObject(_paramCache, jpoTypeDef.newTypeInstance(progName));
258             } else  {
259                 this.registerObject(_paramCache, mqlTypeDef.newTypeInstance(progName));
260             }
261         }
262     }

264     /***
265      * Makes the registration of MxUpdate as application.
266      *
267      * @param _paramCache   parameter cache with MX context
268      * @param _applName     used application name of the MxUpdate Update
269      *                      deployment tool
270      * @param _version      MxUpdate version
271      * @throws MatrixException if registration of MxUpdate failed
272      * @see #PARAM_PROGAPPL
273      */
274     protected void registerMxUpdate(final ParameterCache_mxJPO _paramCache,
275                                     final String _applName,
276                                     final String _version)
277             throws MatrixException
278     {
279         final String progName = _paramCache.getValueString(InstallDataModel_mxJPO.PARAM_PROGAPPL);
280 
281         _paramCache.logInfo("register MxUpdate " + _version);
282         MqlUtil_mxJPO.execMql(_paramCache.getContext(), new StringBuilder()
283                 .append("escape mod prog \"").append(StringUtil_mxJPO.convertMql(progName)).append("\" ")
284                 .append("add property \"appVersion").append(StringUtil_mxJPO.convertMql(_applName)).append("\" ")
285                 .append("value \"").append(StringUtil_mxJPO.convertMql(_version)).append("\""));
286     }

288     /***
289      * The trigger group type is installed.
290      *
291      * @param _paramCache       parameter cache with MX context
292      * @param _path             installation path (needed to fetch the icons)
293      * @param _applName         used application name of the MxUpdate Update
294      *                          deployment tool
295      * @param _applVersion      Mx Update version
296      * @param _authorName       used author name
297      * @param _installerName    used installer name
298      * @param _fileDate         used file date
299      * @param _installedDate    used installed date
300      * @throws Exception if the trigger group type could not be installed
301      */
302     protected void installTriggerGroupType(final ParameterCache_mxJPO _paramCache,
303                                            final File _path,
304                                            final String _applName,
305                                            final String _applVersion,
306                                            final String _authorName,
307                                            final String _installerName,
308                                            final String _fileDate,
309                                            final String _installedDate)
310             throws Exception
311     {
312 
313         final TypeDef_mxJPO trigGrpTypeDef
314                 = _paramCache.getMapping().getTypeDef(InstallDataModel_mxJPO.TYPEDEF_TRIGGER_GROUP);
315         _paramCache.logInfo("check type '" + trigGrpTypeDef.getMxBusType() + "'");
316         final String installed = MqlUtil_mxJPO.execMql(_paramCache.getContext(), new StringBuilder()
317                 .append("escape list type \"")
318                         .append(StringUtil_mxJPO.convertMql(trigGrpTypeDef.getMxBusType())).append('\"'));
319         if ("".equals(installed))  {
320             _paramCache.logDebug("    - create");
321             MqlUtil_mxJPO.execMql(_paramCache.getContext(),
322                     new StringBuilder()
323                         .append("escape add type \"")
324                         .append(StringUtil_mxJPO.convertMql(trigGrpTypeDef.getMxBusType()))
325                         .append("\""));
326         }
327         _paramCache.logDebug("    - update description");
328         final StringBuilder cmd = new StringBuilder()
329                 .append("escape mod type \"")
330                 .append(StringUtil_mxJPO.convertMql(trigGrpTypeDef.getMxBusType())).append("\" ")
331                 .append("description \"Type used to group trigger definitions.\" ");
332 
333         // update type icon
334         final File file = new File(_path, trigGrpTypeDef.getIconPath());
335         if (file.exists())  {
336             _paramCache.logDebug("    - update type icon");
337             cmd.append("icon \"").append(StringUtil_mxJPO.convertMql(file.toString())).append("\" ");
338         }
339 
340         // append property settings
341         final AbstractObject_mxJPO instance = _paramCache.getMapping()
342                                             .getTypeDef(InstallDataModel_mxJPO.TYPEDEF_TYPE)
343                                             .newTypeInstance(trigGrpTypeDef.getMxBusType());
344         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.APPLICATION, _applName, cmd, false);
345         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.AUTHOR, _authorName, cmd, false);
346         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.INSTALLER, _installerName, cmd, false);
347         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.VERSION, _applVersion, cmd, false);
348         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.ORIGINALNAME,
349                            trigGrpTypeDef.getMxBusType(), cmd, false);
350         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.FILEDATE, _fileDate, cmd, true);
351         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.INSTALLEDDATE, _installedDate, cmd, true);
352         MqlUtil_mxJPO.execMql(_paramCache.getContext(), cmd);
353 
354         this.registerObject(_paramCache, instance);
355     }

375                                              final String _applName,

420         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.ORIGINALNAME, polName, cmd, false);
421         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.FILEDATE, _fileDate, cmd, true);

424 
425         this.registerObject(_paramCache, instance);
426     }

499     /***
500      * Creates / updates all needed attributes used as MxUpdate properties.
501      *
502      * @param _paramCache       parameter cache
503      * @param _applName         used application name of the MxUpdate Update
504      *                          deployment tool
505      * @param _applVersion      new MxUpdate version
506      * @param _authorName       used author name
507      * @param _installerName    used installer name
508      * @param _fileDate         used file date
509      * @param _installedDate    used installed date
510      * @throws Exception if update of attribute failed
511      */
512     protected void updateAttributes(final ParameterCache_mxJPO _paramCache,
513                                     final String _applName,
514                                     final String _applVersion,
515                                     final String _authorName,
516                                     final String _installerName,
517                                     final String _fileDate,
518                                     final String _installedDate)
519             throws Exception
520     {
521         for (final PropertyDef_mxJPO propDef : PropertyDef_mxJPO.values())  {
522             if ((propDef.getAttrName(_paramCache) != null) && !"".equals(propDef.getAttrName(_paramCache)))  {
523                 _paramCache.logInfo("check attribute '" + propDef.getAttrName(_paramCache) + "'");
524 
525                 final String exists = MqlUtil_mxJPO.execMql(_paramCache.getContext(),
526                         new StringBuilder().append("list attribute '")
527                                            .append(propDef.getAttrName(_paramCache))
528                                            .append('\''));
529                 if ("".equals(exists))  {
530                     _paramCache.logDebug("    - create");
531                     MqlUtil_mxJPO.execMql(_paramCache.getContext(),
532                             new StringBuilder()
533                                 .append("escape add attribute \"")
534                                 .append(StringUtil_mxJPO.convertMql(propDef.getAttrName(_paramCache)))
535                                 .append("\" type string;"));
536                 }
537 
538                 final StringBuilder cmd = new StringBuilder()
539                     .append("escape mod attribute \"")
540                     .append(StringUtil_mxJPO.convertMql(propDef.getAttrName(_paramCache))).append("\" ");
541 
542                 final AbstractObject_mxJPO instance
543                         = _paramCache.getMapping()
544                                      .getTypeDef(InstallDataModel_mxJPO.TYPEDEF_ATTRIBUTE_STRING)
545                                      .newTypeInstance(propDef.getAttrName(_paramCache));
546 
547                 // check for correct application name

622     protected void makePluginProperty(final ParameterCache_mxJPO _paramCache,
623                                       final File _sourcePath)
624             throws IOException, MatrixException

626         // prepare properties as set
627         final Set<String> props = new TreeSet<String>();
628         for (final TypeDef_mxJPO typeDef : _paramCache.getMapping().getAllTypeDefs())  {
629             if (typeDef.getIconPath() != null)  {

654             }

708      * Register symbolic name for given administration object

715     protected void registerObject(final ParameterCache_mxJPO _paramCache,
716                                   final AbstractObject_mxJPO _instance)
717             throws Exception
718     {
719         final String newSymbName = _instance.getTypeDef().getMxAdminName() + "_"
720                                     + _instance.getName().replaceAll(" ", "");
721         final Set<String> symbolicNames = new HashSet<String>();
722 
723         final String symbProg = _paramCache.getValueString(InstallDataModel_mxJPO.PARAM_SYMB_NAME_PROG);
724 
725         // reads symbolic names of the administration objects
726         final String symbProgIdxOf = new StringBuilder().append(" on program ").append(symbProg).append(' ').toString();
727         final StringBuilder cmd = new StringBuilder()
728                 .append("escape list property on program \"")
729                             .append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
730                     .append(_instance.getTypeDef().getMxAdminName())
731                     .append(" \"").append(_instance.getName()).append("\" ")
732                     .append(_instance.getTypeDef().getMxAdminSuffix());
733         for (final String symbName : MqlUtil_mxJPO.execMql(_paramCache.getContext(), cmd).split("\n"))  {
734             if (!"".equals(symbName))  {
735                 symbolicNames.add(symbName.substring(0, symbName.indexOf(symbProgIdxOf)));
736             }
737         }
738         // check symbolic names
739         final StringBuilder update = new StringBuilder();
740         if (!symbolicNames.contains(newSymbName))  {
741             _paramCache.logDebug("    - register symbolic name '" + newSymbName + "'");
742             update.append("escape add property \"").append(StringUtil_mxJPO.convertMql(newSymbName)).append("\" ")
743                     .append(" on program \"").append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
744                     .append(_instance.getTypeDef().getMxAdminName())
745                     .append(" \"").append(StringUtil_mxJPO.convertMql(_instance.getName())).append("\" ")
746                     .append(_instance.getTypeDef().getMxAdminSuffix())
747                     .append(";\n");
748         }
749         for (final String exSymbName : symbolicNames)  {
750             if (!newSymbName.equals(exSymbName))  {
751                 _paramCache.logDebug("    - remove symbolic name '" + exSymbName + "'");
752                 update.append("escape delete property \"")
753                                 .append(StringUtil_mxJPO.convertMql(exSymbName)).append("\" ")
754                         .append(" on program \"").append(StringUtil_mxJPO.convertMql(symbProg)).append("\" to ")
755                         .append(_instance.getTypeDef().getMxAdminName())
756                         .append(" \"").append(StringUtil_mxJPO.convertMql(_instance.getName())).append("\" ")
757                         .append(_instance.getTypeDef().getMxAdminSuffix())
758                         .append(";\n");
