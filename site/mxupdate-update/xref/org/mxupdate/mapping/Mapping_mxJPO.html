174     /***
175      *
176      * @param _paramCache   parameter cache
177      * @throws MatrixException if the property program {@link #PROP_NAME} could
178      *                         not be read
179      * @throws IOException     if the properties could not be parsed
180      */
181     public Mapping_mxJPO(final ParameterCache_mxJPO _paramCache)
182             throws MatrixException, IOException, Exception
183     {
184         this.properties.putAll(this.readProperties(_paramCache));
185 
186         // map attributes and types
187         for (final Map.Entry<Object, Object> entry : this.properties.entrySet())  {
188             final String key = (String) entry.getKey();
189             final String value = (String) entry.getValue();
190             if (key.startsWith("Mode."))  {
191                 Mode_mxJPO.defineValue(this, key.substring(5), value);
192             } else if (key.startsWith("ParameterDef."))  {
193                 ParameterDef_mxJPO.defineValue(this, key.substring(13), value);
194             } else if (key.startsWith(Mapping_mxJPO.PREFIX_PROPERTYDEF))  {
195                 PropertyDef_mxJPO.defineValue(this, key.substring(Mapping_mxJPO.LENGTH_PROPERTYDEF), value);
196             } else if (key.startsWith("TypeDef."))  {
197                 TypeDef_mxJPO.defineValue(_paramCache, this, key.substring(8), value);
198             } else if (key.startsWith("TypeDefGroup."))  {
199                 TypeDefGroup_mxJPO.defineValue(this, key.substring(13), value);
200             } else if (key.startsWith("TypeDefTree."))  {
201                 TypeDefTree_mxJPO.defineValue(this, key.substring(12), value);
202             } else if (key.startsWith("UpdateCheck."))  {
203                 UpdateCheck_mxJPO.defineValue(this, key.substring(12), value);
204             }
205         }
206         this.typeDefSorted.addAll(this.typeDefMap.values());
207     }

209     /***
210      * Reads the code of the program {@link #PROP_NAME} and evaluates the code
211      * of the program as properties.
212      *
213      * @param _paramCache   parameter cache
214      * @return read properties from MX
215      * @throws MatrixException  if the program could not be read
216      * @throws IOException      if the code of the program could not be parsed
217      *                          as properties
218      * @see #PROP_NAME
219      */
220     protected Properties readProperties(final ParameterCache_mxJPO _paramCache)
221         throws MatrixException, IOException
222     {
223         final String code = MqlUtil_mxJPO.execMql(_paramCache.getContext(),
224                 new StringBuilder()
225                         .append("print prog \"").append(Mapping_mxJPO.PROP_NAME)
226                         .append("\" select code dump"),
227                 false);
228 
229         final InputStream is = new ByteArrayInputStream(code.getBytes());
230         final Properties properties = new Properties();
231         properties.load(is);
232 
233         return properties;
234     }
