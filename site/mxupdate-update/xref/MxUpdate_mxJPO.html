183     /***
184      *
185      * @param _paramCache   parameter cache
186      * @throws MatrixException
187      */
188     private void prepareParams(final ParameterCache_mxJPO _paramCache)
189             throws MatrixException
190     {
191         this.description.clear();
192         this.allParams.clear();
193         this.paramsModes.clear();
194         this.paramsParameters.clear();
195         this.paramsTypeDefs.clear();
196         this.paramsUpdateChecks.clear();
197 
198         ////////////////////////////////////////////////////////////////////////
199         // parameters
200 
201         for (final ParameterDef_mxJPO parameter : _paramCache.getMapping().getAllParameterDefs())  {
202             if (parameter.getParameterList() != null)  {
203                 for (final String param : parameter.getParameterList())  {
204                     final String paramStr = (param.length() > 1)
205                                             ? "--" + param
206                                             : "-" + param;
207                     this.paramsParameters.put(paramStr, parameter);
208                 }
209                 final StringBuilder desc = new StringBuilder().append(parameter.getParameterDesc());
210                 if ((parameter.getDefaultValue() != null)
211                         && (parameter.getType() != ParameterDef_mxJPO.Type.BOOLEAN))  {
212                     desc.append('\n').append("(Default '");
213                     if (parameter.getType() == ParameterDef_mxJPO.Type.LIST)  {
214                         desc.append(parameter.getDefaultValue().replaceAll(",", ", "));
215                     } else  {
216                         desc.append(parameter.getDefaultValue());
217                     }
218                     desc.append("')");
219                 }
220                 this.appendDescription(desc,
221                                        parameter.getParameterList(),
222                                        parameter.getParameterArgs());
223             }
224         }
225 
226         ////////////////////////////////////////////////////////////////////////
227         // modes
228 
229         for (final Mode_mxJPO mode : Mode_mxJPO.values())  {
230             for (final String param : mode.getParameterList(_paramCache))  {
231                 final String paramStr = (param.length() > 1)
232                                         ? "--" + param
233                                         : "-" + param;
234                 this.paramsModes.put(paramStr, mode);
235             }
236             this.appendDescription(mode.getParameterDesc(_paramCache),
237                                    mode.getParameterList(_paramCache),
238                                    null);
239         }
240 
241         ////////////////////////////////////////////////////////////////////////
242         // update checks
243 
244         for (final UpdateCheck_mxJPO updateCheck : UpdateCheck_mxJPO.values())  {
245             for (final String param : updateCheck.getParameterList(_paramCache))  {
246                 if (param.length() == 1)  {
247                     this.paramsUpdateChecks.put("-" + param, updateCheck);
248                 } else  {
249                     this.paramsUpdateChecks.put("--" + param, updateCheck);
250                 }
251             }
252             this.appendDescription(updateCheck.getParameterDesc(_paramCache),
253                                    updateCheck.getParameterList(_paramCache),
254                                    null);
255         }
256 
257         ////////////////////////////////////////////////////////////////////////
258         // type definitions
259 
260         // first create list all type definitions which could be used...
261         final Set<TypeDef_mxJPO> all = new HashSet<TypeDef_mxJPO>();
262         for (final TypeDef_mxJPO typeDef : _paramCache.getMapping().getAllTypeDefs())  {
263             if (!typeDef.isBusCheckExists() || typeDef.existsBusType(_paramCache))  {
264                 all.add(typeDef);
265                 if (typeDef.getParameterList() != null)  {
266                     this.defineParameter(null,
267                             typeDef,
268                             typeDef.getParameterDesc(),
269                             typeDef.getParameterList());
270                 }
271                 if (typeDef.getParameterListOpp() != null)  {
272                     for (final String param : typeDef.getParameterListOpp())  {
273                         final String paramStr = (param.length() == 1)
274                                                 ? "-" + param
275                                                 : "--" + param;
276                         this.paramsTypeDefsOpp.put(paramStr, typeDef);
277                     }
278                     // store description
279                     this.appendDescription(typeDef.getParameterDescOpp(),
280                             typeDef.getParameterListOpp(),
281                             Arrays.asList(new String[]{"MATCH"}));
282                 }
283             }
284         }
285 
286         // define type definition group parameters depending on existing type
287         // definitions (and only if at minimum one type definition of the
288         // groups exists...)
289         for (final TypeDefGroup_mxJPO group : _paramCache.getMapping().getAllTypeDefGroups())  {
290             final Set<TypeDef_mxJPO> curTypeDefs = new HashSet<TypeDef_mxJPO>();
291             for (final String typeDefName : group.getTypeDefList())  {
292                 final TypeDef_mxJPO typeDef = _paramCache.getMapping().getTypeDef(typeDefName);
293                 if (all.contains(typeDef))  {
294                     curTypeDefs.add(typeDef);
295                 }
296             }
297             if (!curTypeDefs.isEmpty())  {
298                 this.defineParameter(curTypeDefs,
299                                      null,
300                                      group.getParameterDesc(),
301                                      group.getParameterList());
302             }
303         }
304    }

306     /***
307      *
308      * @param _paramsList       to which parameter list must the parameter
309      *                          appended
310      * @param _clazz            class implementing the import / export (or
311      *                          <code>null</code> if the complete parameter
312      *                          list must be used)
313      * @param _description      description of the parameter
314      * @param _longParams       list of long parameters strings
315      */
316     private void defineParameter(final Collection<TypeDef_mxJPO> _paramsList,
317                                  final TypeDef_mxJPO _clazz,
318                                  final String _description,
319                                  final Collection<String> _longParams)
320     {
321         final Collection<TypeDef_mxJPO> tmp;
322         if (_clazz == null)  {
323             tmp = _paramsList;
324         } else  {
325             tmp = new HashSet<TypeDef_mxJPO>();
326             // add to given set of parameters
327             if (_paramsList != null)  {
328                 _paramsList.add(_clazz);
329             }
330             tmp.add(_clazz);
331         }
332 
333         // check for long parameters and test for double definition
334         for (final String param : _longParams)  {
335             final String paramStr = (param.length() == 1)
336                                     ? "-" + param
337                                     : "--" + param;
338             this.paramsTypeDefs.put(paramStr, tmp);
339         }
340 
341         // store description
342         this.appendDescription(_description,
343                                _longParams,
344                                Arrays.asList(new String[]{"MATCH"}));
345     }

347     /***
348      * Appends a description for a defined list of parameters. If a parameter
349      * is defined twice, an error is thrown.
350      *
351      * @param _description  description to append
352      * @param _params       related parameters
353      * @param _args         text of the arguments for the list of parameters
354      *                      (or <code>null</code> if not defined)
355      * @see #allParams
356      * @see #description
357      */
358     private void appendDescription(final CharSequence _description,
359                                    final Collection<String> _params,
360                                    final Collection<String> _args)
361     {
362         // check for double parameter definitions
363         for (final String param : _params)  {
364             if (this.allParams.contains(param))  {
365                 throw new Error("double definition of parameter '" + param
366                         + "' with description '" + _description + "'");
367             }
368             this.allParams.add(param);
369         }
370 
371         // check if first parameter is not a short parameter
372         final String firstParam = _params.iterator().next();
373         final StringBuilder param = new StringBuilder();
374         if (firstParam.length() > 1)  {
375             param.append('-').append(firstParam.charAt(0)).append('_');
376         }
377 
378 
379         // append all parameters to the description text
380         boolean first = true;
381         for (final String paramString : _params)  {
382             if (first)  {
383                 first = false;
384             } else  {
385                 param.append(',');
386             }
387             if (paramString.length() > 1)  {
388                 param.append('-');
389             }
390             param.append('-').append(paramString);
391         }
392 
393         // append arguments
394         if (_args != null)  {
395             for (final String arg : _args)  {
396                 param.append(" <").append(arg).append('>');
397             }
398         }
399 
400         if (_description == null)  {
401             throw new Error("descriptions for parameter " + param + " not defined!");
402         }
403 
404         this.description.put(param.toString(), _description.toString());
405     }

407     public void mxMain(final Context _context,
408                        final String... _args)
409             throws Exception
410     {
411         try {
412             final ParameterCache_mxJPO paramCache = new ParameterCache_mxJPO(_context, false);
413 
414             this.prepareParams(paramCache);
415 
416             // to be sure....
417             MqlUtil_mxJPO.execMql(_context, "verbose off", false);
418 
419             Mode_mxJPO mode = null;
420 
421             final Map<Collection<TypeDef_mxJPO>,List<String>> clazz2matches
422                     = new HashMap<Collection<TypeDef_mxJPO>,List<String>>();
423             final Map<TypeDef_mxJPO,List<String>> clazz2matchesOpp
424                     = new HashMap<TypeDef_mxJPO,List<String>>();
425 
426             boolean unknown = false;
427 
428             UpdateCheck_mxJPO versionInfo = null;
429 
430 
431             for (int idx = 0; idx < _args.length; idx++)  {
432                 final String arg = _args[idx];
433                 if (this.paramsTypeDefs.containsKey(arg))  {
434                     final Collection<TypeDef_mxJPO> clazzes = this.paramsTypeDefs.get(arg);
435                     final String name = _args[++idx];
436                     List<String> names = clazz2matches.get(clazzes);
437                     if (names == null)  {
438                         names = new ArrayList<String>();
439                         clazz2matches.put(clazzes, names);
440                     }
441                     names.add(name);
442                 } else if (this.paramsTypeDefsOpp.containsKey(arg))  {
443                     final TypeDef_mxJPO typeDef = this.paramsTypeDefsOpp.get(arg);
444                     final String name = _args[++idx];
445                     List<String> names = clazz2matchesOpp.get(typeDef);
446                     if (names == null)  {
447                         names = new ArrayList<String>();
448                         clazz2matchesOpp.put(typeDef, names);
449                     }
450                     names.add(name);
451                 } else if (this.paramsModes.containsKey(arg)) {
452                     final Mode_mxJPO tmpMode = this.paramsModes.get(arg);
453                     if (mode != null)  {
454                         if ((mode == Mode_mxJPO.HELP) || (tmpMode == Mode_mxJPO.HELP))  {
455                             mode = Mode_mxJPO.HELP;
456                         } else  {
457                             throw new Error("A mode is already defined and could not be defined twice!");
458                         }
459                     } else  {
460                         mode = tmpMode;
461                     }
462                 } else if (this.paramsParameters.containsKey(arg))  {
463                     idx = paramCache.evalParameter(this.paramsParameters.get(arg),
464                                                    _args,
465                                                    idx);
466                 } else if (this.paramsUpdateChecks.containsKey(arg))  {
467                     versionInfo = this.paramsUpdateChecks.get(arg);
468                 } else  {
469                     unknown = true;
470                     paramCache.logError("unknown pararameter "  + arg);
471                 }
472             }
473 
474             if (unknown || (Mode_mxJPO.HELP == mode) || (mode == null))  {
475                 this.printHelp(paramCache);
476             } else if (Mode_mxJPO.EXPORT == mode)  {
477                 this.export(paramCache, clazz2matches, clazz2matchesOpp);
478             } else if (Mode_mxJPO.IMPORT == mode)  {
479                 this.update(paramCache, clazz2matches, clazz2matchesOpp, versionInfo);
480             } else if (Mode_mxJPO.DELETE == mode)  {
481                 this.delete(paramCache, clazz2matches, clazz2matchesOpp);
482             }
483 
484         } catch (final Exception e)  {
485             e.printStackTrace(System.out);
486             throw e;
487         }
488 
489     }

491     /***
492      * Prints the help for the MxUpdate functionality.
493      *
494      * @param _paramCache   parameter cache
495      * @throws IOException if the help could not be printed
496      * @see #description
497      */
498     private void printHelp(final ParameterCache_mxJPO _paramCache)
499             throws IOException
500     {
501         final String prefix = _paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_PREFIX);
502         final int lengthLine = _paramCache.getValueInteger(MxUpdate_mxJPO.PARAM_HELP_LENGTHLINE);
503         final int lengthParam = _paramCache.getValueInteger(MxUpdate_mxJPO.PARAM_HELP_LENGTHPARAMS);
504         final Appendable out = System.out;
505 
506         // first print the usage text with the mode parameters
507         out.append('\n')
508            .append(prefix)
509            .append(_paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_USAGE).trim())
510            .append(' ');
511         boolean first = true;
512         for (final Mode_mxJPO mode : Mode_mxJPO.values())  {
513             final String param = mode.getParameterList(_paramCache).iterator().next();
514             if (first)  {
515                 first = false;
516             } else  {
517                 out.append(" | ");
518             }
519             if (param.length() > 1)  {
520                 out.append('-');
521             }
522             out.append('-');
523             out.append(param);
524         }
525         out.append(" | ... \n\n");
526 
527         // print all parameters with description
528         for (final Map.Entry<String,String> descLine : this.description.entrySet())  {
529 
530             StringBuilder line = new StringBuilder().append(prefix);
531 
532             // append list of parameters
533             if ((descLine.getKey().length() > 3) && (descLine.getKey().charAt(2) == '_'))  {
534                 line.append("   ").append(descLine.getKey().substring(3));
535             } else  {
536                 line.append(descLine.getKey());
537             }
538 
539             // append spaces behind parameter list
540             if (line.length() > lengthParam)  {
541                 out.append(line).append('\n');
542                 line = new StringBuilder().append(prefix);
543             }
544             for (int i = line.length(); i < lengthParam; i++)  {
545                 line.append(' ');
546             }
547 
548             // append parameter description
549             first = true;
550             for (final String partDesc : descLine.getValue().toString().split("\n"))  {
551                 if (first == true)  {
552                     first = false;
553                 } else  {
554                     out.append(line).append('\n');
555                     line = new StringBuilder().append(prefix);
556                     for (int i = 1; i < lengthParam; i++)  {
557                         line.append(' ');
558                     }
559                 }
560                 for (final String desc : partDesc.split(" "))  {
561                     if (!"".equals(desc))  {
562                         if (line.length() > lengthLine)  {
563                             out.append(line).append('\n');
564                             line = new StringBuilder().append(prefix);
565                             for (int i = 1; i < lengthParam; i++)  {
566                                 line.append(' ');
567                             }
568                         }
569                         line.append(' ').append(desc);
570                     }
571                 }
572             }
573             out.append(line).append('\n');
574         }
575 
576         // print copyright
577         for (final String copyright : _paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_COPYRIGHT).split("\n"))  {
578             out.append(prefix).append(copyright).append('\n');
579         }
580         out.append('\n');
581     }

583     /***
584      * Exports matching administration objects to given path.
585      *
586      * @param _paramCache       parameter cache
587      * @param _clazz2matches    map of update classes and the depending
588      *                          list of string which must match to export
589      * @param _clazz2matchesOpp map of update classes and the depending list of
590      *                          match string of MX names which must be excluded
591      *                          from export
592      * @throws Exception if none path or more than one path is defined or if
593      *                   the export failed
594      */
595     protected void export(final ParameterCache_mxJPO _paramCache,
596                           final Map<Collection<TypeDef_mxJPO>,List<String>> _clazz2matches,
597                           final Map<TypeDef_mxJPO,List<String>> _clazz2matchesOpp)
598             throws Exception
599     {
600         final Collection<String> paths = _paramCache.getValueList(MxUpdate_mxJPO.PARAM_PATH);
601 
602         // check for definition of min. / max. one path
603         if (paths.isEmpty())  {
604             throw new Exception("no path is defined, but required for the export!");
605         }
606         if (paths.size() > 1)  {
607             throw new Exception("more than one path is defined, but maximum is allowed for the export!");
608         }
609         final String pathStr = paths.iterator().next();
610 
611         // evaluate all matching administration objects
612         final Map<TypeDef_mxJPO,Set<String>> clazz2names
613                 = this.getMatching(_paramCache, _clazz2matches, _clazz2matchesOpp);
614 
615         // export
616         for (final Map.Entry<TypeDef_mxJPO,Set<String>> entry : clazz2names.entrySet())  {
617             for (final String name : entry.getValue())  {
618                 final AbstractObject_mxJPO instance = entry.getKey().newTypeInstance(name);
619                 final File path = new File(pathStr + File.separator + instance.getPath());
620                 _paramCache.logInfo("export "+instance.getTypeDef().getLogging() + " '" + name + "'");
621                 instance.export(_paramCache, path);
622             }
623         }
624     }

626     /***
627      *
628      * @param _paramCache       parameter cache
629      * @param _clazz2matches    all classes with the depending matches (if
630      *                          paths are defined, the names must match against
631      *                          the MX name; otherwise the found files must
632      *                          match)
633      * @param _clazz2matchesOpp all classes and the depending matches for the
634      *                          files which must be excluded from the matches
635      *                          defined in <code>_class2matched</code>
636      * @param _versionInfo      defines how it is checked if an update is
637      *                          required (see {@link UpdateCheck_mxJPO})
638      * @throws Exception if update failed
639      * @see UpdateCheck_mxJPO
640      */
641     protected void update(final ParameterCache_mxJPO _paramCache,
642                           final Map<Collection<TypeDef_mxJPO>,List<String>> _clazz2matches,
643                           final Map<TypeDef_mxJPO,List<String>> _clazz2matchesOpp,
644                           final UpdateCheck_mxJPO _versionInfo)
645             throws Exception
646     {
647         // get all matching files depending on the update classes
648         final Map<TypeDef_mxJPO,Map<File,String>> clazz2names
649                 = this.evalMatches(_paramCache, _clazz2matches, _clazz2matchesOpp);
650 
651         // evaluate for existing administration objects
652         final Collection<String> wildCardMatch = new HashSet<String>();
653         wildCardMatch.add("*");
654         final Map<TypeDef_mxJPO,Set<String>> existingNames = new HashMap<TypeDef_mxJPO,Set<String>>();
655         for (final TypeDef_mxJPO clazz : clazz2names.keySet())  {
656             if (!existingNames.containsKey(clazz))  {
657                 final AbstractObject_mxJPO instance = clazz.newTypeInstance(null);
658                 existingNames.put(clazz, instance.getMxNames(_paramCache));
659             }
660         }
661         // create if needed (and not in the list of existing objects
