177     /***
178      * Prepares the parameters depending on the <code>_paramCache</code>.
179      *
180      * @param _paramCache       parameter cache
181      * @throws MatrixException if description or parameters could not be
182      *                         prepared
183      */
184     private void prepareParams(final ParameterCache_mxJPO _paramCache)
185         throws MatrixException
186     {
187         this.description.clear();
188         this.allParams.clear();
189         this.paramsModes.clear();
190         this.paramsParameters.clear();
191         this.paramsTypeDefs.clear();
192 
193         ////////////////////////////////////////////////////////////////////////
194         // parameters
195 
196         for (final ParameterDef_mxJPO parameter : _paramCache.getMapping().getAllParameterDefs())  {
197             if (parameter.getParameterList() != null)  {
198                 for (final String param : parameter.getParameterList())  {
199                     final String paramStr = (param.length() > 1)
200                                             ? "--" + param
201                                             : "-" + param;
202                     this.paramsParameters.put(paramStr.toLowerCase(), parameter);
203                 }
204                 final StringBuilder desc = new StringBuilder().append(parameter.getParameterDesc());
205                 if ((parameter.getDefaultValue() != null) && (parameter.getType() != ParameterDef_mxJPO.Type.BOOLEAN))  {
206                     desc.append('\n').append("(Default '");
207                     if (parameter.getType() == ParameterDef_mxJPO.Type.LIST)  {
208                         desc.append(parameter.getDefaultValue().replaceAll(",", ", "));
209                     } else  {
210                         desc.append(parameter.getDefaultValue());
211                     }
212                     desc.append("')");
213                 }
214                 this.appendDescription(desc,
215                                        parameter.getParameterList(),
216                                        parameter.getParameterArgs());
217             }
218         }
219 
220         ////////////////////////////////////////////////////////////////////////
221         // modes
222 
223         for (final Mode_mxJPO mode : Mode_mxJPO.values())  {
224             for (final String param : mode.getParameterList(_paramCache))  {
225                 final String paramStr = (param.length() > 1)
226                                         ? "--" + param
227                                         : "-" + param;
228                 this.paramsModes.put(paramStr.toLowerCase(), mode);
229             }
230             this.appendDescription(mode.getParameterDesc(_paramCache),
231                                    mode.getParameterList(_paramCache),
232                                    null);
233         }
234 
235         ////////////////////////////////////////////////////////////////////////
236         // type definitions
237 
238         // first create list all type definitions which could be used...
239         final Set<TypeDef_mxJPO> all = new HashSet<TypeDef_mxJPO>();
240         for (final TypeDef_mxJPO typeDef : _paramCache.getMapping().getAllTypeDefsSorted())  {
241             if (!typeDef.isBusCheckExists() || typeDef.existsBusType(_paramCache))  {
242                 all.add(typeDef);
243                 if (typeDef.getParameterList() != null)  {
244                     this.defineParameter(null,
245                             typeDef,
246                             typeDef.getParameterDesc(),
247                             typeDef.getParameterList());
248                 }
249                 if (typeDef.getParameterListOpp() != null)  {
250                     for (final String param : typeDef.getParameterListOpp())  {
251                         final String paramStr = (param.length() == 1)
252                                                 ? "-" + param
253                                                 : "--" + param;
254                         this.paramsTypeDefsOpp.put(paramStr.toLowerCase(), typeDef);
255                     }
256                     // store description
257                     this.appendDescription(typeDef.getParameterDescOpp(),
258                             typeDef.getParameterListOpp(),
259                             Arrays.asList(new String[]{"MATCH"}));
260                 }
261             }
262         }
263 
264         // define type definition group parameters depending on existing type
265         // definitions (and only if at minimum one type definition of the
266         // groups exists...)
267         for (final TypeDefGroup_mxJPO group : _paramCache.getMapping().getAllTypeDefGroups())  {
268             final Set<TypeDef_mxJPO> curTypeDefs = new HashSet<TypeDef_mxJPO>();
269             for (final String typeDefName : group.getTypeDefList())  {
270                 final TypeDef_mxJPO typeDef = _paramCache.getMapping().getTypeDef(typeDefName);
271                 if (all.contains(typeDef))  {
272                     curTypeDefs.add(typeDef);
273                 }
274             }
275             if (!curTypeDefs.isEmpty())  {
276                 this.defineParameter(curTypeDefs,
277                                      null,
278                                      group.getParameterDesc(),
279                                      group.getParameterList());
280             }
281         }
282    }

284     /***
285      *
286      * @param _paramsList       to which parameter list must the parameter
287      *                          appended
288      * @param _clazz            class implementing the import / export (or
289      *                          <code>null</code> if the complete parameter
290      *                          list must be used)
291      * @param _description      description of the parameter
292      * @param _longParams       list of long parameters strings
293      */
294     private void defineParameter(final Collection<TypeDef_mxJPO> _paramsList,
295                                  final TypeDef_mxJPO _clazz,
296                                  final String _description,
297                                  final Collection<String> _longParams)
298     {
299         final Collection<TypeDef_mxJPO> tmp;
300         if (_clazz == null)  {
301             tmp = _paramsList;
302         } else  {
303             tmp = new HashSet<TypeDef_mxJPO>();
304             // add to given set of parameters
305             if (_paramsList != null)  {
306                 _paramsList.add(_clazz);
307             }
308             tmp.add(_clazz);
309         }
310 
311         // check for long parameters and test for double definition
312         for (final String param : _longParams)  {
313             final String paramStr = (param.length() == 1)
314                                     ? "-" + param
315                                     : "--" + param;
316             this.paramsTypeDefs.put(paramStr.toLowerCase(), tmp);
317         }
318 
319         // store description
320         this.appendDescription(_description,
321                                _longParams,
322                                Arrays.asList(new String[]{"MATCH"}));
323     }

325     /***
326      * Appends a description for a defined list of parameters. If a parameter
327      * is defined twice, an error is thrown.
328      *
329      * @param _description  description to append
330      * @param _params       related parameters
331      * @param _args         text of the arguments for the list of parameters
332      *                      (or <code>null</code> if not defined)
333      * @see #allParams
334      * @see #description
335      */
336     private void appendDescription(final CharSequence _description,
337                                    final Collection<String> _params,
338                                    final Collection<String> _args)
339     {
340         // check for double parameter definitions
341         for (final String param : _params)  {
342             if (this.allParams.contains(param))  {
343                 throw new Error("double definition of parameter '" + param
344                         + "' with description '" + _description + "'");
345             }
346             this.allParams.add(param);
347         }
348 
349         // check if first parameter is not a short parameter
350         final String firstParam = _params.iterator().next();
351         final StringBuilder param = new StringBuilder();
352         if (firstParam.length() > 1)  {
353             param.append('-').append(firstParam.charAt(0)).append('_');
354         }
355 
356 
357         // append all parameters to the description text
358         boolean first = true;
359         for (final String paramString : _params)  {
360             if (first)  {
361                 first = false;
362             } else  {
363                 param.append(',');
364             }
365             if (paramString.length() > 1)  {
366                 param.append('-');
367             }
368             param.append('-').append(paramString);
369         }
370 
371         // append arguments
372         if (_args != null)  {
373             for (final String arg : _args)  {
374                 param.append(" <").append(arg).append('>');
375             }
376         }
377 
378         if (_description == null)  {
379             throw new Error("descriptions for parameter " + param + " not defined!");
380         }
381 
382         this.description.put(param.toString(), _description.toString());
383     }

385     public void mxMain(final Context _context,
386                        final String... _args)
387             throws Exception
388     {
389         try {
390             final ParameterCache_mxJPO paramCache = new ParameterCache_mxJPO(_context, false);
391 
392             this.prepareParams(paramCache);
393 
394             // to be sure....
395             MqlUtil_mxJPO.execMql(_context, "verbose off", false);
396 
397             Mode_mxJPO mode = null;
398 
399             final Map<Collection<TypeDef_mxJPO>,List<String>> clazz2matches
400                     = new HashMap<Collection<TypeDef_mxJPO>,List<String>>();
401             final Map<TypeDef_mxJPO,List<String>> clazz2matchesOpp
402                     = new HashMap<TypeDef_mxJPO,List<String>>();
403 
404             boolean unknown = false;
405 
406             for (int idx = 0; idx < _args.length; idx++)  {
407                 final String arg = _args[idx].toLowerCase();
408                 if (this.paramsTypeDefs.containsKey(arg))  {
409                     final Collection<TypeDef_mxJPO> clazzes = this.paramsTypeDefs.get(arg);
410                     final String name = _args[++idx];
411                     List<String> names = clazz2matches.get(clazzes);
412                     if (names == null)  {
413                         names = new ArrayList<String>();
414                         clazz2matches.put(clazzes, names);
415                     }
416                     names.add(name);
417                 } else if (this.paramsTypeDefsOpp.containsKey(arg))  {
418                     final TypeDef_mxJPO typeDef = this.paramsTypeDefsOpp.get(arg);
419                     final String name = _args[++idx];
420                     List<String> names = clazz2matchesOpp.get(typeDef);
421                     if (names == null)  {
422                         names = new ArrayList<String>();
423                         clazz2matchesOpp.put(typeDef, names);
424                     }
425                     names.add(name);
426                 } else if (this.paramsModes.containsKey(arg)) {
427                     final Mode_mxJPO tmpMode = this.paramsModes.get(arg);
428                     if (mode != null)  {
429                         if ((mode == Mode_mxJPO.HELP) || (tmpMode == Mode_mxJPO.HELP))  {
430                             mode = Mode_mxJPO.HELP;
431                         } else  {
432                             throw new Error("A mode is already defined and could not be defined twice!");
433                         }
434                     } else  {
435                         mode = tmpMode;
436                     }
437                 } else if (this.paramsParameters.containsKey(arg))  {
438                     idx = paramCache.evalParameter(this.paramsParameters.get(arg),
439                                                    _args,
440                                                    idx);
441                 } else  {
442                     unknown = true;
443                     paramCache.logError("unknown pararameter "  + arg);
444                 }
445             }
446 
447             if (unknown || (Mode_mxJPO.HELP == mode) || (mode == null))  {
448                 this.printHelp(paramCache);
449             } else if (Mode_mxJPO.EXPORT == mode)  {
450                 this.export(paramCache, clazz2matches, clazz2matchesOpp);
451             } else if (Mode_mxJPO.IMPORT == mode)  {
452                 this.update(paramCache, clazz2matches, clazz2matchesOpp);
453             } else if (Mode_mxJPO.DELETE == mode)  {
454                 this.delete(paramCache, clazz2matches, clazz2matchesOpp);
455             }
456 
457         } catch (final Exception e)  {
458             e.printStackTrace(System.out);
459             throw e;
460         }
461 
462     }

464     /***
465      * Prints the help for the MxUpdate functionality.
466      *
467      * @param _paramCache   parameter cache
468      * @throws IOException if the help could not be printed
469      * @see #description
470      */
471     private void printHelp(final ParameterCache_mxJPO _paramCache)
472         throws IOException
473     {
474         final String prefix = _paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_PREFIX);
475         final int lengthLine = _paramCache.getValueInteger(MxUpdate_mxJPO.PARAM_HELP_LENGTHLINE);
476         final int lengthParam = _paramCache.getValueInteger(MxUpdate_mxJPO.PARAM_HELP_LENGTHPARAMS);
477         final Appendable out = System.out;
478 
479         // first print the usage text with the mode parameters
480         out.append('\n')
481            .append(prefix)
482            .append(_paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_USAGE).trim())
483            .append(' ');
484         boolean first = true;
485         for (final Mode_mxJPO mode : Mode_mxJPO.values())  {
486             final String param = mode.getParameterList(_paramCache).iterator().next();
487             if (first)  {
488                 first = false;
489             } else  {
490                 out.append(" | ");
491             }
492             if (param.length() > 1)  {
493                 out.append('-');
494             }
495             out.append('-');
496             out.append(param);
497         }
498         out.append(" | ... \n")
499            .append(_paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_DESCRIPTION));
500 
501         // print all parameters with description
502         for (final Map.Entry<String,String> descLine : this.description.entrySet())  {
503 
504             StringBuilder line = new StringBuilder().append(prefix);
505 
506             // append list of parameters
507             if ((descLine.getKey().length() > 3) && (descLine.getKey().charAt(2) == '_'))  {
508                 line.append("   ").append(descLine.getKey().substring(3));
509             } else  {
510                 line.append(descLine.getKey());
511             }
512 
513             // append spaces behind parameter list
514             if (line.length() > lengthParam)  {
515                 out.append(line).append('\n');
516                 line = new StringBuilder().append(prefix);
517             }
518             for (int i = line.length(); i < lengthParam; i++)  {
519                 line.append(' ');
520             }
521 
522             // append parameter description
523             first = true;
524             for (final String partDesc : descLine.getValue().toString().split("\n"))  {
525                 if (first == true)  {
526                     first = false;
527                 } else  {
528                     out.append(line).append('\n');
529                     line = new StringBuilder().append(prefix);
530                     for (int i = 0; i < lengthParam; i++)  {
531                         line.append(' ');
532                     }
533                 }
534                 for (final String desc : partDesc.split(" "))  {
535                     if (!"".equals(desc))  {
536                         if (line.length() > lengthLine)  {
537                             out.append(line).append('\n');
538                             line = new StringBuilder().append(prefix);
539                             for (int i = 0; i < lengthParam; i++)  {
540                                 line.append(' ');
541                             }
542                         }
543                         line.append(' ').append(desc);
544                     }
545                 }
546             }
547             out.append(line).append('\n');
548         }
549 
550         // print copyright
551         for (final String copyright : _paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_COPYRIGHT).split("\n"))  {
552             out.append(prefix).append(copyright).append('\n');
553         }
554         out.append('\n');
555     }

557     /***
558      * Exports matching administration objects to given path.
559      *
560      * @param _paramCache       parameter cache
561      * @param _clazz2matches    map of update classes and the depending
562      *                          list of string which must match to export
563      * @param _clazz2matchesOpp map of update classes and the depending list of
564      *                          match string of MX names which must be excluded
565      *                          from export
566      * @throws Exception if none path or more than one path is defined or if
567      *                   the export failed
568      */
569     protected void export(final ParameterCache_mxJPO _paramCache,
570                           final Map<Collection<TypeDef_mxJPO>,List<String>> _clazz2matches,
571                           final Map<TypeDef_mxJPO,List<String>> _clazz2matchesOpp)
572         throws Exception
573     {
574         final Collection<String> paths = _paramCache.getValueList(MxUpdate_mxJPO.PARAM_PATH);
575 
576         // check for definition of min. / max. one path
577         if (paths.isEmpty())  {
578             throw new Exception("no path is defined, but required for the export!");
579         }
580         if (paths.size() > 1)  {
581             throw new Exception("more than one path is defined, but maximum is allowed for the export!");
582         }
583         final String pathStr = paths.iterator().next();
584 
585         // evaluate all matching administration objects
586         final Map<TypeDef_mxJPO,Set<String>> clazz2names
587                 = this.getMatching(_paramCache, _clazz2matches, _clazz2matchesOpp);
588 
589         // export
590         for (final Map.Entry<TypeDef_mxJPO,Set<String>> entry : clazz2names.entrySet())  {
591             for (final String name : entry.getValue())  {
592                 final AbstractObject_mxJPO instance = entry.getKey().newTypeInstance(name);
593                 final File path = new File(pathStr + File.separator + instance.getPath());
594                 _paramCache.logInfo("export "+instance.getTypeDef().getLogging() + " '" + name + "'");
595                 instance.export(_paramCache, path);
596             }
597         }
598     }

623     /***
624      *
625      * @param _paramCache       parameter cache
626      * @param _clazz2matches    map of update classes and the depending
627      *                          list of string which must match to delete
628      * @param _clazz2matchesOpp map of update classes and the depending list of
629      *                          match string of MX names which must be excluded
630      *                          from deletion
631      * @throws Exception if the delete of administration objects failed
632      */
633     protected void delete(final ParameterCache_mxJPO _paramCache,
634                           final Map<Collection<TypeDef_mxJPO>,List<String>> _clazz2matches,
635                           final Map<TypeDef_mxJPO,List<String>> _clazz2matchesOpp)
636             throws Exception
637     {
638         // check for definition of min. / max. one path
639         if (_paramCache.getValueList(MxUpdate_mxJPO.PARAM_PATH).isEmpty())  {
640             throw new Exception("no path is defined, but required for the delete!");
641         }
642 
643         // evaluate all matching administration objects
644         final Map<TypeDef_mxJPO,Set<String>> clazz2MxNames
645                 = this.getMatching(_paramCache, _clazz2matches, _clazz2matchesOpp);
646 
647         // get all matching files depending on the update classes
648         final Map<TypeDef_mxJPO,Map<File,String>> clazz2FileNames
649                 = this.evalMatches(_paramCache, _clazz2matches, _clazz2matchesOpp);
650 
651         // and now loop throw the list of file names and compare to existing
652         for (final Map.Entry<TypeDef_mxJPO,Set<String>> entry : clazz2MxNames.entrySet())  {
653             final Collection<String> fileNames = clazz2FileNames.containsKey(entry.getKey())
654                                                  ? clazz2FileNames.get(entry.getKey()).values()
655                                                  : null;

731                                                                      entry.getKey(),

734                                                                      allFiles,
735                                                                      matchedFiles,
736                                                                      null);
737                 // get all matching files depending on the update classes
738                 // (and LAST file match)
739                 if (lastFileMatchNeeded)  {

741                     for (final File file : allFiles)  {

747                                      entry.getKey(),

753                 }
754             }

756         } else  {
757             /// get all files
758             final Set<File> allFiles = new HashSet<File>();
759             for (final String path : paths)  {
760                 allFiles.addAll(this.getAllFiles(new File(path), ignorePaths, ignoreFiles));
761             }
762             final Set<File> matchedFiles = new HashSet<File>();
763             // get all matching files depending on the update classes

793             }
794         }

797         final Map<TypeDef_mxJPO,Map<File,String>> ret = new HashMap<TypeDef_mxJPO,Map<File,String>>();

804                     for (final String matchOpp : matchOpps)  {
805                         if (StringUtil_mxJPO.match(fileEntry.getValue(), matchOpp))  {
806                             allowed = false;
807                             break;
808                         }
809                     }
810                     if (allowed)  {
811                         files.put(fileEntry.getKey(), fileEntry.getValue());
812                     }
813                 }
814                 ret.put(entry.getKey(), files);
815             } else  {
816                 ret.put(entry.getKey(), entry.getValue());
817             }
818         }
