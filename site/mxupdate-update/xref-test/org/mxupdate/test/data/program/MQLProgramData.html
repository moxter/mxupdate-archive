21  package org.mxupdate.test.data.program;
22  
23  import matrix.util.MatrixException;
24  
25  import org.mxupdate.test.AbstractTest;
26  
27  /***
28   * The class is used to define all MQL program objects used to create / update
29   * and to export.
30   *
31   * @author The MxUpdate Team
32   * @version $Id: MQLProgramData.java 594 2009-10-29 00:37:31Z tim.moxter $
33   */
34  public class MQLProgramData
35      extends AbstractProgramData<MQLProgramData>
36  {
37      /***
38       * Start marker for the update code.
39       */
40      private static final String MARK_START
41              = "################################################################################\n"
42              + "# START NEEDED MQL UPDATE FOR THIS PROGRAM                                     #\n"
43              + "################################################################################";
44  
45      /***
46       * End marker for the update code.
47       */
48      private static final String MARK_END
49              = "################################################################################\n"
50              + "# END NEEDED MQL UPDATE FOR THIS PROGRAM                                       #\n"
51              + "################################################################################";
52  
53      /***
54       * Initializes this MQL program.
55       *
56       * @param _test     related test instance
57       * @param _name     name of the MQL program
58       */
59      public MQLProgramData(final AbstractTest _test,
60                            final String _name)
61      {
62          super(_test, AbstractTest.CI.PRG_MQL_PROGRAM, _name);
63      }
64  
65      /***
66       * Creates this MQL program within MX.
67       *
68       * @return this MQL program instance
69       * @throws MatrixException if create of the MQL program failed
70       */
71      @Override()
72      public MQLProgramData create()
73          throws MatrixException
74      {
75          if (!this.isCreated())  {
76              final StringBuilder cmd = new StringBuilder()
77                      .append("escape add program \"").append(AbstractTest.convertMql(this.getName()))
78                      .append("\" mql ")
79                      .append("code \"").append(AbstractTest.convertMql(this.getCode())).append("\"");
80              this.append4Create(cmd);
81              cmd.append(";\n")
82                 .append("escape add property ").append(this.getSymbolicName())
83                 .append(" on program eServiceSchemaVariableMapping.tcl")
84                 .append(" to program \"").append(AbstractTest.convertMql(this.getName())).append("\"");
85  
86              this.getTest().mql(cmd);
87              this.setCreated(true);
88          }
89          return this;
90      }
91  
92      /***
93       * Returns the configuration item file name of this MQL program. The
94       * configuration item file name of a MQL program excludes the
95       * <code>.tcl</code> extension.
96       *
97       * @return file name of a JPO
98       */
99      @Override()
100     public String getCIFileName()
101     {
102         final String ciFileName = super.getCIFileName();
103         return ciFileName.replaceAll("//.tcl$", "");
104     }
105 
106     /***
107      * The related configuration item file is the {@link #code} of the program.
108      *
109      *  @return {@link #code} of the program
110      *  @see #code
111      */
112     @Override()
113     public String ciFile()
114     {
115         final StringBuilder cmd = new StringBuilder()
116            .append(MQLProgramData.MARK_START)
117            .append("\nmql escape mod program ${NAME} ");
118         this.append4CIFileValues(cmd);
119         cmd.append("\n")
120            .append(MQLProgramData.MARK_END)
121            .append("\n");
122 
123         final StringBuilder ciFile = new StringBuilder();
124         for (final String line : cmd.toString().split("\n"))  {
125             if (this.getName().endsWith(".tcl"))  {
126                 ciFile.append('#');
127             }
128             ciFile.append(line).append('\n');
129         }
130         ciFile.append('\n').append(this.getCode());
131 
132         return ciFile.toString();
133     }
