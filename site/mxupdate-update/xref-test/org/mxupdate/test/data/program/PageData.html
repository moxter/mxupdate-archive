33  /***
34   * The class is used to define all page program objects used to create / update
35   * and to export.
36   *
37   * @author The MxUpdate Team
38   * @version $Id: PageData.java 617 2009-11-02 02:15:16Z tim.moxter $

40  public class PageData
41      extends AbstractAdminData<PageData>
42  {
43      /***
44       * Within export the description and mime type must be defined.
45       *
46       * @see #PageData(AbstractTest, String)
47       */
48      private static final Set<String> REQUIRED_EXPORT_VALUES = new HashSet<String>(2);
49      static  {
50          PageData.REQUIRED_EXPORT_VALUES.add("description");
51          PageData.REQUIRED_EXPORT_VALUES.add("mime");
52      }
53  
54      /***
55       * Separator used between the configuration item update code the the
56       * page content itself.
57       */
58      private static final String PAGE_CONTENT_SEPARATOR
59              = "# do not change the next three lines, they are needed as separator information:\n"
60              + "################################################################################\n"
61              + "# PAGE CONTENT                                                                 #\n"
62              + "################################################################################";
63  
64      /***
65       * Related code of this program.
66       *
67       * @see #setContent(String)
68       * @see #getCode()
69       */
70      private String content = "";

120     /***
121      * The related configuration item file is the {@link #content} of the page.
122      *
123      *  @return {@link #content} of the page
124      *  @see #content
125      */
126     @Override()
127     public String ciFile()
128     {
129         final StringBuilder cmd = new StringBuilder()
130                 .append("mql escape mod page \"${NAME}\" file [file join \"${FILE}\"]");
131         this.append4CIFileValues(cmd);
132 
133         // append embedded inquiry code
134         if (this.content != null)  {
135             cmd.append("\n\n").append(PageData.PAGE_CONTENT_SEPARATOR).append('\n')
136                .append(this.content);
137         }
138 
139         return cmd.toString();
140     }

142     /***
143      * Creates this MQL program within MX.
144      *
145      * @return this MQL program instance
146      * @throws MatrixException if create of the MQL program failed
147      */
148     @Override()
149     public PageData create()
150         throws MatrixException
151     {
152         if (!this.isCreated())  {
153             this.setCreated(true);
154 
155             final StringBuilder cmd = new StringBuilder()
156                     .append("escape add page \"").append(AbstractTest.convertMql(this.getName())).append("\"");
157             this.append4Create(cmd);
158             // append content
159             if (this.content != null)  {
160                 cmd.append(" content \"").append(AbstractTest.convertMql(this.content)).append("\"");
161             }
162 
163             this.getTest().mql(cmd);
164         }
165         return this;
166     }

168     /***
169      * Checks the export of this page all values and the embedded page content
170      * are correct defined. The original method is overwritten to test the page
171      * content.
172      *
173      * @param _exportParser     parsed export
174      * @throws MatrixException if check failed
175      */
176     @Override()
177     public void checkExport(final ExportParser _exportParser)
178         throws MatrixException
179     {
180         super.checkExport(_exportParser);
181 
182         // check no symbolic name is defined
183         Assert.assertTrue((_exportParser.getSymbolicName() == null),
184                           "check that no symbolic name is defined");
185 
186         // check embedded inquiry code
187         final PageExportParser exportParser = (PageExportParser) _exportParser;
188         if (this.content == null)  {
189             Assert.assertEquals(exportParser.pageContent,
190                                 "",
191                                 "check that no embedded page content is defined");
192         } else  {
193             Assert.assertEquals(exportParser.pageContent,
194                                 this.content,
195                                 "check that correct embedded page content is defined");
196         }
197     }

213     /***
214      * The class overwrites the original export parser to handle the embedded
215      * page content within configuration item update files.
216      */
217     private static class PageExportParser
218         extends ExportParser
219     {
220         /***
221          * Defines the embedded page content within the configuration item
222          * update code.
223          *
224          * @see #extractUpdateCode(String)
225          */
226         private final String pageContent;
227 
228         /***
229          * Initializes the inquiry export parser.
230          *
231          * @param _ci       type of the configuration item
232          * @param _code     exported configuration item update code
233          */
234         public PageExportParser(final AbstractTest.CI _ci,
235                                    final String _code)
236         {
237             super(_ci, _code);
238             final int idx = _code.indexOf(PageData.PAGE_CONTENT_SEPARATOR);
239             if (idx >= 0)  {
240                 this.pageContent = _code.substring(idx + PageData.PAGE_CONTENT_SEPARATOR.length()).trim();
241             } else  {
242                 this.pageContent = "";
243             }
244         }
245 
246         /***
247          * Extracts from the <code>_origCode</code> the update code without
248          * header. Because the configuration item update code of an page
249          * includes also the content of the page, this page content must be
250          * removed from the original configuration item code so that the
251          * update code could be extracted correct.
252          *
253          * @param _origCode     original code from which the update code must
254          *                      be extracted
255          * @return extracted update code (without header and page content)
256          */
257         @Override()
258         protected String extractUpdateCode(final String _origCode)
259         {
260             final String origCode;
261             final int idx = _origCode.indexOf(PageData.PAGE_CONTENT_SEPARATOR);
262             if (idx >= 0)  {
263                 origCode = _origCode.substring(0, idx);
264             } else  {
265                 origCode = _origCode;
266             }
267             return super.extractUpdateCode(origCode);
268         }
269     }
