242     /***
243      * Appends the program related MQL commands to create this program. This
244      * includes:
245      * <ul>
246      * <li>{@link #user}</li>
247      * <li>{@link #isHidden() hidden flag}</li>
248      * <li>{@link #deferred}</li>
249      * <li>{@link #needsBusinessObject needs business object}</li>
250      * <li>{@link #downloadable} flag</li>
251      * <li>{@link #pipe} flag</li>
252      * <li>{@link #pooled} flag</li>
253      * </ul>
254      *
255      * @param _cmd  string builder used to append MQL commands
256      * @throws MatrixException if used object could not be created
257      */
258     @Override()
259     protected void append4Create(final StringBuilder _cmd)
260         throws MatrixException
261     {
262         super.append4Create(_cmd);
263 
264         // user
265         if (this.getUser() != null)  {
266             this.getUser().create();
267             _cmd.append(" execute user \"").append(AbstractTest.convertMql(this.getUser().getName())).append("\"");
268         }
269         // hidden flag
270         if ((this.isHidden() != null) && this.isHidden())  {
271             _cmd.append(" hidden");
272         }
273         // deferred flag
274         if (this.deferred)  {
275             _cmd.append(" execute deferred");
276         }
277         // needs business object
278         if (this.needsBusinessObject)  {
279             _cmd.append(" needsbusinessobject");
280         }
281         // downloable flag
282         if (this.downloadable)  {
283             _cmd.append(" downloadable");
284         }
285         // pipe flag
286         if (this.pipe)  {
287             _cmd.append(" pipe");
288         }
289         // pooled flag
290         if (this.pooled)  {
291             _cmd.append(" pooled");
292         }
293     }

345     /***
346      * Checks the export of this data piece if all values are correct defined.
347      * Because the handling of a program is different (the TCL update code
348      * is embedded), the values must be checked directly via MQL commands.
349      *
350      * @param _exportParser     parsed export
351      * @throws MatrixException if check failed
352      */
353     @Override()
354     public void checkExport(final ExportParser _exportParser)
355         throws MatrixException
356     {
357         Assert.assertTrue(!"".equals(this.getTest().mql("escape list program \"" + AbstractTest.convertMql(this.getName()) + "\"")),
358                           "check program is created");
359         Assert.assertTrue(!"".equals(this.getTest().mql("escape list property to program \"" + AbstractTest.convertMql(this.getName()) + "\"")),
360                           "check that the program is registered");
361         Assert.assertEquals(this.getTest().mql("escape list property to program \"" + AbstractTest.convertMql(this.getName()) + "\""),
362                             this.getSymbolicName()
363                                     + " on program eServiceSchemaVariableMapping.tcl to program "
364                                     + this.getName(),
365                             "check that the MQL program is registered with correct symbolic name");
366         // check hidden
367         Assert.assertEquals(Boolean.valueOf(this.getTest().mql("escape print program \""
368                                         + AbstractTest.convertMql(this.getName()) + "\" select hidden dump")).booleanValue(),
369                             (this.isHidden() != null) ? this.isHidden().booleanValue() : false,
370                             "hidden flag correct defined");
371         // check description
372         Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName()) + "\" select description dump"),
373                             (this.getValue("description") != null) ? this.getValue("description") : "",
374                             "correct description defined");
375         // check user
376         Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName()) + "\" select user dump"),
377                             (this.getUser() != null) ? this.getUser().getName() : "",
378                             "correct user assigned");
379         // execution is deferred
380         Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName()) + "\" select execute dump"),
381                             (this.deferred) ? "deferred" : "immediate",
382                             "check execution");
383         // needs business object
384         Assert.assertEquals(Boolean.valueOf(this.getTest().mql("escape print program \""
385                                     + AbstractTest.convertMql(this.getName()) + "\" select doesneedcontext dump")).booleanValue(),
386                             this.needsBusinessObject,
387                             "needs business object flag");
388         // downloadable
389         Assert.assertEquals(Boolean.valueOf(this.getTest().mql("escape print program \""
390                                     + AbstractTest.convertMql(this.getName()) + "\" select downloadable dump")).booleanValue(),
391                             this.downloadable,
392                             "downloadable flag");
393         // pipe
394         Assert.assertEquals(Boolean.valueOf(this.getTest().mql("escape print program \""
395                                     + AbstractTest.convertMql(this.getName()) + "\" select ispipedprogram dump")).booleanValue(),
396                             this.pipe,
397                             "pipe flag");
398 
399         // check properties
400         for (final PropertyDef prop : this.getProperties())  {
401             Assert.assertTrue(!"".equals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName())
402                                             + "\" select property[" + prop.getName() + "] dump")),
403                               "property '" + prop.getName() + "' is defined");
404             Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName())
405                                             + "\" select property[" + prop.getName() + "].value dump"),
406                                 (prop.getValue() != null) ? prop.getValue() : "",
407                                 "value of property '" + prop.getName() + "' is defined");
408             if (prop.getTo() != null)  {
409                 Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName())
410                                     + "\" select property[" + prop.getName() + "].to.name dump"),
411                                     prop.getTo().getName(),
412                                     "to of property '" + prop.getName() + "' is defined");
413             } else  {
414                 Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName())
415                                             + "\" select property[" + prop.getName() + "].to dump"),
416                                     "",
417                                     "to of property '" + prop.getName() + "' is not defined");
418             }
419         }
420     }
