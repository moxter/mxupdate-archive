21  package org.mxupdate.test.data.program;
22  
23  import org.mxupdate.test.AbstractTest;
24  import org.mxupdate.test.AbstractTest.CI;
25  import org.mxupdate.test.data.AbstractData;
26  
27  /***
28   * The class is used to define all program objects used to create / update and
29   * to export.
30   *
31   * @author The MxUpdate Team
32   * @version $Id: AbstractProgramData.java 515 2009-08-16 14:08:51Z tim.moxter $
33   * @param <T>   defines the class which is derived from this class
34   */
35  public abstract class AbstractProgramData<T extends AbstractProgramData<?>>
36      extends AbstractData<T>
37  {
38      /***
39       * Related code of this program.
40       *
41       * @see #setCode(String)
42       * @see #getCode()
43       */
44      private String code = "";
45  
46      /***
47       * Initialize the values for program objects.
48       *
49       * @param _test                 related test case
50       * @param _ci                   related configuration type
51       * @param _name                 name of the program
52       * @param _filePrefix           prefix for the file name
53       * @param _ciPath               path of the configuration item file
54       */
55      protected AbstractProgramData(final AbstractTest _test,
56                                final CI _ci,
57                                final String _name,
58                                final String _filePrefix,
59                                final String _ciPath)
60      {
61          super(_test, _ci, _name, _filePrefix, _ciPath, null);
62      }
63  
64      /***
65       * The related configuration item file is the {@link #code} of the program.
66       *
67       *  @return {@link #code} of the program
68       *  @see #code
69       */
70      @Override
71      public String ciFile()
72      {
73          return this.code;
74      }
75  
76      /***
77       * Defines the {@link #code} for this program.
78       *
79       * @param _code     new code of this program
80       * @return this program instance
81       * @see #code
82       */
83      @SuppressWarnings("unchecked")
84      public T setCode(final String _code)
85      {
86          this.code = _code;
87          return (T) this;
88      }
89  
90      /***
91       * Returns the {@link #code} of this program.
92       *
93       * @return related {@link #code} of this program
94       * @see #code
95       */
96      public String getCode()
97      {
98          return this.code;
99      }
