33  /***
34   * The class is used to define all administration objects which could have
35   * triggers used to create / update and to export.
36   *
37   * @author The MxUpdate Team
38   * @version $Id: AbstractDataWithTrigger.java 797 2010-05-16 22:47:14Z tim.moxter $
39   * @param <DATAWITHTRIGGERS>    defines the class which is derived from this
40   *                              class
41   */
42  public abstract class AbstractDataWithTrigger<DATAWITHTRIGGERS extends AbstractDataWithTrigger<?>>
43      extends AbstractDataWithAttribute<DATAWITHTRIGGERS>
44  {
45      /***
46       * Defines all assigned triggers to this administration object.
47       *
48       * @see #addTrigger(AbstractTrigger)
49       */
50      private final Set<AbstractTrigger<?>> triggers = new HashSet<AbstractTrigger<?>>();
51  
52      /***
53       * Initialize the values for administration objects with triggers.
54       *
55       * @param _test                 related test case
56       * @param _ci                   related configuration type
57       * @param _name                 name of the administration object
58       * @param _requiredExportValues defines the required values of the
59       *                              export within the configuration item
60       *                              file
61       */
62      protected AbstractDataWithTrigger(final AbstractTest _test,
63                                        final CI _ci,
64                                        final String _name,
65                                        final Map<String,String> _requiredExportValues)
66      {
67          super(_test, _ci, _name, _requiredExportValues);
68      }
69  
70      /***
71       * Appends a new <code>_trigger</code> to the list of all
72       * {@link #triggers}.
73       *
74       * @param _trigger  trigger to append
75       * @return this instance casted to T
76       * @see #triggers
77       */
78      @SuppressWarnings("unchecked")
79      public DATAWITHTRIGGERS addTrigger(final AbstractTrigger<?> _trigger)
80      {
81          this.triggers.add(_trigger);
82          return (DATAWITHTRIGGERS) this;
83      }
84  
85      /***
86       * {@inheritDoc}
87       * Creates all programs from depending {@link #triggers}.
88       *
89       * @see #triggers
90       */
91      @Override()
92      @SuppressWarnings("unchecked")
93      public DATAWITHTRIGGERS createDependings()
94          throws MatrixException
95      {
96          super.createDependings();
97  
98          // create programs
99          for (final AbstractDataWithTrigger.AbstractTrigger<?> trig : this.triggers)  {
100             trig.getProgram().create();
101         }
102 
103         return (DATAWITHTRIGGERS) this;
104     }

106     /***
107      * Appends the MQL commands to define all {@link #triggers} within a
108      * create.
109      *
110      * @param _cmd  string builder used to append MQL commands
111      * @throws MatrixException if programs could not be created or thrown from
112      *                         called method in super class
113      * @see #triggers
114      */
115     @Override()
116     protected void append4Create(final StringBuilder _cmd)
117         throws MatrixException
118     {
119         super.append4Create(_cmd);
120         for (final AbstractTrigger<?> trigger : this.triggers)  {
121             trigger.append4Create(_cmd);
122         }
123     }

125     /***
126      * Evaluates all 'adds' for triggers in the configuration item file.
127      *
128      * @param _needAdds     set with add strings used to append the adds for

130      */
131     @Override
132     protected void evalAdds4CheckExport(final Set<String> _needAdds)
133     {
134         super.evalAdds4CheckExport(_needAdds);

140     /***
141      * Defines a common interface for triggers.
142      *
143      * @param <T>   class of the derived trigger
144      */
145     public abstract static class AbstractTrigger<T extends AbstractDataWithTrigger.AbstractTrigger<?>>
146     {
147         /***
148          * Event type of this trigger.
149          */
150         private final String eventType;
151 
152         /***
153          * Kind of this trigger (action, check, overrider).
154          */
155         private final String kind;
156 
157         /***
158          * Called program of this trigger.
159          */
160         private final AbstractProgramData<?> program;
161 
162         /***
163          * Input value of this trigger.
164          *
165          * @see #setInput(String)
166          */
167         private String input;
168 
169         /***
170          * Initializes this trigger.
171          *
172          * @param _eventType    event type of this trigger
173          * @param _kind         kind of this trigger
174          * @param _program      used program of this trigger
175          */
176         private AbstractTrigger(final String _eventType,
177                                 final String _kind,
178                                 final AbstractProgramData<?> _program)
179         {
180             this.eventType = _eventType;
181             this.kind = _kind;
182             this.program = _program;
183         }
184 
185         /***
186          * Returns the defined trigger {@link #program}.
187          *
188          * @return trigger program
189          * @see #program
190          */
191         public AbstractProgramData<?> getProgram()
192         {
193             return this.program;
194         }
195 
196         /***
197          * Appends this trigger will so that the range is defined while the
198          * administration object is created.
199          *
200          * @param _cmd  string builder where the MQL trigger will be appended
201          * @throws MatrixException if {@link #program} could not be created
202          */
203         protected void append4Create(final StringBuilder _cmd)
204             throws MatrixException
205         {
206             this.program.create();
207             _cmd.append(" trigger ").append(this.eventType)
208                 .append(' ').append(this.kind)
209                 .append(" \"").append(AbstractTest.convertMql(this.program.getName()))
210                 .append('\"');
211             if (this.input != null)  {
212                 _cmd.append(" input \"").append(AbstractTest.convertMql(this.input)).append('\"');
213             }
214         }
215 
216         /***
217          * Appends the add statement in TCL code for this trigger.
218          *
219          * @param _needAdds     set with add strings used to append the adds
220          *                      for this trigger
221          */
222         protected void evalAdds4CheckExport(final Set<String> _needAdds)
223         {
224             final StringBuilder cmd = new StringBuilder()
225                 .append("trigger ").append(this.eventType)
226                 .append(' ').append(this.kind)
227                 .append(" \"").append(AbstractTest.convertTcl(this.program.getName()))
228                 .append("\"")
229                 .append(" input \"");
230             if (this.input != null)  {
231                 cmd.append(AbstractTest.convertMql(this.input));
232             }
233             cmd.append("\"");
234             _needAdds.add(cmd.toString());
235         }
236 
237         /***
238          * Defines the input value of this trigger.
239          *
240          * @param _input    input value
241          * @return this trigger instance
242          */
243         @SuppressWarnings("unchecked")
244         public T setInput(final String _input)
245         {
246             this.input = _input;
247             return (T) this;
248         }
249     }
