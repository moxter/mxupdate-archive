193     /***
194      * Create the related rule in MX for this rule data instance.
195      *
196      * @return this rule data instance
197      * @throws MatrixException if create failed
198      */
199     @Override()
200     public RuleData create() throws MatrixException
201     {
202         if (!this.isCreated())  {
203             this.setCreated(true);
204 
205             final StringBuilder cmd = new StringBuilder();
206             cmd.append("escape add rule \"").append(AbstractTest.convertMql(this.getName()))
207                .append("\" ");
208             // owner access
209             cmd.append(" owner ").append(this.ownerAccess);
210             if (this.ownerAccessFilter != null)  {
211                 cmd.append(" filter \"").append(AbstractTest.convertMql(this.ownerAccessFilter)).append("\" ");
212             }
213             // owner revoke
214             if (this.ownerRevoke != null)  {
215                 cmd.append(" revoke owner ").append(this.ownerRevoke);
216                 if (this.ownerRevokeFilter != null)  {
217                     cmd.append(" filter \"").append(AbstractTest.convertMql(this.ownerRevokeFilter)).append("\" ");
218                 }
219             }
220             // public access
221             cmd.append(" public ").append(this.publicAccess);
222             if (this.publicAccessFilter != null)  {
223                 cmd.append(" filter \"").append(AbstractTest.convertMql(this.publicAccessFilter)).append("\" ");
224             }
225             // public revoke
226             if (this.publicRevoke != null)  {
227                 cmd.append(" revoke public ").append(this.publicRevoke);
228                 if (this.publicRevokeFilter != null)  {
229                     cmd.append(" filter \"").append(AbstractTest.convertMql(this.publicRevokeFilter)).append("\" ");
230                 }
231             }
232             this.append4Create(cmd);
233 
234             this.getTest().mql(cmd);
235         }
236         return this;
237     }

239     /***
240      * Appends the add statements for the owner / public access / revokes.
241      *
242      * @param _needAdds     set where all adds are appended
243      */
244     @Override()
245     protected void evalAdds4CheckExport(final Set<String> _needAdds)
246     {
247         super.evalAdds4CheckExport(_needAdds);
248         // owner access
249         final StringBuilder ownerAccess = new StringBuilder()
250             .append("owner \"").append(this.ownerAccess).append("\"");
251         if (this.ownerAccessFilter != null)  {
252             ownerAccess.append(" filter \"").append(AbstractTest.convertTcl(this.ownerAccessFilter)).append("\"");
253         }
254         _needAdds.add(ownerAccess.toString());
255         // owner revoke
256         if (this.ownerRevoke != null)  {
257             final StringBuilder ownerRevoke = new StringBuilder()
258                 .append("revoke owner \"").append(this.ownerRevoke).append("\"");
259             if (this.ownerRevokeFilter != null)  {
260                 ownerRevoke.append(" filter \"").append(AbstractTest.convertTcl(this.ownerRevokeFilter)).append("\"");
261             }
262             _needAdds.add(ownerRevoke.toString());
263         }
264         // public access
265         final StringBuilder publicAccess = new StringBuilder()
266             .append("public \"").append(this.publicAccess).append("\"");
267         if (this.publicAccessFilter != null)  {
268             publicAccess.append(" filter \"").append(AbstractTest.convertTcl(this.publicAccessFilter)).append("\"");
269         }
270         // public revoke
271         if (this.publicRevoke != null)  {
272             final StringBuilder publicRevoke = new StringBuilder()
273                 .append("revoke public \"").append(this.publicRevoke).append("\"");
274             if (this.publicRevokeFilter != null)  {
275                 publicRevoke.append(" filter \"").append(AbstractTest.convertTcl(this.publicRevokeFilter)).append("\"");
276             }
277             _needAdds.add(publicRevoke.toString());
278         }
279         _needAdds.add(publicAccess.toString());
280     }
