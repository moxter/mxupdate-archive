166     /***
167      * Returns the name of the configuration item file.
168      *
169      * @return name of the CI file
170      */
171     public String getCIFileName()
172     {
173         final char[] charName = this.name.toCharArray();
174         final StringBuilder fileName = new StringBuilder().append(this.getCI().filePrefix);
175         for (int idx = 0; idx < charName.length; idx++)  {
176             final char ch = charName[idx];
177             if (ch == '@')  {
178                 fileName.append("@@");
179             } else if ((ch < '(' || ch > ')')
180                     && (ch < '+' || ch > '.')
181                     && (ch < '0' || ch > '9')
182                     && (ch < 'A' || ch > 'Z')
183                     && (ch < 'a' || ch > 'z')
184                     && (ch != ' ') && (ch != '=') && (ch != '_'))  {
185 
186                 final String hex = String.valueOf(Integer.toHexString(ch));
187                 fileName.append('@');
188                 switch (hex.length())  {
189                     case 1:
190                         fileName.append('0').append(hex);
191                         break;
192                     case 3:
193                         fileName.append("u0").append(hex);
194                         break;
195                     case 4:
196                         fileName.append('u').append(hex);
197                         break;
198                     default:
199                         fileName.append(hex);
200                         break;
201                 }
202             } else  {
203                 fileName.append(ch);
204             }
205         }
206         fileName.append(".tcl");
207 
208         return fileName.toString();
209     }

249     /***
250      * Exports this data piece from MX. The returned values from the export are
251      * checked for:
252      * <ul>
253      * <li>value is returned</li>
254      * <li>returned export value includes exact one export for given
255      *     {@link #ci configuration item type}</li>
256      * <li>name of the exported object is equal {@link #name}</li>
257      * <li>path of the exported object is equal {@link #ciPath}</li>
258      * <li>file name of the export object is equal returned value from
259      *     {@link #getCIFileName()}</li>
260      * </ul>
261      *
262      * @param _params       parameters
263      * @return parsed export
264      * @throws IOException      if the parameter could not be encoded
265      * @throws MatrixException  if MQL calls failed
266      */
267     public ExportParser export(final String... _params)
268         throws IOException, MatrixException
269     {
270         final Map<String,Collection<String>> files = new HashMap<String,Collection<String>>(1);
271         files.put(this.ci.updateType, Arrays.asList(new String[]{this.getName()}));
272 
273         final Map<String,String> params = new HashMap<String,String>();
274         if (_params != null)  {
275             for (int idx = 0; idx < _params.length; idx += 2)  {
276                 params.put(_params[idx], _params[idx + 1]);
277             }
278         }
279 
280         final Map<String,Collection<Map<String,String>>> bck =
281                 this.test.<Map<String,Collection<Map<String,String>>>>jpoInvoke("org.mxupdate.plugin.Export",
282                                                                                 "exportByName",
283                                                                                 files,
284                                                                                 params)
285                     .getValues();
286 
287         // check existence and element is defined
288         Assert.assertNotNull(bck);
289         Assert.assertTrue(bck.containsKey(this.ci.updateType));
290         Assert.assertEquals(bck.get(this.ci.updateType).size(), 1, "one element is returned");
291 
292         // parse first element
293         final Map<String,String> exportDesc = bck.get(this.getCI().updateType).iterator().next();
294         final ExportParser ret = this.parseExport(this.ci, exportDesc.get("code"));
295 
296         // check returned configuration item name
297         Assert.assertEquals(exportDesc.get("name"),
298                             this.name,
299                             "returned name is equal to given name");
300         // check path of the configuration item update file
301         Assert.assertEquals(exportDesc.get("path"),
302                             this.getCI().filePath,
303                             "check path where the configuration item update file is located is correct");
304         // check file name of the configuration item update file
305         Assert.assertEquals(exportDesc.get("filename"),
306                             this.getCIFileName(),
307                             "check that the correct configuration item file name is returned");
308 
309         return ret;
310     }

337     /***
338      * Checks for given <code>_tag</code> in the <code>_exportParser</code>
339      * if the <code>_value</code> exists and is correct defined. If
340      * <code>_value</code> is <code>null</code>, it is checked that no value is
341      * defined.
342      *
343      * @param _exportParser     parsed export
344      * @param _kind             kind of the check
345      * @param _tag              tag to check
346      * @param _value            value to check (or <code>null</code> if value
347      *                          is not defined)
348      */
349     public void checkSingleValue(final ExportParser _exportParser,
350                                  final String _kind,
351                                  final String _tag,
352                                  final String _value)
353     {
354         if (_value != null)  {
355             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").size(),
356                                 1,
357                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is defined");
358             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").get(0),
359                                 _value,
360                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is " + _value);
361 
362         } else  {
363             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").size(),
364                                 0,
365                                 "check " + _kind + " '" + this.getName() + "' that no " + _tag + " is defined");
366         }
367     }
