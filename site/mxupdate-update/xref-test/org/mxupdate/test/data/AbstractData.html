228     /***
229      * Returns the name of the configuration item file.
230      *
231      * @return name of the CI file
232      */
233     public String getCIFileName()
234     {
235         final char[] charName = this.name.toCharArray();
236         final StringBuilder fileName = new StringBuilder().append(this.filePrefix);
237         for (int idx = 0; idx < charName.length; idx++)  {
238             final char ch = charName[idx];
239             if (ch == '@')  {
240                 fileName.append("@@");
241             } else if ((ch < '(' || ch > ')')
242                     && (ch < '+' || ch > '.')
243                     && (ch < '0' || ch > '9')
244                     && (ch < 'A' || ch > 'Z')
245                     && (ch < 'a' || ch > 'z')
246                     && (ch != ' ') && (ch != '=') && (ch != '_'))  {
247 
248                 final String hex = String.valueOf(Integer.toHexString(ch));
249                 fileName.append('@');
250                 switch (hex.length())  {
251                     case 1:
252                         fileName.append('0').append(hex);
253                         break;
254                     case 3:
255                         fileName.append("u0").append(hex);
256                         break;
257                     case 4:
258                         fileName.append('u').append(hex);
259                         break;
260                     default:
261                         fileName.append(hex);
262                         break;
263                 }
264             } else  {
265                 fileName.append(ch);
266             }
267         }
268         fileName.append(".tcl");
269 
270         return fileName.toString();
271     }

281     /***
282      * Appends the defined {@link #values} to the TCL code <code>_cmd</code> of
283      * the configuration item file.
284      *
285      * @param _cmd  string builder with the TCL commands of the configuration
286      *              item file
287      * @see #values
288      */
289     protected void append4CIFileValues(final StringBuilder _cmd)
290     {
291         for (final Map.Entry<String,String> entry : this.values.entrySet())  {
292             _cmd.append(' ').append(entry.getKey()).append(" \"")
293                 .append(AbstractTest.convertTcl(entry.getValue()))
294                 .append('\"');
295         }
296         // check for add values
297         final Set<String> needAdds = new HashSet<String>();
298         this.evalAdds4CheckExport(needAdds);
299         for (final String needAdd : needAdds)  {
300             _cmd.append(" add ").append(needAdd);
301         }
302     }

352     /***
353      * Checks the export of this data piece if all values are correct defined.
354      *
355      * @param _exportParser     parsed export
356      * @throws MatrixException if check failed
357      */
358     public void checkExport(final ExportParser _exportParser)
359         throws MatrixException
360     {
361         Assert.assertEquals(_exportParser.getFileName(),
362                             this.getCIFileName(),
363                             "check file name");
364         Assert.assertEquals(_exportParser.getName(),
365                             this.getName(),
366                             "check name");
367         Assert.assertEquals(_exportParser.getSymbolicName(),
368                             this.getSymbolicName(),
369                             "check symbolic name");
370         // check for all required values
371         for (final String valueName : this.requiredExportValues)  {
372             Assert.assertEquals(_exportParser.getLines("/mql/" + valueName + "/@value").size(),
373                                 1,
374                                 "minimum and maximum one " + valueName + " is defined");
375         }
376         // check for defined values
377         for (final Map.Entry<String,String> entry : this.values.entrySet())  {
378             Assert.assertEquals(_exportParser.getLines("/mql/" + entry.getKey() + "/@value").size(),
379                                                        1,
380                                                        "minimum and maximum one " + entry.getKey() + " is defined");
381             Assert.assertEquals(_exportParser.getLines("/mql/" + entry.getKey() + "/@value").get(0),
382                                 "\"" + AbstractTest.convertTcl(entry.getValue()) + "\"",
383                                 entry.getKey() + " is correct defined");
384         }
385         // check for add values
386         final Set<String> needAdds = new HashSet<String>();
387         this.evalAdds4CheckExport(needAdds);
388         final List<String> foundAdds = _exportParser.getLines("/mql/add/@value");
389         Assert.assertEquals(foundAdds.size(), needAdds.size(), "all adds defined");
390         for (final String foundAdd : foundAdds)  {
391             Assert.assertTrue(needAdds.contains(foundAdd), "check that add '" + foundAdd + "' is defined");
392         }
393     }
