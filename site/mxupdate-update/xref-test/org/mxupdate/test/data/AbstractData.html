239     /***
240      * Returns the name of the configuration item file.
241      *
242      * @return name of the CI file
243      */
244     public String getCIFileName()
245     {
246         final char[] charName = this.name.toCharArray();
247         final StringBuilder fileName = new StringBuilder().append(this.filePrefix);
248         for (int idx = 0; idx < charName.length; idx++)  {
249             final char ch = charName[idx];
250             if (ch == '@')  {
251                 fileName.append("@@");
252             } else if ((ch < '(' || ch > ')')
253                     && (ch < '+' || ch > '.')
254                     && (ch < '0' || ch > '9')
255                     && (ch < 'A' || ch > 'Z')
256                     && (ch < 'a' || ch > 'z')
257                     && (ch != ' ') && (ch != '=') && (ch != '_'))  {
258 
259                 final String hex = String.valueOf(Integer.toHexString(ch));
260                 fileName.append('@');
261                 switch (hex.length())  {
262                     case 1:
263                         fileName.append('0').append(hex);
264                         break;
265                     case 3:
266                         fileName.append("u0").append(hex);
267                         break;
268                     case 4:
269                         fileName.append('u').append(hex);
270                         break;
271                     default:
272                         fileName.append(hex);
273                         break;
274                 }
275             } else  {
276                 fileName.append(ch);
277             }
278         }
279         fileName.append(".tcl");
280 
281         return fileName.toString();
282     }

312         for (final Map.Entry<String,String> entry : this.values.entrySet())  {

317         // check for add values

373     /***
374      * Exports this data piece from MX. The returned values from the export are
375      * checked for:
376      * <ul>
377      * <li>value is returned</li>
378      * <li>returned export value includes exact one export for given
379      *     {@link #ci configuration item type}</li>
380      * <li>name of the exported object is equal {@link #name}</li>
381      * <li>path of the exported object is equal {@link #ciPath}</li>
382      * <li>file name of the export object is equal returned value from
383      *     {@link #getCIFileName()}</li>
384      * </ul>
385      *
386      * @return parsed export
387      * @throws IOException      if the parameter could not be encoded
388      * @throws MatrixException  if MQL calls failed
389      */
390     public ExportParser export()
391         throws IOException, MatrixException
392     {
393         final Map<String,Collection<String>> params = new HashMap<String,Collection<String>>(1);
394         params.put(this.ci.updateType, Arrays.asList(new String[]{this.getName()}));
395         final Map<String,Collection<Map<String,String>>> bck =
396                 this.test.<Map<String,Collection<Map<String,String>>>>jpoInvoke("org.mxupdate.plugin.Export",
397                                                                                 "exportByName",
398                                                                                 params)
399                     .getValues();
400 
401         // check existence and element is defined
402         Assert.assertNotNull(bck);
403         Assert.assertTrue(bck.containsKey(this.ci.updateType));
404         Assert.assertEquals(bck.get(this.ci.updateType).size(), 1, "one element is returned");
405 
406         // parse first element
407         final Map<String,String> exportDesc = bck.get(this.getCI().updateType).iterator().next();
408         final ExportParser ret = this.parseExport(this.ci, exportDesc.get("code"));
409 
410         // check returned configuration item name
411         Assert.assertEquals(exportDesc.get("name"),
412                             this.name,
413                             "returned name is equal to given name");
414         // check path of the configuration item update file
415         Assert.assertEquals(exportDesc.get("path"),
416                             this.ciPath,
417                             "check path where the configuration item update file is located is correct");
418         // check file name of the configuration item update file
419         Assert.assertEquals(exportDesc.get("filename"),
420                             this.getCIFileName(),
421                             "check that the correct configuration item file name is returned");
422 
423         return ret;
424     }

439     /***
440      * Checks the export of this data piece if all values are correct defined.
441      *
442      * @param _exportParser     parsed export
443      * @throws MatrixException if check failed
444      */
445     public void checkExport(final ExportParser _exportParser)
446         throws MatrixException
447     {
448         Assert.assertEquals(_exportParser.getName(),
449                             this.getName(),
450                             "check name");
451         Assert.assertEquals(_exportParser.getSymbolicName(),
452                             this.getSymbolicName(),
453                             "check symbolic name");
454         // check for all required values
455         for (final String valueName : this.requiredExportValues)  {
456             Assert.assertEquals(_exportParser.getLines("/mql/" + valueName + "/@value").size(),
457                                 1,
458                                 "required check that minimum and maximum one " + valueName + " is defined");
459         }
460         // check for defined values
461         for (final Map.Entry<String,String> entry : this.values.entrySet())  {
462             Assert.assertEquals(_exportParser.getLines("/mql/" + entry.getKey() + "/@value").size(),
463                                                        1,
464                                                        "check that minimum and maximum one "
465                                                                + entry.getKey() + " is defined");
466             Assert.assertEquals(_exportParser.getLines("/mql/" + entry.getKey() + "/@value").get(0),
467                                 "\"" + AbstractTest.convertTcl(entry.getValue()) + "\"",
468                                 "check that " + entry.getKey() + " is correct defined");
469         }
470         // check for add values
471         final Set<String> needAdds = new HashSet<String>();
472         this.evalAdds4CheckExport(needAdds);
473         final List<String> foundAdds = _exportParser.getLines("/mql/add/@value");
474         Assert.assertEquals(foundAdds.size(), needAdds.size(), "all adds defined");
475         for (final String foundAdd : foundAdds)  {
476             Assert.assertTrue(needAdds.contains(foundAdd), "check that add '" + foundAdd + "' is defined");
477         }
478     }
