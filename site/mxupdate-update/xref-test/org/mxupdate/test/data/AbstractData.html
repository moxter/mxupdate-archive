169     /***
170      * Returns the name of the configuration item file.
171      *
172      * @return name of the CI file
173      */
174     public String getCIFileName()
175     {
176         final char[] charName = this.name.toCharArray();
177         final StringBuilder fileName = new StringBuilder().append(this.getCI().filePrefix);
178         for (int idx = 0; idx < charName.length; idx++)  {
179             final char ch = charName[idx];
180             if (ch == '@')  {
181                 fileName.append("@@");
182             } else if ((ch < '(' || ch > ')')
183                     && (ch < '+' || ch > '.')
184                     && (ch < '0' || ch > '9')
185                     && (ch < 'A' || ch > 'Z')
186                     && (ch < 'a' || ch > 'z')
187                     && (ch != ' ') && (ch != '=') && (ch != '_'))  {
188 
189                 final String hex = String.valueOf(Integer.toHexString(ch));
190                 fileName.append('@');
191                 switch (hex.length())  {
192                     case 1:
193                         fileName.append('0').append(hex);
194                         break;
195                     case 3:
196                         fileName.append("u0").append(hex);
197                         break;
198                     case 4:
199                         fileName.append('u').append(hex);
200                         break;
201                     default:
202                         fileName.append(hex);
203                         break;
204                 }
205             } else  {
206                 fileName.append(ch);
207             }
208         }
209         fileName.append(".tcl");
210 
211         return fileName.toString();
212     }

277     /***
278      * Makes an update for given <code>_code</code>.
279      *
280      * @param _code         TCL update code
281      * @param _params       parameters
282      * @return values from the called dispatcher
283      * @throws Exception  if update failed
284      */
285     @SuppressWarnings("unchecked")
286     public DATA updateWithCode(final String _code,
287                                final String... _params)
288         throws Exception
289     {
290         final Map<String,String> files = new HashMap<String,String>();
291         files.put(this.getCIFileName(), _code);
292         final Map<String,String> params = new HashMap<String,String>();
293         if (_params != null)  {
294             for (int idx = 0; idx < _params.length; idx += 2)  {
295                 params.put(_params[idx], _params[idx + 1]);
296             }
297         }
298         final Map<?,?> bck = this.getTest().executeEncoded("Update", params, "FileContents", files);
299         if (bck.get("exception") != null)  {
300             throw (Exception) bck.get("exception");
301         }
302         return (DATA) this;
303     }

305     /***
306      * Exports this data piece from MX. The returned values from the export are
307      * checked for:
308      * <ul>
309      * <li>value is returned</li>
310      * <li>returned export value includes exact one export for given
311      *     {@link #ci configuration item type}</li>
312      * <li>name of the exported object is equal {@link #name}</li>
313      * <li>path of the exported object is equal {@link #ciFile()}</li>
314      * <li>file name of the export object is equal returned value from
315      *     {@link #getCIFileName()}</li>
316      * </ul>
317      *
318      * @param _params       parameters
319      * @return parsed export
320      * @throws Exception if export failed
321      */
322     public ExportParser export(final String... _params)
323         throws Exception
324     {
325         final Map<String,Collection<String>> files = new HashMap<String,Collection<String>>(1);
326         files.put(this.ci.updateType, Arrays.asList(new String[]{this.getName()}));
327 
328         final Map<String,String> params = new HashMap<String,String>();
329         if (_params != null)  {
330             for (int idx = 0; idx < _params.length; idx += 2)  {
331                 params.put(_params[idx], _params[idx + 1]);
332             }
333         }
334 
335         final Map<?,?> bck = this.test.executeEncoded("Export",
336                                                       params,
337                                                       "TypeDef", this.ci.updateType,
338                                                       "Name", this.getName());
339 
340         // check for no exception
341         if (bck.get("exception") != null)  {
342             throw new MatrixException((Exception) bck.get("exception"));
343         }
344 
345         // extract values
346         final Map<?,?> values = (Map<?,?>) bck.get("values");
347 
348         // check existence and element is defined
349         Assert.assertNotNull(values);
350         Assert.assertEquals((String) values.get("TypeDef"),
351                             this.ci.updateType,
352                             "check correct type definition");
353 
354         // parse first element
355         final ExportParser ret = this.parseExport(this.ci, (String) values.get("Code"));
356 
357         // check returned configuration item name
358         Assert.assertEquals((String) values.get("Name"),
359                             this.name,
360                             "returned name is equal to given name");
361         if (this.ci != AbstractTest.CI.PRG_JPO)  {
362             // check path of the configuration item update file
363             Assert.assertEquals((String) values.get("FilePath"),
364                                 this.getCI().filePath,
365                                 "check path where the configuration item update file is located is correct");
366         } else  {
367             // check path for JPOs (because of packages...)
368             final String path;
369             if (this.name.indexOf('.') < 0)  {
370                 path = this.ci.filePath;
371             } else  {
372                 path = this.ci.filePath + "/" + this.name.replaceAll("//.[^.]*$", "").replaceAll("//.", "/");
373             }
374             Assert.assertEquals((String) values.get("FilePath"),
375                                 path,
376                                 "check path where the configuration item update file is located is correct");
377         }
378         // check file name of the configuration item update file
379         Assert.assertEquals((String) values.get("FileName"),
380                             this.getCIFileNameFromExport(),
381                             "check that the correct configuration item file name is returned");
382 
383         return ret;
384     }

428     /***
429      * Checks for given <code>_tag</code> in the <code>_exportParser</code>
430      * if the <code>_value</code> exists and is correct defined. If
431      * <code>_value</code> is <code>null</code>, it is checked that no value is
432      * defined.
433      *
434      * @param _exportParser     parsed export
435      * @param _kind             kind of the check
436      * @param _tag              tag to check
437      * @param _value            value to check (or <code>null</code> if value
438      *                          is not defined)
439      */
440     public void checkSingleValue(final ExportParser _exportParser,
441                                  final String _kind,
442                                  final String _tag,
443                                  final String _value)
444     {
445         if (_value != null)  {
446             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").size(),
447                                 1,
448                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is defined");
449             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").get(0),
450                                 _value,
451                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is " + _value);
452 
453         } else  {
454             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").size(),
455                                 0,
456                                 "check " + _kind + " '" + this.getName() + "' that no " + _tag + " is defined");
457         }
458     }
