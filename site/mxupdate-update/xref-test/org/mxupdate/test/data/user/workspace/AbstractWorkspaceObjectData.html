127     /***
128      * Creates this workspace object.
129      *
130      * @return this workspace object instance
131      * @throws MatrixException if create failed
132      */
133     @SuppressWarnings("unchecked")
134     @Override()
135     public DATA create()
136         throws MatrixException
137     {
138         if (!this.isCreated())  {
139             final StringBuilder cmd = new StringBuilder()
140                     .append("escape add ").append(this.mxAdminType).append(" \"")
141                     .append(AbstractTest.convertMql(this.getName()))
142                     .append("\" user \"")
143                     .append(AbstractTest.convertMql(this.user.getName()))
144                     .append('\"');
145 
146             // append values
147             this.append4Create(cmd);
148 
149             this.getTest().mql(cmd);
150 
151             this.setCreated(true);
152         }
153         return (DATA) this;
154     }

156     /***
157      * Appends the MQL commands to define the hidden flag and the
158      * {@link #visible users}Êwithin a create.
159      *
160      * @param _cmd  string builder used to append MQL commands
161      * @throws MatrixException if used object could not be created
162      * @see #values
163      */
164     @Override()
165     protected void append4Create(final StringBuilder _cmd)
166         throws MatrixException
167     {
168         super.append4Create(_cmd);
169 
170         // hidden flag
171         if (this.isHidden())  {
172             _cmd.append(" hidden");
173         } else  {
174             _cmd.append(" !hidden");
175         }
176 
177         // visible users
178         if (!this.visible.isEmpty())  {
179             _cmd.append(" visible ");
180             boolean first = true;
181             for (final String user : this.visible)  {
182                 if (first)  {
183                     first = false;
184                 } else  {
185                     _cmd.append(',');
186                 }
187                 _cmd.append('\"').append(AbstractTest.convertMql(user)).append('\"');
188             }
189         }
190     }

192     /***
193      * Returns the part of the CI file to create this workspace object of an
194      * user.
195      *
196      * @return part of the CI file to create this workspace object of an user
197      */
198     @Override()
199     public String ciFile()
200     {
201         final StringBuilder cmd = new StringBuilder()
202                 .append("mql escape add ").append(this.mxAdminType).append(" \"")
203                 .append(AbstractTest.convertTcl(this.getName()))
204                 .append("\" //\n    user \"${NAME}\"");
205 
206         // hidden flag
207         if (this.isHidden())  {
208             cmd.append(" //\n    hidden");
209         } else  {
210             cmd.append(" //\n    !hidden");
211         }
212 
213         // visible users
214         if (!this.visible.isEmpty())  {
215             for (final String user : this.visible)  {
216                 cmd.append(" //\n    visible \"")
217                    .append(AbstractTest.convertTcl(user))
218                    .append('\"');
219             }
220         }
221 
222         this.append4CIFileValues(cmd);
223         return cmd.toString();
224     }

226     /***
227      * Checks the export of this data piece if all values are correct defined.
228      * The hidden flag and the {@link #visible visible users} are checked.
229      *
230      * @param _exportParser     parsed export
231      * @throws MatrixException if check failed
232      */
233     @Override()
234     public void checkExport(final ExportParser _exportParser)
235         throws MatrixException
236     {
237         super.checkExport(_exportParser);
238 
239         // check hidden flag
240         final Set<String> main = new HashSet<String>(_exportParser.getLines("/mql/"));
241         if (this.isHidden())  {
242             Assert.assertTrue(main.contains("hidden") || main.contains("hidden //"),
243                               "check that " + this.mxAdminType + " '" + this.getName() + "' is hidden");
244         } else  {
245             Assert.assertTrue(main.contains("!hidden") || main.contains("!hidden //"),
246                               "check that " + this.mxAdminType + " '" + this.getName() + "' is not hidden");
247         }
248 
249         // check visible users
250         final List<String> foundVisible = _exportParser.getLines("/mql/visible/@value");
251         for (final String user : foundVisible)  {
252             final String tmpUser = user.replaceAll("^\"", "").replaceAll("\"$", "");
253             Assert.assertTrue(this.visible.contains(tmpUser),
254                               "check that '" + tmpUser + "' was defined visible");
255         }
256         Assert.assertEquals(foundVisible.size(),
257                             this.visible.size(),
258                             "check that all visible users are defined");
259 
260         // check properties
261         final Set<String> foundProps = new HashSet<String>(_exportParser.getLines("/mql/property/@value"));
262         for (final PropertyDef property : this.getProperties())  {
263             final StringBuilder key = new StringBuilder()
264                     .append("\"").append(AbstractTest.convertTcl(property.getName())).append("\"");
265             if (property.getTo() != null)  {
266                 key.append(" to ").append(property.getTo().getCI().getMxType()).append(" \"")
267                     .append(AbstractTest.convertTcl(property.getTo().getName())).append("\"");
268             }
269             if (property.getValue() != null)  {
270                 key.append(" value \"").append(AbstractTest.convertTcl(property.getValue())).append("\"");
271             }
272             foundProps.remove(key.toString());
273         }
274         Assert.assertTrue(foundProps.isEmpty(), "check that all properties are defined");
275     }
