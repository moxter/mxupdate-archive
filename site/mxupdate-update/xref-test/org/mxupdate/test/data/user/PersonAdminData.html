275     /***
276      * Appends the person specific values to the TCL update code
277      * <code>_cmd</code> of the configuration item file. This includes:
278      * <ul>
279      * <li>{@link #access}</li>
280      * <li>{@link #adminAccess administration access}</li>
281      * <li>{@link #types}</li>
282      * <li>{@link #groups}<li>
283      * <li>{@link #roles}</li>
284      * <li>{@link #passwordNeverExpires password never expires flag}</li>
285      * <li>{@link #wantsEmail wants email flag}</li>
286      * <li>{@link #wantsIconMail wants icon mail flag}</li>
287      * </ul>
288      *
289      * @param _cmd  string builder with the TCL commands of the configuration
290      *              item file
291      * @see #values
292      */
293     @Override()
294     protected void append4CIFileValues(final StringBuilder _cmd)
295     {
296         super.append4CIFileValues(_cmd);
297         _cmd.append(" access ").append(StringUtil_mxJPO.joinTcl(',', false, this.access, "none"))
298             .append(" admin ").append(StringUtil_mxJPO.joinTcl(',', false, this.adminAccess, "none"));
299         if (!this.types.isEmpty())  {
300             _cmd.append(" type ").append(StringUtil_mxJPO.joinTcl(',', false, this.types, null)).append(";\n");
301         }
302         // groups
303         for (final GroupData group : this.groups)  {
304             _cmd.append(" assign group \"").append(AbstractTest.convertTcl(group.getName())).append("\"");
305         }
306         // roles
307         for (final RoleData role : this.roles)  {
308             _cmd.append(" assign role \"").append(AbstractTest.convertTcl(role.getName())).append("\"");
309         }
310         // password never expires flag
311         if (this.passwordNeverExpires != null)  {
312             _cmd.append(' ');
313             if (!this.passwordNeverExpires)  {
314                 _cmd.append('!');
315             }
316             _cmd.append("neverexpires");
317         }
318         // wants email
319         if (this.wantsEmail != null)  {
320             _cmd.append(' ')
321                 .append(this.wantsEmail ? "enable" : "disable")
322                 .append(" email");
323         }
324         // wants icon mail
325         if (this.wantsIconMail != null)  {
326             _cmd.append(' ')
327                 .append(this.wantsIconMail ? "enable" : "disable")
328                 .append(" iconmail");
329         }
330     }

354         super.append4Create(_cmd);

356         // assign access
357         _cmd.append(" access ").append(StringUtil_mxJPO.joinMql(',', false, this.access, "none"));
358         // assign administration access
359         if (!this.adminAccess.isEmpty())  {
360             _cmd.append(" admin ").append(StringUtil_mxJPO.joinMql(',', false, this.adminAccess, null));
361         }
362         // assign access
363         if (!this.types.isEmpty())  {
364             _cmd.append(" type ").append(StringUtil_mxJPO.joinMql(',', false, this.types, null));
365         }
366         // groups
367         for (final GroupData group : this.groups)  {
368             group.create();
369             _cmd.append(" assign group \"").append(AbstractTest.convertMql(group.getName())).append("\"");
370         }
371         // roles
372         for (final RoleData role : this.roles)  {
373             role.create();
374             _cmd.append(" assign role \"").append(AbstractTest.convertMql(role.getName())).append("\"");
375         }
376         // password never expires flag
377         if (this.passwordNeverExpires != null)  {
378             _cmd.append(' ');
379             if (!this.passwordNeverExpires)  {
380                 _cmd.append('!');
381             }
382             _cmd.append("neverexpires");
383         }
384         // wants email
385         if (this.wantsEmail != null)  {
386             _cmd.append(' ')
387                 .append(this.wantsEmail ? "enable" : "disable")
388                 .append(" email");
389         }

398     /***
399      * Checks the export of this person if all values are correct defined. It
400      * is checked that
401      * <ul>
402      * <li>all {@link #access} is correct defined</li>
403      * <li>all {@link #adminAccess administration access} is correct defined
404      *     </li>
405      * <li>all {@link #groups} are correct defined</li>
406      * <li>all {@link #roles} are correct defined</li>
407      * <li>{@link #passwordNeverExpires password never expires flag} is defined
408      *     if set to true</li>
409      * <li>{@link #wantsEmail wants email flag} is correct defined</li>
410      * <li>{@link #wantsIconMail wants icon mail flag} is correct defined</li>
411      * </ul>
412      *
413      * @param _exportParser     parsed export
414      * @throws MatrixException if check failed
415      */
416     @Override()
417     public void checkExport(final ExportParser _exportParser)
418         throws MatrixException
419     {
420         super.checkExport(_exportParser);
421 
422         // check access
423         this.checkSingleValue(_exportParser,
424                               "person",
425                               "access",
426                               "\"" + StringUtil_mxJPO.joinTcl(',', false, this.access, "none") + "\"");
427 
428         // check administration access
429         this.checkSingleValue(_exportParser,
430                               "person",
431                               "admin",
432                               "\"" + StringUtil_mxJPO.joinTcl(',', false, this.adminAccess, "none") + "\"");
433 
434         // prepare list of all assignments
435         final Set<String> assigns = new HashSet<String>();
436         for (final GroupData group : this.groups)  {
437             assigns.add("group \"" + AbstractTest.convertTcl(group.getName()) + "\"");
438         }
439         for (final RoleData role : this.roles)  {
440             assigns.add("role \"" + AbstractTest.convertTcl(role.getName()) + "\"");
441         }
442 
443         // remove all assigned groups / roles in the TCL update file
444         for (final String assign : _exportParser.getLines("/mql/assign/@value"))  {
445             if (assigns.contains(assign))  {
446                 assigns.remove(assign);
447             } else  {
448                 Assert.assertNotNull(assign, "assign " + assign + "not known");
449             }
450         }
451         Assert.assertEquals(assigns.size(), 0, "check that all assignments for groups / roles are defined");
452 
453         // password never expires flag
454         this.checkValueExists(_exportParser, "person", "neverexpires", (this.passwordNeverExpires != null) && this.passwordNeverExpires);
455         this.checkValueExists(_exportParser, "person", "!neverexpires", (this.passwordNeverExpires == null) || !this.passwordNeverExpires);
456 
457         // email / icon mail flags
458         final Set<String> enabled = new HashSet<String>(_exportParser.getLines("/mql/enable/@value"));
459         final Set<String> disabled = new HashSet<String>(_exportParser.getLines("/mql/disable/@value"));
460         if ((this.wantsEmail != null) && this.wantsEmail)  {
461             Assert.assertTrue(enabled.contains("email"), "check email enabled");
462             Assert.assertFalse(disabled.contains("email"), "check email not disabled");
463         } else  {
464             Assert.assertFalse(enabled.contains("email"), "check email not enabled");
465             Assert.assertTrue(disabled.contains("email"), "check email disabled");
466         }
467         if ((this.wantsIconMail == null) || this.wantsIconMail)  {
468             Assert.assertTrue(enabled.contains("iconmail"), "check icon mail enabled");
469             Assert.assertFalse(disabled.contains("iconmail"), "check icon mail not disabled");
470         } else  {
471             Assert.assertFalse(enabled.contains("iconmail"), "check icon mail not enabled");
472             Assert.assertTrue(disabled.contains("iconmail"), "check icon mail disabled");
473         }
474     }
