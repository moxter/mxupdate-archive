143     /***
144      * Tests, if the MQL program within MX is created and registered with the
145      * correct symbolic name.
146      *
147      * @param _description  description of the test data
148      * @param _mqlProgram   data collection to test
149      * @throws Exception if test failed
150      */
151     @Test(dataProvider = "mqlprograms", description = "test update of non existing MQL programs")
152     public void simpleUpdate(final String _description,
153                              final MQLProgramData _mqlProgram)
154         throws Exception
155     {
156         // create user
157         if (_mqlProgram.getUser() != null)  {
158             _mqlProgram.getUser().create();
159         }
160         // create referenced property value
161         for (final PropertyDef prop : _mqlProgram.getProperties())  {
162             if (prop.getTo() != null)  {
163                 prop.getTo().create();
164             }
165         }
166 
167         // first update with original content
168         _mqlProgram.update();
169 
170         // the replace code (removing TCL update commands)
171         final StringBuilder cmd = new StringBuilder()
172                 .append("escape mod program \"").append(AbstractTest.convertMql(_mqlProgram.getName()))
173                 .append("\" code \"").append(AbstractTest.convertMql(_mqlProgram.getCode())).append("\"");
174         this.mql(cmd);
175 
176         // export
177         final ExportParser exportParser = _mqlProgram.export();
178         _mqlProgram.checkExport(exportParser);
179 
180         // set creator as user, 'test' as description, not hidden
181         this.mql("escape mod program \"" + AbstractTest.convertMql(_mqlProgram.getName())
182                 + "\" execute user creator execute immediate !needsbusinessobject !downloadable !pipe !pooled description test !hidden");
183 
184         // second update with delivered content
185         _mqlProgram.updateWithCode(exportParser.getOrigCode())
186                    .checkExport();
187 
188         // third update with delivered content (without changing the code)
189         _mqlProgram.updateWithCode(exportParser.getOrigCode())
190                    .checkExport();
191 
192         Assert.assertEquals(exportParser.getOrigCode(),
193                             _mqlProgram.export().getOrigCode(),
194                             "check that separators correct removed");
195     }

210         Assert.assertTrue(xml.indexOf("<code><![CDATA[<]Inserted_by_ENOVIA]Inserted_by_ENOVIA>]]></code>") >= 0,
211                           "check translation of the CDATA conversion");
212     }
