57      /***
58       * {@inheritDoc}
59       * The source code is set to an empty class.
60       */
61      @Override()
62      protected JPOProgramData createCleanNewData(final JPOProgramData _original)
63      {
64          final String prgName = _original.getName();
65          final StringBuilder code = new StringBuilder();
66          if (prgName.indexOf('.') > 0)  {
67              code.append("package ").append(prgName.replaceAll("//.[^.]*$", "")).append(";\n");
68          }
69          code.append("public class ").append(prgName.replaceAll(".*//.", "")).append("_mxJPO {}");
70  
71          return this.createNewData(_original.getName().substring(AbstractTest.PREFIX.length())).setCode(code.toString());
72      }

133     /***
134      * Tests, if the JPO within MX is created and registered with the correct
135      * symbolic name.
136      *
137      * @param _description  description of the test data
138      * @param _jpoProgram   data collection to test
139      * @throws Exception if test failed
140      */
141     @Test(dataProvider = "data",
142           description = "test update of non existing JPOs")
143     public void simpleUpdate(final String _description,
144                              final JPOProgramData _jpoProgram)
145         throws Exception
146     {
147         // first update with original content
148         _jpoProgram.createDependings()
149                    .update();
150 
151         // the replace code (removing TCL update commands)
152         final StringBuilder cmd = new StringBuilder()
153                 .append("escape mod program \"").append(AbstractTest.convertMql(_jpoProgram.getName()))
154                 .append("\" code \"").append(AbstractTest.convertMql(_jpoProgram.getCode())).append("\"");
155         this.mql(cmd);
156 
157         // export
158         final ExportParser exportParser = _jpoProgram.export();
159         _jpoProgram.checkExport(exportParser);
160 
161         // set creator as user, 'test' as description, not hidden
162         this.mql("escape mod program \"" + AbstractTest.convertMql(_jpoProgram.getName())
163                 + "\" execute user creator execute immediate !needsbusinessobject !downloadable !pipe !pooled description test !hidden");
164 
165         // second update with delivered content
166         _jpoProgram.updateWithCode(exportParser.getOrigCode())
167                    .checkExport();
168 
169         // third update with delivered content (without changing the code)
170         _jpoProgram.updateWithCode(exportParser.getOrigCode())
171                    .checkExport();
172 
173         Assert.assertEquals(exportParser.getOrigCode(),
174                             _jpoProgram.export().getOrigCode(),
175                             "check that separators correct removed");
176     }

178     /***
179      * Check that the end tag of CDATA is correct translated to
180      * 'Inserted_by_ENOVIA'.
181      *
182      * @throws Exception if test failed
183      */
184     @Test(description = "check that the end tag of CDATA is correct translated to 'Inserted_by_ENOVIA'")
185     public void checkCDataTranslation()
186         throws Exception
187     {
188         final String name = AbstractTest.PREFIX + "_Test";
189         this.mql("add prog " + name + " java code '<]]>'");
190         final String xml = this.mql("export prog " + name + " xml");
191         Assert.assertTrue(xml.indexOf("<code><![CDATA[<]Inserted_by_ENOVIA]Inserted_by_ENOVIA>]]></code>") >= 0,
192                           "check translation of the CDATA conversion");
193     }
