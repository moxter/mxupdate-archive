252     /***
253      * Tests an update of non existing person. The result is tested with by
254      * exporting the person and checking the result.
255      *
256      * @param _description  description of the test case
257      * @param _person       person to test
258      * @throws Exception if test failed
259      */
260     @Test(dataProvider = "persons", description = "test update of non existing person")
261     public void simpleUpdate(final String _description,
262                              final PersonAdminData _person)
263         throws Exception
264     {
265         // create referenced property value
266         for (final PropertyDef prop : _person.getProperties())  {
267             if (prop.getTo() != null)  {
268                 prop.getTo().create();
269             }
270         }
271         // create site
272         if (_person.getSite() != null)  {
273             _person.getSite().create();
274         }
275         // create assigned roles
276         for (final RoleData role : _person.getRoles())  {
277             role.create();
278         }
279         // create assigned groups
280         for (final GroupData group : _person.getGroups())  {
281             group.create();
282         }
283         // create cue properties
284         for (final CueData<PersonAdminData> cue : _person.getCues())  {
285             for (final PropertyDef prop : cue.getProperties())  {
286                 if (prop.getTo() != null)  {
287                     prop.getTo().create();
288                 }
289             }
290         }
291         // create filter properties
292         for (final FilterData<PersonAdminData> filter : _person.getFilters())  {
293             for (final PropertyDef prop : filter.getProperties())  {
294                 if (prop.getTo() != null)  {
295                     prop.getTo().create();
296                 }
297             }
298         }
299         // create query properties
300         for (final QueryData<PersonAdminData> query : _person.getQueries())  {
301             for (final PropertyDef prop : query.getProperties())  {
302                 if (prop.getTo() != null)  {
303                     prop.getTo().create();
304                 }
305             }
306         }
307         // create table properties
308         for (final TableData<PersonAdminData> table : _person.getTables())  {
309             for (final PropertyDef prop : table.getProperties())  {
310                 if (prop.getTo() != null)  {
311                     prop.getTo().create();
312                 }
313             }
314         }
315         // create tip properties
316         for (final TipData<PersonAdminData> tip : _person.getTips())  {
317             for (final PropertyDef prop : tip.getProperties())  {
318                 if (prop.getTo() != null)  {
319                     prop.getTo().create();
320                 }
321             }
322         }
323         // create tool set properties and programs
324         for (final ToolSetData<PersonAdminData> toolSet : _person.getToolSets())  {
325             for (final PropertyDef prop : toolSet.getProperties())  {
326                 if (prop.getTo() != null)  {
327                     prop.getTo().create();
328                 }
329             }
330             for (final AbstractProgramData<?> prog : toolSet.getPrograms())  {
331                 prog.create();
332             }
333         }
334         // create view properties
335         for (final ViewData<PersonAdminData> view : _person.getViews())  {
336             for (final PropertyDef prop : view.getProperties())  {
337                 if (prop.getTo() != null)  {
338                     prop.getTo().create();
339                 }
340             }
341         }
342 
343         // first update with original content
344         this.update(_person.getCIFileName(), _person.ciFile());
345         final ExportParser exportParser = _person.export();
346         _person.checkExport(exportParser);
347 
348         // second update with delivered content
349         this.update(_person.getCIFileName(), exportParser.getOrigCode());
350         _person.checkExport(_person.export());
351     }
