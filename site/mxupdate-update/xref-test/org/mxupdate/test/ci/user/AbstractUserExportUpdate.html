42  /***
43   * Prepares test data for users (group, role and person).
44   *
45   * @author The MxUpdate Team
46   * @version $Id: AbstractUserExportUpdate.java 580 2009-10-17 18:58:51Z tim.moxter $
47   * @param <USER> user class which is tested
48   */
49  public abstract class AbstractUserExportUpdate<USER extends AbstractUserData<USER>>
50      extends AbstractDataExportUpdate<USER>

52      /***
53       * Prepares the test data.
54       *
55       * @param _logText  text used for the description (logging purpose)
56       * @param _datas    specific test data to append
57       * @return prepared test data
58       */
59      @Override()
60      protected Object[][] prepareData(final String _logText,
61                                       final Object[]... _datas)
62      {
63          final List<Object[]> ret = new ArrayList<Object[]>();
64          ret.addAll(Arrays.asList(_datas));
65  
66          this.appendCue(_logText, ret);
67          this.appendFilter(_logText, ret);
68          this.appendQuery(_logText, ret);
69          this.appendTable(_logText, ret);
70          this.appendTip(_logText, ret);
71          this.appendToolSet(_logText, ret);
72          this.appendView(_logText, ret);
73  
74          return super.prepareData(_logText, ret.toArray(new Object[ret.size()][]));
75      }

743     /***
744      * Prepares the test data for views.
745      *
746      * @param _name                     name of the user data
747      * @param _viewName                 name of the view
748      * @param _activeCueNames           names of active cues
749      * @param _notActiveCueNames        name of inactive cues
750      * @param _activeFilterNames        name of active filters
751      * @param _notActiveFilterNames     name of inactive filters
752      * @param _activeTipNames           name of active tips
753      * @param _notActiveTipNames        name of inactive tips
754      * @param _activeToolSetNames       name of active tool sets
755      * @param _notActiveToolSetNames    name of inactive tool sets
756      * @param _activeTableNames         name of active tables
757      * @param _notActiveTableNames      name of inactive tables
758      * @return user test data used for views
759      */
760     private USER prepareViewTestData(final String _name,
761                                      final String _viewName,
762                                      final String[] _activeCueNames,
763                                      final String[] _notActiveCueNames,
764                                      final String[] _activeFilterNames,
765                                      final String[] _notActiveFilterNames,
766                                      final String[] _activeTipNames,
767                                      final String[] _notActiveTipNames,
768                                      final String[] _activeToolSetNames,
769                                      final String[] _notActiveToolSetNames,
770                                      final String[] _activeTableNames,
771                                      final String[] _notActiveTableNames)
772     {
773         final USER userData = this.createNewData(_name);
774         final ViewData<USER> view = userData.newView(_viewName);
775         if (_activeCueNames != null)  {
776             for (final String name : _activeCueNames)  {
777                 view.addActive(userData.newCue(name));
778             }
779         }
780         if (_notActiveCueNames != null)  {
781             for (final String name : _notActiveCueNames)  {
782                 view.addNotActive(userData.newCue(name));
783             }
784         }
785         if (_activeFilterNames != null)  {
786             for (final String name : _activeFilterNames)  {
787                 view.addActive(userData.newFilter(name));
788             }
789         }
790         if (_notActiveFilterNames != null)  {
791             for (final String name : _notActiveFilterNames)  {
792                 view.addNotActive(userData.newFilter(name));
793             }
794         }
795         if (_activeTipNames != null)  {
796             for (final String name : _activeTipNames)  {
797                 view.addActive(userData.newTip(name));
798             }
799         }
800         if (_notActiveTipNames != null)  {
801             for (final String name : _notActiveTipNames)  {
802                 view.addNotActive(userData.newTip(name));
803             }
804         }
805         if (_activeToolSetNames != null)  {
806             for (final String name : _activeToolSetNames)  {
807                 view.addActive(userData.newToolSet(name));
808             }
809         }
810         if (_notActiveToolSetNames != null)  {
811             for (final String name : _notActiveToolSetNames)  {
812                 view.addNotActive(userData.newToolSet(name));
813             }
814         }
815         if (_activeTableNames != null)  {
816             for (final String name : _activeTableNames)  {
817                 view.addActive(userData.newTable(name));
818             }
819         }
820         if (_notActiveTableNames != null)  {
821             for (final String name : _notActiveTableNames)  {
822                 view.addNotActive(userData.newTable(name));
823             }
824         }
825         return userData;
826     }
