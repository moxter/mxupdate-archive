24  import java.util.Arrays;
25  import java.util.List;

27  import org.mxupdate.test.AbstractDataExportUpdate;
28  import org.mxupdate.test.data.AbstractAdminData;

30  /***
31   * Abstract user interface test class.
32   *
33   * @author The MxUpdate Team
34   * @version $Id: AbstractUITest.java 738 2010-02-21 16:33:51Z tim.moxter $
35   * @param <DATA> class of the data
36   */
37  abstract class AbstractUITest<DATA extends AbstractAdminData<?>>
38      extends AbstractDataExportUpdate<DATA>
39  {
40      /***
41       * Prepares the test data.
42       *
43       * @param _logText  text used for the description (logging purpose)
44       * @param _datas    specific test data to append
45       * @return prepared test data
46       */
47      @Override()
48      protected Object[][] prepareData(final String _logText,
49                                       final Object[]... _datas)
50      {
51          final List<Object[]> ret = new ArrayList<Object[]>();
52          ret.add(new Object[]{
53                  _logText + " without anything (to test required fields)",
54                  this.createNewData("hello \" test")});
55          ret.add(new Object[]{
56                  _logText + " with other symbolic name",
57                  this.createNewData("hello \" test").setSymbolicName(_logText + "_Test")});
58          ret.add(new Object[]{
59                  _logText + " with description",
60                  this.createNewData("hello \" test").setValue("description", "complex description \"test\"")});
61          ret.add(new Object[]{
62                  _logText + " with default hidden flag",
63                  this.createNewData("hello \" test").setHidden(null)});
64          ret.add(new Object[]{
65                  _logText + " with hidden flag false",
66                  this.createNewData("hello \" test").setHidden(false)});
67          ret.add(new Object[]{
68                  _logText + " with hidden flag true",
69                  this.createNewData("hello \" test").setHidden(true)});
70  
71          ret.addAll(Arrays.asList(_datas));
72          return super.prepareData(_logText, ret.toArray(new Object[ret.size()][]));
73      }
74  }
