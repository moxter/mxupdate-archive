397     /***
398      * Returns depending on the <code>_key</code> related value within the
399      * properties.
400      *
401      * @param _key      searched key within property settings
402      * @return value of the property
403      * @throws Exception if setting could not be fetched
404      * @see #settings
405      */
406     private String getSetting(final String _key)
407         throws Exception
408     {
409         if (this.settings == null)  {
410             final DefaultMavenSettingsBuilder builder = new DefaultMavenSettingsBuilder();
411             final File settingsFile = new File(new File(System.getProperty("user.home")), ".m2/settings.xml");
412             this.settings = builder.buildSettings(settingsFile);
413         }
414         String tmp = null;
415         for (final String active : this.settings.getActiveProfiles())  {
416             tmp = ((Profile) this.settings.getProfilesAsMap().get(active)).getProperties().getProperty(_key);
417             if (tmp != null)  {
418                 break;
419             }
420         }
421         final String ret;
422         if (tmp != null)  {
423             ret = tmp;
424         } else  {
425             ret = "";
426         }
427         return ret;
428     }

499      * @return set of found values

502     public Set<String> mqlAsSet(final CharSequence _cmd)

504     {
505         final String bck = this.mql(_cmd);
506         final Set<String> ret;
507         if ("".equals(bck))  {
508             ret = new HashSet<String>(0);

510             ret = new HashSet<String>(Arrays.asList(bck.split("\n")));
511         }
512         return ret;

516      * Export given configuration item <code>_type</code> with
517      * <code>_name</code>. The returned values from the export are checked for:
518      * <ul>
519      * <li>value is returned</li>
520      * <li>the returned export value includes exact one export for given

522      * <li>the name of the exported object is equal <code>_name</code></li>
523      * </ul>
524      *
525      * @param _type     type to export
526      * @param _name     name to export
527      * @return map with the exported object
528      * @throws Exception  if MQL calls failed
529      */
530     protected Export export(final CI _type,
531                             final String _name)
532         throws Exception
533     {
534         final Map<?,?> bck = this.executeEncoded("Export",
535                                                  null,
536                                                  "TypeDef", _type.updateType,
537                                                  "Name", _name);
538         // extract values
539         final Map<?,?> values = (Map<?,?>) bck.get("values");
540 
541         Assert.assertNotNull(bck);
542         Assert.assertEquals((String) values.get("TypeDef"),
543                             _type.updateType,
544                             "check correct type definition");
545 
546         final Export ret =  new Export(values);
547 
548         Assert.assertEquals(ret.getName(), _name, "returned name is equal to given name");
549 
550         return ret;
551     }

570     /***
571      * Cleanups given configuration <code>_type</code>. This means that all
572      * existing administration object which starts with {@link #PREFIX} are
573      * deleted. Only if {@link CI#wildcardSearch} is <i>true</i>, a search via
574      * wild card is done; otherwise a list of all administration object is
575      * used.
576      *
577      * @param _type     type of configuration item
578      * @throws MatrixException if cleanup for given configuration item
579      *                         <code>_type</code> failed
580      */
581     protected void cleanup(final CI _type)
582         throws MatrixException
583     {
584         // administration objects
585         if (_type.getMxType() != null)  {
586             final Set<String> elements;
587             if (_type == AbstractTest.CI.UI_TABLE)  {
588                 elements = this.mqlAsSet("escape list table system");
589                 for (final String element : elements)  {
590                     if (element.startsWith(AbstractTest.PREFIX))  {
591                         this.mql("escape delete " + _type.mxType + " \"" + AbstractTest.convertMql(element) + "\" system");
592                     }
593                 }
594             } else  {
595                 if (_type.wildcardSearch)  {
596                     elements = this.mqlAsSet("escape list " + _type.mxType + " \"" + AbstractTest.PREFIX + "*\"");
597                 } else  {
598                     elements = this.mqlAsSet("escape list " + _type.mxType);
599                 }
600                 if (_type == AbstractTest.CI.DM_INTERFACE)  {
601                     for (final String element : elements)  {
602                         this.mql("escape mod " + _type.mxType + " \"" + AbstractTest.convertMql(element) + "\" remove derived");
603                     }
604                 }
605                 for (final String element : elements)  {
606                     if (element.startsWith(AbstractTest.PREFIX))  {
607                         this.mql("escape delete " + _type.mxType + " \"" + AbstractTest.convertMql(element) + "\"");
608                     }
609                 }
610             }
611         }
612         // business objects
613         if (_type.getBusType() != null)  {
614             final Set<String> busIds = this.mqlAsSet("escape temp query bus \""
615                     + AbstractTest.convertMql(_type.getBusType())
616                     + "\" \"" + AbstractTest.PREFIX + "*\" * select id dump '\t' recordseparator '\n'");
617             for (final String busId : busIds)  {
618                 this.mql("delete bus " + busId.split("\t")[3]);
619             }
620         }
621     }

623     /***
624      * Calls given <code>_method</code> in <code>_jpo</code>.
625      *
626      * @param <T>           class which is returned
627      * @param _jpo          name of JPO to call
628      * @param _method       method of the called <code>_jpo</code>
629      * @param _parameters   list of all parameters for the <code>_jpo</code>
630      *                      which are automatically encoded encoded
631      * @return returned value from the called <code>_jpo</code>
632      * @throws IOException      if the parameter could not be encoded
633      * @throws MatrixException  if the called <code>_jpo</code> throws an
634      *                          exception
635      * @see JPOReturn
636      */
637     public <T> JPOReturn<T> jpoInvoke(final String _jpo,
638                                       final String _method,
639                                       final Object... _parameters)
640         throws IOException, MatrixException
641     {
642         // encode parameters
643         final String[] paramStrings = new String[_parameters.length];
644         int idx = 0;
645         for (final Object parameter : _parameters)  {
646             final ByteArrayOutputStream out = new ByteArrayOutputStream();
647             final ObjectOutputStream oos = new ObjectOutputStream(out);
648             oos.writeObject(parameter);
649             oos.close();
650             paramStrings[idx++] = new String(Base64.encodeBase64(out.toByteArray()));
651         }
652 
653         return new JPOReturn<T>(JPO.invoke(this.context,
654                                            _jpo,
655                                            null,
656                                            _method,
657                                            paramStrings,
658                                            Object.class));
659     }

662     /***
663      * Calls given <code>_method</code> in of the MxUpdate eclipse plug-in
664      * dispatcher. The MX context {@link #mxContext} is connected to the
665      * database if not already done.
666      *
667      * @param _method       method of the called <code>_jpo</code>
668      * @param _params       extra parameters
669      * @param _arguments    list of all parameters for the <code>_jpo</code>
670      *                      which are automatically encoded encoded
671      * @return returned value from the called <code>_jpo</code>
672      * @throws IOException      if the parameter could not be encoded
673      * @throws MatrixException  if the called <code>_jpo</code> throws an
674      *                          exception
675      * @throws ClassNotFoundException if the class which is decoded from the
676      *                          returned string value could not be found
677      * @see #mxContext
678      * @see #connect()
679      */
680     public Map<?,?> executeEncoded(final String _method,
681                                       final Map<String,String> _params,
682                                       final Object... _arguments)
683         throws IOException, MatrixException, ClassNotFoundException
684     {
685         // prepare arguments in a map
686         final Map<String,Object> arguments;
687         if ((_arguments == null) || (_arguments.length == 0))  {
688             arguments = null;
689         } else  {
690             arguments = new HashMap<String,Object>();
691             for (int idx = 0; idx < _arguments.length; )  {
692                 arguments.put((String) _arguments[idx++], _arguments[idx++]);
693             }
694         }
695 
696         // prepare MQL statement with encoded parameters
697         final StringBuilder cmd = new StringBuilder()
698             .append("exec prog ").append("org.mxupdate.plugin.Dispatcher \"")
699             .append(this.encode(_params)).append("\" \"")
700             .append(this.encode(_method)).append("\" \"")

748     protected final <T> T decode(final String _arg)
