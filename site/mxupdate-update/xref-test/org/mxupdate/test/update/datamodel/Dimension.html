107     /***
108      * Makes an update for given <code>_code</code>.
109      *
110      * @param _fileName     name of the file to update
111      * @param _code         TCL update code
112      * @param _params       parameters
113      * @return values from the called dispatcher
114      * @throws Exception  if update failed
115      */
116     @Deprecated()
117     public Map<?,?> update(final String _fileName,
118                            final String _code,
119                            final String... _params)
120         throws Exception
121     {
122         final Map<String,String> files = new HashMap<String,String>();
123         files.put(_fileName, _code);
124         final Map<String,String> params = new HashMap<String,String>();
125         if (_params != null)  {
126             for (int idx = 0; idx < _params.length; idx += 2)  {
127                 params.put(_params[idx], _params[idx + 1]);
128             }
129         }
130         final Map<?,?> bck = this.executeEncoded("Update", params, "FileContents", files);
131 
132         return bck;
133     }

451     /***
452      * Checks that the new created or updated dimension {@link #DIMENSION_NAME}
453      * is equal to the defined update code {@link #DIMENSION_CODE}.
454      *
455      * @throws MatrixException if MQL select statements failed
456      */
457     private void checkAll()
458         throws MatrixException
459     {
460         Assert.assertTrue(!"".equals(this.mql("list dimension " + Dimension.DIMENSION_NAME)),
461                           "dimension was not created!");
462         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
463                                     + " select description dump"),
464                             "a simple description",
465                             "Check that description is set");
466         Assert.assertEquals(this.mql("list property to dimension " + Dimension.DIMENSION_NAME),
467                             "dimension_" + Dimension.DIMENSION_NAME
468                                     + " on program eServiceSchemaVariableMapping.tcl to dimension "
469                                     + Dimension.DIMENSION_NAME,
470                             "Check for symbolic name registration.");
471 
472         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
473                                             + " select unit dump @")
474                                     .replaceAll("name1", "")
475                                     .replaceAll("name2", "")
476                                     .replaceAll("name 3", ""),
477                             "@@",
478                             "check that exact three units are defined");
479 
480         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
481                                     + " select unit[name1] dump"),
482                             "TRUE",
483                             "check that unit 'name1' exists");
484         this.checkUnit("name1", "default", "TRUE");
485         this.checkUnit("name1", "label", "label1");
486         this.checkUnit("name1", "description", "description 1");
487         this.checkUnit("name1", "multiplier", "1.0");
488         this.checkUnit("name1", "offset", "0.0");
489         this.checkUnit("name1", "setting[SettingKey2]", "TRUE");
490         this.checkUnit("name1", "setting[SettingKey2].value", "SettingValue2");
491         this.checkUnit("name1", "setting[to]", "TRUE");
492         this.checkUnit("name1", "setting[to].value", "SettingValue");
493         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
494                                             + " select unit[name1].setting dump @")
495                                     .replaceAll("to", "")
496                                     .replaceAll("SettingKey2", ""),
497                             "@",
498                             "Check that only two settings are defined");
499         this.checkUnit("name1", "property[Zest].value", "Zest");
500         this.checkUnit("name1", "property[Zest].to", "");
501         this.checkUnit("name1", "property[_test3].value", "Test3");
502         this.checkUnit("name1", "property[_test3].to", "");
503         this.checkUnit("name1", "property[abc].value", "abc");
504         this.checkUnit("name1", "property[abc].to", "");
505         this.checkUnit("name1", "property[abc2].value", "abc2");
506         this.checkUnit("name1", "property[abc2].to", "");
507         this.checkUnit("name1", "property[test2].value", "Test2");
508         this.checkUnit("name1", "property[test2].to", "");
509         this.checkUnit("name1", "property[test].value", "Test");
510         this.checkUnit("name1", "property[test].to", "type Part");
511         this.checkUnit("name1", "systemunit", "");
512 
513         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
514                                     + " select unit[name2] dump"),
515                             "TRUE",
516                             "Check that unit 'name2' exists");
517         this.checkUnit("name2", "default", "FALSE");
518         this.checkUnit("name2", "label", "label2");
519         this.checkUnit("name2", "description", "description 2");
520         this.checkUnit("name2", "multiplier", "10.0");
521         this.checkUnit("name2", "offset", "20.0");
522         this.checkUnit("name2", "property", "");
523         this.checkUnit("name2", "systemunit", "name2");
524         this.checkUnit("name2", "systemunit[Duration Units]", "name2");
525 
526         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
527                                     + " select unit[name 3] dump"),
528                             "TRUE",
529                             "Check that unit 'name 3' exists");
530         this.checkUnit("name 3", "default", "FALSE");
531         this.checkUnit("name 3", "label", "label 3");
532         this.checkUnit("name 3", "description", "description3");
533         this.checkUnit("name 3", "multiplier", "30.0");
534         this.checkUnit("name 3", "offset", "40.0");
535         this.checkUnit("name 3", "property", "");
536         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
537                                             + " select 'unit[name 3].systemunit[]' dump @")
538                                     .replaceAll("name2", "")
539                                     .replaceAll("name 3", ""),
540                             "@",
541                             "Check that exact two system infos are defined");
542         this.checkUnit("name 3", "systemunit[Duration Units]", "name2");
543         this.checkUnit("name 3", "systemunit[Test]", "name 3");
