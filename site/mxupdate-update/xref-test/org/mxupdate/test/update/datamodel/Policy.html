173     /***
174      * Checks that states where no symbolic name is defined in the TCL update
175      * does not define a symbolic name (with empty name).
176      *
177      * @throws Exception if MQL execution failed
178      * @see http://code.google.com/p/mxupdate/issues/detail?id=29
179      */
180     public void testProperties()
181         throws Exception
182     {
183         final Map<String,String> params = new HashMap<String,String>();
184         params.put("POLICY_" + Policy.POLICY_NAME + ".tcl",
185                    Policy.POLICY_UPDATE_CODE);
186         this.jpoInvoke("org.mxupdate.plugin.Update", "updateByContent", params);
187 
188         Assert.assertTrue(!"".equals(this.mql("list policy MxUpdate_Test")),
189                           "policy was not created!");
190 
191         // check that only 11 properties are defined....
192         // 4 state properties + installer + installed date + original name +
193         // file date + author + version + application
194         final Set<String> propNames = new HashSet<String>();
195         final String propNamesStr = this.mql("print policy MxUpdate_Test select property.name dump '\n'");
196         for (final String propName : propNamesStr.split("\n"))  {
197             propNames.add(propName);
198         }
199         Assert.assertTrue(propNames.size() == 11, "check that all properties are defined");
200         Assert.assertTrue(this.mql("print policy MxUpdate_Test").indexOf("property  value ") < 0,
201                           "Update did set an empty property!");
202     }

204     /***
205      * Test for for an update for existing policy with empty property and state
206      * names without symbolic names. The empty property must be removed after
207      * the update and for the state without symbolic name no registration must
208      * be done.
209      *
210      * @throws Exception if MQL execution failed
211      * @see http://code.google.com/p/mxupdate/issues/detail?id=28
212      * @see http://code.google.com/p/mxupdate/issues/detail?id=29
213      */
214     public void testNullProperties()
215         throws Exception
216     {
217         this.mql("add policy MxUpdate_Test state Pending property \"\" value Test");
218 
219         final Map<String,String> params = new HashMap<String,String>();
220         params.put("POLICY_" + Policy.POLICY_NAME + ".tcl",
221                    Policy.POLICY_UPDATE_CODE);
222         this.jpoInvoke("org.mxupdate.plugin.Update", "updateByContent", params);
223 
224         // check that only 11 properties are defined....
225         // 4 state properties + installer + installed date + original name +
226         // file date + author + version + application
227         final Set<String> propNames = new HashSet<String>();
228         final String propNamesStr = this.mql("print policy MxUpdate_Test select property.name dump '\n'");
229         for (final String propName : propNamesStr.split("\n"))  {
230             propNames.add(propName);
231         }
232         Assert.assertTrue(!propNames.contains(""), "Update did not remove empty property!");
233         Assert.assertTrue(this.mql("print policy MxUpdate_Test").indexOf("property  value Test") < 0,
234                           "Update did not remove empty property!");
235         Assert.assertTrue(propNames.size() == 11, "check that all properties are defined");
236     }

238     /***
239      * Tests that multiple symbolic names for a single state could be defined.
240      *
241      * @throws Exception if test failed
242      */
243     public void testMultipleStateSymbolicNames()
244         throws Exception
245     {
246         final Map<String,String> params = new HashMap<String,String>();
247         params.put("POLICY_" + Policy.POLICY_NAME + ".tcl",
248                    "updatePolicy \"${NAME}\" {"
249                     + "  state \"Pending\"  {"
250                     + "    registeredName \"state_Exists\""
251                     + "    registeredName \"state_Pending\""
252                     + "  }"
253                     + "}");
254         this.jpoInvoke("org.mxupdate.plugin.Update", "updateByContent", params);
255 
256         Assert.assertEquals(this.mql("print policy " + Policy.POLICY_NAME
257                                 + " select property[state_Exists].value dump"),
258                             "Pending",
259                             "test that symbolic name 'state_Exists' is correct registered");
260         Assert.assertEquals(this.mql("print policy " + Policy.POLICY_NAME
261                                 + " select property[state_Pending].value dump"),
262                             "Pending",
263                             "test that symbolic name 'state_Pending' is correct registered");
264     }
